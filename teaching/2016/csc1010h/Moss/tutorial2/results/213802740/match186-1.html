def main():
    cardholder_name = input("Enter the card holder's name:\n")
    card_number = input("Enter the 16 digit card number (no spaces please):\n")
    issue_date = input("Enter the date of issue (mm:yy):\n")
    expiry_date = input("Enter the expiry date (mm:yy):\n")
    security_code = int(input("Enter the 3 digit security code:\n"))
    credit_limit = float(input("Enter the credit limit (Rand):\n"))
    balance = float(input("Enter the balance (Rand):\n"))
    available = str("{0:.2f}".format(credit_limit-balance)+" "+"Rand")
    credit_limit2 = str("{0:.2f}".format(credit_limit)+" " +"Rand")
    issue_date2 = str(issue_date[0:2]+"/"+str(issue_date[3:]))
    balance2 = str("{0:.2f}".format(balance)+" " +"Rand")
    card_number2 = str(card_number[0:4])+" "+str(card_number[4:8])+" "+str(card_number[8:12])+" "+str(card_number[12:17])
    expiry_date2 = str(expiry_date[0:2])+"/"+str(expiry_date[3:])
   
    
    print("+------------- Credit Card Account --------------+")
    print("| Card holder:   {0:<32}".format(cardholder_name.title())+"|")
    print("| Card number:   {0:<32}".format(card_number2)+"|")
    print("| Date of issue: {0:<9}".format(issue_date2)+"Expiry date: {0:<10}".format(expiry_date2)+"|")
    print("| Security code: {0:<32}".format(security_code)+"|")
    print("| Credit limit:  {0:<32}".format(credit_limit2)+"|")
    print("| Available:     {0:<32}".format(available)+"|")
    print("| Balance:       {0:<32}".format(balance2)+"|")
    print("+------------------------------------------------+")
main()
