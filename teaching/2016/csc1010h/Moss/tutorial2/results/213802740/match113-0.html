#The program produces a credit card and the information on it
def main():
    name=input("Enter the card holder's name:\n")       #prompt the user to enter name


    card_number=input('Enter the 16 digit card number (no spaces please):\n')   #ask the user for the number
    date_of_issue=input('Enter the date of issue (mm:yy):\n')  # prompt the user to enter date of issue
    expiry_date=input('Enter the expiry date (mm:yy):\n')   # prompt the user to enter the expiry date
    security_code=int(input('Enter the 3 digit security code:\n'))   #ask the user to enter the security code
    credit_limit=int(input('Enter the credit limit (Rand):\n'))    #ask the user to enter the credit limit
    balance=int(input('Enter the balance (Rand):\n'))    #prompt the user to enter the balance
    available=credit_limit-balance  # the program calculates the available amount
    #program prepares round the amounts entered into rands and cents
    credit_limit="{0:.2f}".format(credit_limit)
    balance='{0:.2f}'.format(balance)
    available='{0:.2f}'.format(available)
    
    #The program prints the results
    word=' Credit Card Account '
    print('+'+'{0:-^48}'.format(word)+'+')
    print('|', 'Card holder:   ''{:<32}'.format(name)+'|')
    print('|', 'Card number:   '+str(card_number)[0:4],str(card_number)[4:8],str(card_number)[8:12],'{:<17}'.format(str(card_number)[12:16])+'|')
    print('|', 'Date of issue: '+str(date_of_issue).replace(':','/'),'  ','Expiry date:','{:<10}'.format(expiry_date.replace(':','/'))+'|')
    print('|', 'Security code: '+'{:<32}'.format(security_code)+'|')
    print('|', 'Credit limit:  '+'{:<32}'.format(credit_limit+' Rand')+'|')
    print('|', 'Available:     '+'{:<32}'.format(available+' Rand')+'|')
    print('|', 'Balance:       '+'{:<32}'.format(balance+' Rand')+'|')
    sign='-'
    print('+'+'{0:-^48}'.format(sign)+'+')
    
main()

    side=int(input('Enter the length of the side (cm):\n'))
