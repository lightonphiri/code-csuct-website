import math

def main():
    # prompt user for input
    side = eval(input('Enter the length of the side (cm):\n'))
    
    # Calculating the area using given side
    Area = (3*math.sqrt(3)/2)*side**2
    
    # printing the area
    print('The area of a hexagon of side {0} is {1:2.3f}.'.format(side,Area))
