name_of_card_holder = input("Enter the card holder's name: \n")

DOI = date_of_issue.replace(":","/")
expiry_date = input("Enter the expiry date (mm:yy): \n")
DOE = expiry_date.replace(":","/")
security_code = input("Enter the 3 digit security code: \n")
limit = float(input("Enter the credit limit (Rand): \n"))
limitee = "{0:.2f}".format(limit)
limiter = str(limitee)+" Rand"
balance = float(input("Enter the balance (Rand):\n"))
balancee = "{0:.2f}".format(balance)
balancer = str(balancee)+" Rand"
available = limit-balance
availablee = "{0:.2f}".format(available)
availabler = str(availablee)+" Rand"
last= "-"

print ("+{0:-^48}+".format(credit))
print ("| Card holder:   {0:<32}|".format(name_of_card_holder))
print ("| Card number:   {0:<5}{1:<5}{2:<5}{3:<17}|".format(number[0:4],number[4:8],number[8:12],number[12:16]))
print ("| Date of issue: {0:<9}Expiry date: {1:<10}|".format(DOI,DOE))

print ("| Credit limit:  {0:<32}|".format(limiter))

print ("| Balance:       {0:<32}|".format(balancer))
