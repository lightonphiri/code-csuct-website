def main():
    print('Hello, I am Eugene Junior.')


    first_name = input('What is your first name?\n')
    last_name = input('What is your last name?\n')
    print('Hi',first_name[0:1]+'.'+last_name[0:1]+'.!')
    present_year = int(input('What year is this?\n'))
    age = int(input('What age are you?\n'))
    a = present_year-age-1                                  # use variablea and b to find possible years of birth
    b = a+1
    print('Okay, so you  were born in '+str(a)+'/'+str(b)[2:4]+'.')
    height = float(input('How tall are you (metres)?\n'))
    its_feet = str(round(height//0.304799995))
    its_inches = str(round((height%0.304799995)/0.02540003))
    print("That's",its_feet+"'"+its_inches+'"!')


    print("I'm going to tell you your Star Wars name.")
    first_name_of_mother = input("What's your mother's first name?\n")
    city_name = input ("What's the name of the city nearest to your place of birth?\n")
    name = last_name[0:3]+first_name[0:2]
    surname = first_name_of_mother[-2:-1:1]+first_name_of_mother[-1]+city_name[-3:-1:1]+city_name[-1]
    print('Your Star Wars name is',name.title()+' '+surname.title()+'.')
main()
