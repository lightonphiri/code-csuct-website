    left(90)
    
    # move the turtle forward
    forward(100)

    right(135)
    
    # moove the turtle forward
    forward(200)
    
    # change the pen colour
    pencolor('magenta')
    
    # lift up the pen
    penup()
    
    # go back to the initial position
    home()
    
    # turn the turtle 90 degrees to the left
    left(90)
    
    # move the turtle forward
    forward(distance)
    
    # put the pen down
    pendown()
    
    # turn the turtle 90 degrees to the right
    right(90)
    
    # move the turtle foward
    forward(200)
    
    # lift the pen up
    penup()

    left(90)
    
    # move the turtle forward
    forward(dimension/2)
    
    # put the pen down
    pendown()

    back(dimension)
    
    # exit the turtle window
    exitonclick()
