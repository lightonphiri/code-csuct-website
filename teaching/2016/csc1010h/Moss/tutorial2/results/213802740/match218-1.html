date_of_issue=input("Enter the date of issue (mm:yy):\n") #prompt the user to enter the date the credit card was issued
expiry_date=input("Enter the expiry date (mm:yy):\n") #prompt the user to enter the expiry date of the credit card 
security_code=input("Enter the 3 digit security code:\n") #prompt the user to enter the security code
limit=float(input("Enter the credit limit (Rand):\n")) #prompt the user to enter the credit card limit
balance=float(input("Enter the balance (Rand):\n"))  #prompt the user to enter balance
available=limit-balance          #calculating the available amount

#converting the amounts to rands
limit="{0:.2f}".format(limit)
limit_in_rands=str(limit)+" Rand"
available="{:.2f}".format(available)
available_in_rands=str(available)+" Rand"
balance="{:.2f}".format(balance)
balance_in_rands=str(balance)+" Rand"
