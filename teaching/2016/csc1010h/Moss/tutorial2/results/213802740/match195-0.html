def main():
    #inputs
    #user name
    user_name = input("Enter the card holder's name: \n")      #input user name
    user_name_final = user_name.title()                        #changes first letter of user name into a capital letter
    
    #card number
    card_number = input("Enter the 16 digit card number (no spaces please): \n")      #input card number
    card_number_1 = str(card_number[0:4])                                             #separates the 1st 4 numbers of the card number
    card_number_2 = str(card_number[4:8])                                             #separates the 4th to 8th numbers of the card number
    card_number_3 = str(card_number[8:12])                                            #separates the 8th to 12th numbers of the card number
    card_number_4 = str(card_number[12:])                                             #separates the last 4 numbers of the card number
    
    #date of issue
    date_of_issue = input("Enter the date of issue (mm:yy): \n")       #input the date that the card was issued
    date_of_issue_1 = str(date_of_issue[0:2])                          #separates the first two digits of the "date of issue"
    date_of_issue_2 = str(date_of_issue[3:])                           #separates the last two digits of the "date of issue"
    date_of_issue_final = date_of_issue_1 + "/" + date_of_issue_2      #separates the first 2 and last two digits of the "date of issue" by a slash
    
    #expiry date
    expiry_date = input("Enter the expiry date (mm:yy): \n")      #input the expiry date
    expiry_date_1 = str(expiry_date[0:2])                         #separates the first two digits of the "expiry date"
    expiry_date_2 = str(expiry_date[3:])                          #separates the last two digits of the "expiry date"
    expiry_date_final = expiry_date_1 + "/" + expiry_date_2       #separates the first 2 and the last 2 digits of the "expiry date" by a slash
    
    #securty code
    security_code = input("Enter the 3 digit security code: \n")      #input the security code
    
    #credit limit
    credit_limit = input("Enter the credit limit (Rand): \n")     #input the credit limit
    credit_limit_final = float(credit_limit)                      #converts the credit limit into a float
    
    #balance
    balance = input("Enter the balance (Rand): \n")      #input the balance
    balance_final = float(balance)     #converts the balance into a float
    balance_final_string = str(balance_final)
    
    #available space in account
    available = credit_limit_final - balance_final      #finds the available space in the account by minusing the credit limit (float) by the balance (float)
    available_final = float(available)                  #converts the available space amount into a float
    available_final_string = str(available_final)
    
    #output
    print("+-------------", "Credit Card Account", "--------------+")     
    print("|", "{0:14} {1:31}".format("Card holder:", user_name_final), "|")
    print("|", "{0:14} {1} {2} {3} {4:16}".format("Card number:", card_number_1, card_number_2, card_number_3, card_number_4), "|")
    print("|", "{0} {1}    {2} {3:9}".format("Date of issue:", date_of_issue_final, "Expiry date:", expiry_date_final), "|")
    print("|", "{0} {1:31}".format("Security code:", security_code), "|")
    print("|", "{0:14} {1}{2} {3:22}".format("Credit limit:", credit_limit_final, "0", "Rand"), "|")
    print("|", "{0:14} {1:31} {2}".format("Available:", available_final_string+"0"+" "+"Rand", "|"))
    print("|", "{0:14} {1:31} {2}".format("Balance:", balance_final_string+"0"+" "+"Rand", "|"))
    print("+------------------------------------------------+")

def main():
    #input
    side = input("Enter the length of the side (cm): \n")      #input the side of the hexagon   
    side_float = float(side)                                   #converts the side of the hexagon into a float
    
    #calculations
    import math
    
    area = ((3 * (math.sqrt(3))) / 2) * (side_float ** 2)      #calculates the area of the hexagon
    area_round = round(area, 3)                                #rounds off the area to three digits
    area_round_string = str(area_round)                  #converts the rounded off area to a string
    
    #output
    print("The area of a hexagon of side", side, "is", area_round_string + ".")
