# Get input from user

def main():
    name = input("Enter the card holder's name: \n")
    card_number = input("Enter the 16 digit card number (no spaces please): \n")
    issue = input("Enter the date of issue (mm:yy): \n")
    issuestring = (issue[:2])+"/"+(issue[3:5])
    expiry = input("Enter the expiry date (mm:yy): \n")
    expirystring = (expiry[:2])+"/"+(expiry[3:5])
    security = input("Enter the 3 digit security code: \n")
    limit = float(input("Enter the credit limit (Rand): \n"))
    balance = float(input("Enter the balance (Rand): \n"))
    available = (limit-balance)
    
    print("+-------------","Credit Card Account","--------------+")
    print("| Card holder:", "{0:<34}".format(name)+"|") 


    print("| Card number:", "{0:<8}".format(card_number[:4]),"{0:<7}".format(card_number[4:8]),"{0:<8}".format(card_number[8:12]),"{0:<8}".format(card_number[12:16])+"|")
    print("| Date of issue:", "{0:<31}".format(issuestring),"|") 
    print("| Security Code:", "{0:<14}".format(security) + "Expiry date:","{0:<7}".format(expirystring)+"|", sep = "")
    print("| Credit limit:", "{0:<33}".format(str("{:0.2f}".format(limit))+  "Rand")+"|") 
    print("| Available:", "{0:<36}".format(str("{:0.2f}".format(available))+  "Rand")+"|") 
    print("| Balance:", "{0:<38}".format(str("{:0.2f}".format(balance))+  "Rand")+"|") 
    print("+------------------------------------------------+")
    
main()

import math

# Get length of side from user

length = int(input("Enter the length of the side (cm): \n"))

def main():
    area = (math.sqrt(3)*(3/2)*length**2)
    print("The area of a hexagon of side",length,"is",("{0:5.3f}".format(area))+".")

main()
