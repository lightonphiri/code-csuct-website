def main():
    
    name=input('Enter the card holder\'s name:\n')
    card_num=input('Enter the 16 digit card number (no spaces please):\n')
    date_iss=input('Enter the date of issue (mm:yy):\n')
    date_exp=input('Enter the expiry date (mm:yy):\n')
    sec_code=input('Enter the 3 digit security code:\n')
    cred_lim=float(input('Enter the credit limit (Rand):\n'))
    balance=float(input('Enter the balance (Rand):\n'))
    avail= cred_lim -balance
    num=str(card_num[0:4]+' '+card_num[4:8]+' '+card_num[8:12]+' '+card_num[12:])
    issue_date=date_iss[0:2]+'/'+date_iss[3:]
    exp_date=date_exp[0:2]+'/'+date_exp[3:]
    credit_lim=str('{0:.2f}'.format(cred_lim))+' Rand'
    available=str('{0:.2f}'.format(avail))+' Rand'
    bal=str('{0:.2f}'.format(balance))+' Rand'
    
    print('+------------- Credit Card Account --------------+')
    print('| Card holder:   {0:<32}'.format(name.title())+'|')
    print('| Card number:   {0:<32}'.format(num)+'|')
    print('| Date of issue: {0:<8}'.format(issue_date),'Expiry date: {0:<10}|'.format(exp_date))
    print('| Security code: {0:<32}|'.format(sec_code))
    print('| Credit limit:  {0:<32}'.format(credit_lim)+'|')
    print('| Available:     {0:<32}'.format(available)+'|')
    print('| Balance:       {0:<31}'.format(bal),'|')
    print('+------------------------------------------------+')
