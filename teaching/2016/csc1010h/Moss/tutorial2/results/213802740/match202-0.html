def main():
    
    import math
    #Prompt user for length of side
    length=int(input("Enter the length of the side (cm):\n"))
    
    #Calculate the area
    area=((3*math.sqrt(3))/2)*length*length
    
    #Display the area
    print("The area of a hexagon of side "+str(length)+" is "+"{0:.3f}".format(area)+".")
