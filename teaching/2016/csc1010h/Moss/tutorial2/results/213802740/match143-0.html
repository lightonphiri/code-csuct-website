# Defining a function that acts as an interactive responder
def main():
    # Obtaining information from user
    card_holder = input("Enter the card holder's name:\n")
    card_number = input("Enter the 16 digit card number (no spaces please):\n")
    issue_date = input("Enter the date of issue (mm:yy):\n")
    expiry_date = input("Enter the expiry date (mm:yy):\n")
    security_code = input("Enter the 3 digit security code:\n")
    credit_limit = float(input("Enter the credit limit (Rand):\n"))
    balance = float(input("Enter the balance (Rand):\n"))
    amount_available = credit_limit - balance
    
    # Printing results
    print("+------------- Credit Card Account --------------+")
    print("| Card holder:  ", "{0:<31}".format(card_holder),"|")
    print("| Card number:  ", "{0:<4}".format(card_number[:4]), "{0:<4}".format(card_number[4:8]), "{0:<4}".format(card_number[8:12]), "{0:<16}".format(card_number[12:16]), "|")
    print("| Date of issue: ", issue_date[:2], "/", "{0:<6}".format(issue_date[3:]), "Expiry date: ", expiry_date[:2], "/", "{0:<7}".format(expiry_date[3:]), "|", sep = "")
    print("| Security code:", "{0:<31}".format(security_code), "|")
    print("| Credit limit: ", "{0:<31}".format(str("{:0.2f}".format(credit_limit)) + " Rand"), "|")                  # Two decimal points?
    print("| Available:    ", "{0:<31}".format(str("{:0.2f}".format(amount_available)) + " Rand"), "|")              # Two decimal points?
    print("| Balance:      ", "{0:<31}".format(str("{:0.2f}".format(balance)) + " Rand"), "|")                       # Two decimal points?
    print("+------------------------------------------------+")

def main():

    print ("Hi ", first_name[0:1], ".", last_name[0:1], ".!", sep = "")
    
    # Determining year born
    year = int(input("What year is this?\n"))
    age = int(input("What age are you?\n"))
    birth_year1 = year - age - 1
    birth_year2 = year - age
    birth_year_string = str(birth_year2)    # Converting the year into a string (so that I can string slice it the next step)
    print("Okay, so you were born in ", birth_year1, '/', birth_year_string[2:], ".", sep = "")
    
    # Determining height in feet(') and inches(")
    height = float(input("How tall are you (metres)?\n"))
    height_in_feet = height // 0.3048
    height_in_inches = height / 0.3048 % 1 * 12
    print("That's ", round(height_in_feet),"'",round(height_in_inches),'"!', sep = "")
    
    # Determining Star Wars name
    print("I'm going to tell you your Star Wars name.")
    mothers_name = input("What's your mother's first name?\n")

import math

# Defining a function that calculates and print the area of a hexagon
def main():
    side_length = int(input("Enter the length of the side (cm):\n"))
    area = (((3 * math.sqrt(3))/2)*(side_length**2))
    print ("The area of a hexagon of side {0} is {1:0.3f}.".format(side_length, area))
