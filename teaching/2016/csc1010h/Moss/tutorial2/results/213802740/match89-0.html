import math
def main():
    side = eval(input("Enter the length of the side (cm):\n"))
    formula = 3*math.sqrt(3)
    area = (formula/2)*(side**2)
    print("The area of a hexagon of side", side, "is {:^.3f}".format(area)+".")
main()
