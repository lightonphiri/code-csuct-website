    side_length=int(input("Enter the length of the side (cm):\n"))
    #calculate area of hexagon
    import math
    Area=((3*math.sqrt(3))/2)*side_length**2
    print("The area of a hexagon of side {0:1} is {1:.3f}.".format(side_length,Area))
