print('| Card number:   {0:<31}'.format(card),'|')
print('| Date of issue: {0:<2}    Expiry date: {1:9}'.format(date,expiring),'|')

def main():
    length=int(input('Enter the length of the side (cm):\n'))
    area=((3*math.sqrt(3)/2)*pow(length,2))
    print('The area of a hexagon of side {0} is {1:.3f}.'.format(length,area))
