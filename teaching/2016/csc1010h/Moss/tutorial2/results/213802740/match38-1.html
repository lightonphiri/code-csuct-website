    n4=number[12:16]

    i1=issued[0:2]
    i2=issued[3:5]
    
    expired=input("Enter the expiry date (mm:yy):\n")
    e1=expired[0:2]
    e2=expired[3:5]
    
    security=input("Enter the 3 digit security code:\n")
    
    limit=eval(input("Enter the credit limit (Rand):\n"))
    
    balance=float(input("Enter the balance (Rand):\n"))
    
    available=limit-balance
    
    print("+------------- Credit Card Account --------------+")
    print("| Card holder:   {0:31} |".format(name))
    print("| Card number:   {0} {1} {2} {3:16} |".format(n1,n2,n3,n4))

    print("| Credit limit:  {0:.2f}".format(limit),"Rand", "{0:17} |".format(""))
