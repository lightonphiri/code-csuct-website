def main():
    name=input("Enter the card holder's name:\n")
    digits=input("Enter the 16 digit card number (no spaces please):\n")
    A=digits[0:4] 
    B=digits[4:8] 
    C=digits[8:12]
    D=digits[12:16]
    issue=input('Enter the date of issue (mm:yy):\n')
    s=issue.replace(':','/')
    expiry=input("Enter the expiry date (mm:yy):\n")
    p=expiry.replace(':','/')
    Three_d=input("Enter the 3 digit security code:\n")
    limit=float(input("Enter the credit limit (Rand):\n"))
    limit1='{0:.2f}'.format(limit)
    balance=float(input("Enter the balance (Rand):\n"))
    balance1='{0:.2f}'.format(balance)
    ava1=eval(limit1)-eval(balance1)
    ava='{0:.2f}'.format(ava1)
    R1=str(limit1)+' Rand'
    R2=str(balance1)+' Rand'
    R3=str(ava)+' Rand'
    print('+{0:-^48}'.format(' Credit Card Account ')+'+')
    print('| Card holder:   {0:32}'.format(name)+'|')
    print('| Card number:  ',A,B,C,D,'            |')
    print('| Date of issue: {0:8}'.format(s)+' Expiry date: {0:10}'.format(p)+'|')
    print('| Security code: {0:32}'.format(Three_d)+'|')
    print('| Credit limit:  {0:<32}|'.format(R1))
    print('| Available:     {0:<32}|'.format(R3))
    print('| Balance:       {0:<32}|'.format(R2))
    print('+{0:-^48}'.format('-')+'+')
main()
