def main():
    holder = input("Enter the card holder's name:\n")
    num = input("Enter the 16 digit card number (no spaces please):\n")
    issue = input("Enter the date of issue (mm:yy):\n")
    expiry = input("Enter the expiry date (mm:yy):\n")
    security = input("Enter the 3 digit security code:\n")
    limit = float(input("Enter the credit limit (Rand):\n"))
    balance = float(input("Enter the balance (Rand):\n"))   
    available = float(limit - balance)
    
    print("+------------- Credit Card Account --------------+")
    print("|{0}   {1:}{2:>20}".format(" Card holder:", holder, "|"))
    print("| Card number:  ", num[0:4], num[4:8], num[8:12], num[12:16]+"             |")
    print("| Date of issue: "+issue[0:2] +"/"+issue[3:5] +"    "+"Expiry date: "+expiry[0:2] +"/"+expiry[3:5] +"     |")
    print("| Security code: "+security+'                             |')
    print("| Credit limit:  {0:.2f} Rand                   |".format(limit))
    print("| Available:     {0:.2f} Rand                   |".format(available))
    print("| Balance:       {0:.2f} Rand                      |".format(balance))
    print("+------------------------------------------------+")
    
    
main()
