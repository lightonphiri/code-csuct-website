def main():
    import math
    length=int(input('Enter the length of the side (cm):\n'))
    leng=length**2
    square=3*math.sqrt(3)
    area=square/2*leng
    print('The area of a hexagon of side {0:1} is {1:2.3f}.'.format(length,area))
