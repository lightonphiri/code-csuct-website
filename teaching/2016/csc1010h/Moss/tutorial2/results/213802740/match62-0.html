def main():


    hol_name = input("Enter the card holder's name:\n")
    hol_acc = input("Enter the 16 digit card number (no spaces please):\n")
    exp_iss = input("Enter the date of issue (mm:yy):\n")
    exp_dat = input("Enter the expiry date (mm:yy):\n")
    hol_sec_code = int(input("Enter the 3 digit security code:\n"))
    cred_lim = float(input("Enter the credit limit (Rand):\n"))
    cred_lim_input = "{:.2f} Rand".format(cred_lim)
    bal = float(input("Enter the balance (Rand):\n"))
    #SlicedVariables
    sliced_acc = hol_acc[:4]+" "+hol_acc[4:8]+" "+hol_acc[8:12]+" "+hol_acc[12:]
    sliced_date1 = exp_dat[:2]+"/"+exp_dat[3:5]
    sliced_date2 = exp_iss[:2]+"/"+exp_iss[3:5]
    #AvailableBalance
    aval_bal = cred_lim - bal
    avail_bal = "{:.2f} Rand".format(aval_bal)
    bal_imp = "{:.2f} Rand".format(bal)
    #CreditCard
    print("+------------- Credit Card Account --------------+")
    print("| Card holder:   {0:<31} |".format(hol_name))
    print("| Card number:   {0:<32}|".format( sliced_acc))
    print("| Date of issue: {0:}    Expiry date: {1:<10}|".format(sliced_date2,sliced_date1))
    print("| Security code: {0:<32}|".format(hol_sec_code))
    print("| Credit limit:  {0:<32}|".format(cred_lim_input))
    print("| Available:     {0:<32}|".format(avail_bal))
    print("| Balance:       {0:<32}|".format(bal_imp))
    print("+------------------------------------------------+")
    
    
    
    
main()
