def main():
    name = input("Enter the card holder's name:\n")
    number = input("Enter the 16 digit card number (no spaces please):\n")
    date = input("Enter the date of issue (mm:yy):\n")
    expiry = input("Enter the expiry date (mm:yy):\n")
    digit = input('Enter the 3 digit security code:\n')
    credit = int(input('Enter the credit limit (Rand):\n'))
    balance = int(input('Enter the balance (Rand):\n'))
    # changing format of expriry date
    date2 = date[0:2]
    date3 = date[3:]
    date1 = date2+'/'+date3
    expiry2 = expiry[0:2]
    expiry3 = expiry[3:]
    expiry1 = expiry2+'/'+expiry3
    # separating the numbers 
    n = number[0:4]
    p = number[4:8]
    a = number[8:12]
    k = number[12:16]
    # getting the values and turning them to look like the output
    available = credit-balance
    credit1 = str(credit)+'.00 Rand'
    balance1 = str(balance)+'.00 Rand'
    available1 = str(available)+'.00 Rand'
    # print out the results
    print("+------------- Credit Card Account --------------+")
    print("| Card holder:   {0:<32}|".format(name))
    print("| Card number:  ",n,p,a,"{0:<17}|".format(k))
    print("| Date of issue:",date1,'   Expiry date: {0:<10}|'.format(expiry1))
    print("| Security code: {0:<32}|".format(digit))
    print("| Credit limit:  {0:<32}|".format(credit1))
    print("| Available:     {0:<32}|".format(available1))
    print("| Balance:       {0:<32}|".format(balance1))
    print("+------------------------------------------------+")
main()
