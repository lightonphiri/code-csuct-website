name = input("Enter the card holder's name:\n")
number = input("Enter the 16 digit card number (no spaces please):\n")
issue = input("Enter the date of issue (mm:yy):\n")
expiry = input("Enter the expiry date (mm:yy):\n")
code = input("Enter the 3 digit security code:\n")
limit = eval(input("Enter the credit limit (Rand):\n"))
balance = input("Enter the balance (Rand):\n")

if balance[:1] == "0":
    balance = balance[1:]
    
balance_new = eval(balance)
    
       
available = float(limit - balance_new)
part1 = number[0:4]
part2 = number[4:8]
part3 = number[8:12]
part4 = number[12:16]
date1 = issue[0:2]
date2 = issue[3:5]
expiry1 = expiry[0:2]
expiry2 = expiry[3:5]
rand = " Rand"
bal1 = "{0:.2f}".format(balance_new)
bal2 = str(bal1+ rand)
avail1 = "{0:.2f}".format(available)
avail2 = str(avail1+ rand)
lim1 = "{0:.2f}".format(limit)
lim2 = str(lim1+ rand)
print("+------------- Credit Card Account --------------+")
print("| Card holder:   {0:<32}|".format(name))
print("| Card number:   {0} {1} {2} {3:<17}|".format(part1,part2,part3,part4))
print("| Date of issue: {0}/{1}    Expiry date: {2}/{3:<7}|".format(date1,date2,expiry1,expiry2))
print("| Security code: {0:<32}|".format(code))
print("| Credit limit:  {0:<32}|".format(lim2))
print("| Available:     {0:<32}|".format(avail2))
print("| Balance:       {0:<32}|".format(bal2))
