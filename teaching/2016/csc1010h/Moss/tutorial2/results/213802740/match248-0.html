from math import sqrt

def main():
    side = int(input('Enter the length of the side (cm):\n'))
    area = ((3 * sqrt(3)) / 2) * side ** 2
  
    
    print('The area of a hexagon of side',side, 'is '+'{0:.3f}'.format(round(area,3))+'.')
