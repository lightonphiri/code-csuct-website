import math  # Enter the import math function,because we are gonna use it.
def main():  # Define the main.
    side = int(input("Enter the length of the side (cm):\n"))# Prompt user to enter the length of the side in cm.
    # The area of a hexagon is:(3*math.sqrt(3)\2 multiplied by side^2).
    a = 3*math.sqrt(3)/2 # Store 3*math.sqrt(3)\2 in a variable
    area = a*side**2 # Multiply the variable above by the side^2(assign all this to a variable)
    print("The area of a hexagon of side",side,"is {0:^6.3f}".format(area)+".")
