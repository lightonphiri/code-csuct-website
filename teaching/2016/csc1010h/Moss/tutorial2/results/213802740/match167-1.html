def main():
    len_side=int(input("Enter the length of the side (cm):\n"))
    # The formula to calculate the area of a regular triangel
    Area=3*math.sqrt(3)*len_side**2/2
    print("The area of a hexagon of side {0} is {1}.".format(len_side,round(Area,3)))
