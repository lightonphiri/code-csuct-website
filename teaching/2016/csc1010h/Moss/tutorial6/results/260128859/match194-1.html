def equals(w1,w2):
    if len(w1)==0 or len(w2)==0:
        return True
    elif len(w1)!=len(w2):
        return False
    elif w1[0]==w2[0]:
        if len(w1)==1:
            return True
        else:
            return equals(w1[1:],w2[1:])
    else:
        return False

    if n==0:
        return 0
    elif n==1:
        return 1
