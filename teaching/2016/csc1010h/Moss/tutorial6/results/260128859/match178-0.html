import string

def palindrome(uu):
    if len(uu)==1:
        return uu.isalpha()


    elif uu=="":
        return True
    elif uu[0] in string.punctuation or uu[0] in string.whitespace:
        return palindrome(uu[1:])
    elif uu[-1] in string.punctuation or uu[-1] in string.whitespace:
        return palindrome(uu[:-1])
    elif uu[0].lower()==uu[-1].lower():
        return palindrome(uu[1:-1])
    else:
        return False
