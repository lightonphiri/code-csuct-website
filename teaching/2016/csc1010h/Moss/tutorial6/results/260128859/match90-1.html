def query(pattern, word):
    if pattern == "" and word == "":
        return True 
    elif pattern[0] == word[0] or pattern[0] == "?":
        return query(pattern[1:], word[1:])
    else:
        return False

def match(pattern, word):


    if pattern == "*" and word == "":
        return True
    elif pattern[0] == word[0]:
        return match(pattern[1:], word[1:])
    elif pattern[-1] == word[-1]:
        return match(pattern[:-1], word[:-1])
    else:
        return False
