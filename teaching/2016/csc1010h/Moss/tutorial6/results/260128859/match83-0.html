def palindrome (s):

      return True
   elif len(s)==2:
      if s[0]==s[1] or s[0] and s[-1] in string.whitespace or s[0] and s[-1] in string.punctuation:
         return True
      elif s[0] in string.punctuation or s[0] in string.whitespace and s[-1] in string.punctuation or s[-1] in string.whitespace:
         return True
      elif s[0] in string.punctuation or s[0] in string.whitespace:
         return True
      elif s[-1] in string.punctuation or s[-1] in string.whitespace:
         return True
      else:
         return False
   elif s[0] in string.punctuation or s[0] in string.whitespace:
      return palindrome(s[1:])
   elif s[-1] in string.punctuation or s[-1] in string.whitespace:
      return palindrome(s[:-1])

      return False

def equals(word_one,word_two):
    if word_one=='' and word_two=='':
        return True
    elif word_one=='' and word_two!='':
        return False
    elif word_one!='' and word_two=='':
        return False
    elif len(word_one)==1 and len(word_two)==1:
        if word_one==word_two:
            return True
        else:
            return False
    elif word_one[0]==word_two[0]:
        return equals(word_one[1:],word_two[1:])
    else:
        return False

def sum_to_n(n):
    if n==1:
        return 1
    elif n<=0:
        m=-n
        return -1*(m+sum_to_n(m-1)-1)
        
    else:
        return n+sum_to_n(n-1)
