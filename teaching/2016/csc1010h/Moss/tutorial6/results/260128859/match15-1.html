import string
def palindrome (s):
 if len(s) == 1 or len(s) == 0:
  return True
 elif s[0] in string.punctuation or s[0] in string.whitespace:
  return palindrome(s[1:])
 elif s[-1] in string.punctuation or s[-1] in string.whitespace:
  return palindrome(s[:-1])


 elif s[0].lower() == s[-1].lower():
  return palindrome (s[1:-1])
 else:
  return False

def equals(word_one, word_two):
    if len(word_one)==len(word_two):
        if word_one == "" and word_two == "":
            return True


        elif word_one[0]==word_two[0]:
            return equals(word_one[1:], word_two[1:])
        else:
            return False
    else:
        return False

def query(pattern, word):
    if len(pattern)==len(word):
        if pattern == "" and word == "":
            return True
        elif pattern[0]==word[0] or pattern[0]=="?":


            return query(pattern[1:], word[1:])
        else:
            return False
    else:
        return False

def match(pattern, word):
    if pattern == "" and word == "":
        return True    
    if pattern=="*":
        return True
    elif word == '' and pattern != '*':
        return False
    elif pattern[0]=="*":
        return match(pattern[1:], word)
    if pattern[0] in word:
        word=word[word.find(pattern[0]):]
    if pattern[0]==word[0] or pattern[0]=="?":
        return match(pattern[1:], word[1:])
    else:
        return False

        return 2

def sum_to_n(n):
    if n == 1:
        return 1
    elif n == 0:
        return 0
    else:
        return n + sum_to_n(n-1)
