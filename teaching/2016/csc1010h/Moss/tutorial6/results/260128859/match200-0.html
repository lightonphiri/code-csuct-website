import string

def palindrome (k):
   if len(k) == 1:
      return k.isalpha()
   elif k[0] in string.punctuation or k[0] in string.whitespace:
      return palindrome(k[1:])
   elif k[-1] in string.punctuation or k[-1] in string.whitespace:
      return palindrome(k[:-1])
   elif k[0].lower() == k[-1].lower() and len(k)==2:
      return True
   elif k[0].lower() == k[-1].lower():
      return palindrome (k[1:-1])
   else:
      return False
