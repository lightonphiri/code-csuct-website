            return (s.isalpha())   
   elif len(str(s)) == 1:
      return s.isalpha()
   elif s[0] in string.punctuation or s[0] in string.whitespace:
      return palindrome(s[1:])

def equals(w_1, w_2):
    if (w_1 == "" and w_2 == "") and len(w_1) == len(w_2):
        return True
    elif (w_1 == "") and (w_2 != ""):
        return False
    elif (w_2 == "") and (w_1 != ""):
        return False    
    elif w_1[0].lower() == w_2[0].lower():
        return equals(w_1[1:-1].lower(),w_2[1:-1].lower())
    else:
        return False
    
    
def query(p, w):
    if w == "" and p == "?":
        return True     
    if len(p) == 0 and len(w) == 0:
        return True
    if  (len(p) == 1 and len(w) == 1) and (p == "?"):
        return True    
    if (len(p) > 1 and p[0] == '?') or (len(p) != 0 and len(w) !=0 and p[0] == w[0]):
        return query(p[1:].lower(),w[1:].lower())    
    else:
        return False
