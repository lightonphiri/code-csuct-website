def match(pattern,word):
    if pattern=='*' and len(word)>0:
        return True
    elif pattern =='' and len(word)>0:
        return False
    elif word == '' and len(pattern)>0:
        if '*' in pattern:
            pass
        else:
            return False
    elif len(pattern)>len(word) and '*' in pattern:
        pattern = pattern[:pattern.find("*")] + pattern[pattern.find("*")+1:]
        return match(pattern,word)
    elif pattern =='' and word =='':
        return True
    elif pattern[0] =='?':
        return match(pattern[1:],word[1:])
    elif pattern[0] =='*':
        pattern = pattern[1:]
        return match(pattern,word[1:][word[1:].find(pattern[0]):])
    elif pattern[0] == word[0]:
        return match(pattern[1:],word[1:])

        return False
