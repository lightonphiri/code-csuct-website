import string

def equals(word_one,word_two):
    if len(word_one) != len(word_two):
        return False
    if len(word_one) and len(word_two) < 2 and word_one == word_two:
        return True
    if word_one == '' or word_two == '':
        return False
    if word_one[0] in string.punctuation or word_one[0] in string.whitespace:
        return equals(word_one[:-1],word_two[:-1])
    elif word_one[-1] in string.punctuation or word_one[-1] in string.whitespace:
        return equals(word_one[:-1],word_two[:-1])      
    else:
        if word_one[0].lower() == word_two[0].lower():
            return equals(word_one[1::],word_two[1::])
        else:
            return False

    if pattern == '':
        return not word

        return False

        if pattern[0] == '?':
            return query(pattern[1::],word[1::])
        else:
            return pattern[0] == word[0] and query(pattern[1:],word[1:])
