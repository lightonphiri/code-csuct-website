def equals(word_one,word_two):
    if len(word_one) == len(word_two):
        if word_one =="" and word_two == "":
            return "True"
        elif word_one[0] != word_two[0]:
            return "False"
        else:
            return equals(word_one[1:],word_two[1:])
    else:
        return "False"

        if pattern =="" and word == "":

    else:
        return "False"

def match(pattern,word):
    if pattern =="" and word == "":
            return "True"
    elif pattern[0]=="*":
        if len(pattern)==1:
            return True
        elif len(word)==0 and pattern[1].isalpha():
            return False
        else:
            return match(pattern[1:],word[word[1:].find(pattern[1])+1:])
    elif pattern[0] == "?":
        return match(pattern[1:],word[1:])
    elif pattern[0] != word[0]:
        return "False"
    else:
        return match(pattern[1:],word[1:])
