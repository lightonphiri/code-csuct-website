def equals(word_one,word_two):
    if len(word_one)==0 and len(word_two)==0:
        return True
    elif len(word_one)==0 or len(word_two)==0:
        return False
    elif word_one[0:]==word_two[0:]:
        return equals(word_one[1:],word_two[1:])
    else:
        return False
