def equals(word_one,word_two):
    
    if len(word_one)!= len(word_two):
        return False
    elif len(word_one)==0 and len(word_two) ==0:
        return True
    elif word_one[-1]== word_two[-1]:
        return equals(word_one[1:],word_two[1:])
    else: 
        return False

def population(n):
    if n==0:
        return 0
    elif n==1:
        return 2
    else:
        return population(n-1)+population(n-2)

        return 0

        return n + sum_to_n(n-1)
