def equals(word_one,word_two):
    
    if len(word_one)==0 and len(word_two)==0:
        return True
    
    elif len(word_one)!=len(word_two):
        return False
    if word_one[-1]==word_two[-1]:
        return equals(word_one[:-1],word_two[:-1])
    else:
        return False
    #part two,query function
    
def query(pattern,word):
    if len(pattern)==0 and len(word)==0:
        return True
    elif len(pattern)!=len(word):
        return False


    elif pattern[0]==word[0] or pattern[0]=="?":
        return query(pattern[1:],word[1:])
    else:
        return False

def match(pattern,word):
    if len(word)==0 and (pattern[0]=="*" or pattern[-1]=="*"):
        return True
    
    elif (word[-1]!=pattern[-1] and pattern[-1]!="*") or (word[0]!=pattern[0] and pattern[0]!="*"):
        return False
    
    elif word[0]==pattern[0]:
        return match(pattern[1:],word[1:])
    
    elif word[-1]==pattern[-1]:
        return match(pattern[:-1],word[:-1])
    elif (pattern[0]=="*" or pattern[-1]=="*") and len(word)!=0:
        return match(pattern,word[:-1])
    else:
        return False
