def equals(word_one,word_two):
    if word_one == '' and word_two == '':
        return True
    elif len(word_one) != len(word_two):
        return False
    elif word_one[0] == word_two[0]:


        return equals(word_one[1:],word_two[1:])
    else:
        return False
    
def query(pattern,word):
    if len(pattern) == len(word) or len(pattern) ==  0  and len(word) == 0 :
        return True
    elif pattern[0] == word[0]:
        return query(pattern[1:], word[1:])
    else:
        return False
def match(pattern,word):
    if pattern == '*' and word == '':
        return True


    elif pattern[0] == '?' and word != '':
        if pattern[1] == word[0]:
            return match(pattern[1:],word[1:])
        else:
            return True


    elif pattern == '*' and word != '':
        return True
    elif pattern[0] == '*':
        if pattern[-1] == word[-1]:
            return match(pattern[:-1],word[:-1])
        else:
            return match(pattern[1:],word[1:])       
    elif pattern[0] != '*' or pattern[-1] != '*':


        if pattern[0] == word[0]:
            return match(pattern[1:],word[1:])  
        else:
            return False
    else:
        return False

def sum_to_n(n):
    if n==0:
        return 0
    elif n > 0:
        return n + sum_to_n(n-1)
    else:
        return n + sum_to_n(n+1)
