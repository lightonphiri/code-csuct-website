import string

def palindrome (s):
   if len(s) == 1: # Checks if the string length is 1
      return s.isalpha()
   elif s[0] == s[-1] and len(s) == 2: # Checks if the string length is 2 and if first and last char are the same
      return s.isalpha()
   elif s[0] in string.punctuation or s[0] in string.whitespace: # Removes punctuation or whitespace from first position
      return palindrome(s[1:])
   elif s[-1] in string.punctuation or s[-1] in string.whitespace: # Removes punctuation or whitespace from last position
      return palindrome(s[:-1])
   elif s[0].lower() == s[-1].lower(): # Checks if first letters are the same
      return palindrome (s[1:-1])
   else:
      return False

def equals(word_one, word_two):
    if len(word_one) == 0 and len(word_two) == 0: return True # Checks if both word length are empty
    elif len(word_one) == 0 or len(word_two) == 0: return False # Checks if one of the word length are empty
    elif not(word_one[0] == word_two[0]): return False # Checks if first letters are the same
    else: return equals(word_one[1:], word_two[1:]) # Returns both words without first letters

def query(pattern, word):
    if len(pattern) == 0 and len(word) == 0: return True # Checks if both word length are empty 
    elif len(pattern) == 0 or len(word) == 0: return False # Checks if one of the word length are empty
    elif pattern[0] == word[0]: return query(pattern[1:], word[1:]) # Checks if first letters are the same


    elif pattern[0] == '?': return query(pattern[1:], word[1:]) # Checks if first letter of the pattern is ?
    else: return False
    
def match(pattern, word):
    if len(pattern) == 0 and len(word) == 0: return True # Checks if both word length are empty 
    elif len(pattern) == 0 or len(word) == 0: return False # Checks if one of the word length are empty


    elif pattern[0] == word[0]: return match(pattern[1:], word[1:]) # Checks if first letters are the same
    elif pattern[0] == '?': return match(pattern[1:], word[1:]) # Checks if first letter of the pattern is ?
    elif pattern[0] == '*': # Checks if first letter of the pattern is *
        if len(pattern) == len(word): return match(pattern[1:], word[1:])
        elif len(pattern) > len(word): return match(pattern[1:], word[:])
        else: return match(pattern[:], word[1:])
    else: return False
