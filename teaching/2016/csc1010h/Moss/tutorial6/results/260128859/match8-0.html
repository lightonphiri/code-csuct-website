import string

def palindrome (s):
   if len(s)==2 and s[0]==s[1]:
      return s.isalpha()
   elif len(s) == 1:
      return True 
   elif s[0] in string.punctuation or s[0] in string.whitespace:
      return palindrome(s[1:])
   elif s[-1] in string.punctuation or s[-1] in string.whitespace:
      return palindrome(s[:-1])


   elif s[0].lower() == s[-1].lower():
      return palindrome (s[1:-1])
   else:
      return False

def equals(word_one,word_two):
    if len(word_one)==len(word_two):
        if len(word_one)==1 and word_one[0]==word_two[0] :
            return True


        elif word_one[0]==word_two[0]:
            return equals(word_one[1:],word_two[1:])
        else:
            return False   
    else:
        return False
    
def query(word_1,word_2):
    if len(word_1)==1 and word_1[0]==word_2[0]:
        return True


    elif len(word_1)==1 and len(word_2)==1 and word_1[0]=="?":
        return True
    elif word_1[0]=="?":
        return query(word_1[1:],word_2[1:])
    elif word_1[0]==word_2[0]:
        return query(word_1[1:],word_2[1:])
    else:
        return False

def match(pattern,word):
    if len(pattern)==1 and pattern[0]==word[0]:
        return True
    elif len(pattern)==1 and pattern[0]=="*":
        return True


    elif len(pattern)==1 and pattern[0]=="?":
        return True
    elif pattern[0]=="?":
        return match(pattern[1:],word[1:])
    elif pattern[0]=="*" and len(pattern)<=len(word):
        return match(pattern[:0:-1],word[:0:-1])
    elif pattern[0]=="*" and len(pattern)>len(word):
        return match(pattern[:0:-1],word[::-1])


    elif pattern[0]==word[0]:
        return match(pattern[1:],word[1:])
    else:
        return False
