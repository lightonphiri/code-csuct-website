def is_leap_year():
    year=eval(input('enter the year: '))
    
    if year % 4 ==0 and year % 100==0 and year % 400==0:
        print(year,'is a leap year.')
    elif year % 4==0 and year % 100!=0:
        print(year,'is a leap year')
    else:
        print(year,'is not a leap year')
    return
