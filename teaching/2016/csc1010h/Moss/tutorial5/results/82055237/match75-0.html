def is_leap_year(year):
    a=year
    if len(str(a))!=4:
        return False
    else:
        n=1000
        while True:
            b=a-n
            if b==0:
                return True         
            elif b<0:
                return False
            n+=4
    
def month_name(number):
    months=["January","February","March","April","May","June","July","August","September","October","November","December"]
    if number==1:
        return "January"
    else:
        return months[number-1]

def days_in_month(month_number, year):
    while True:
        if 0<=month_number<=7:
            if month_number%2==0:
                if month_number==2:
                    if is_leap_year(year)==True:
                        return 29
                    else:
                        return 28
                else:
                    return 30
            else:
                return 31
        elif 7<month_number<=12:
            if month_number%2==0 :
                return 31
            else:
                return 30
        
def first_day_of_year(year):
    day=(1+5*((year-1)%4)+4*((year-1)%100)+6*((year-1)%400))%7
    return day
def first_day_of_month(month_number , year):
    for i in range(month_number-1):
        if i ==0:
            sum_days = days_in_month(i+1,year)
        else:
            sum_days += days_in_month(i+1,year)
    return (sum_days + first_day_of_year(year))%7

def code_for(letters):
    sym=[".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-.._","-.--","--.."]
    alpha=["a","b","c","d","e","f","g","h","i","j","k","l","m","n","o","p","q","r","s","t","u","v","w","x","y","z"]
    n=0
    dicti={}
    for i in range(len(alpha)):
        dicti[alpha[n]]=sym[n]
        n+=1
    return dicti[letters.lower()] 

def transmitt_letter(letters):
    u=code_for(letters)
    w=""
    for v in range(len(u)):
        if v ==len(u)-1:
            w+=u[v]
        else:
            w+=u[v]+'^'
    return w

def transmit_letter(letters):
    return print(transmitt_letter(letters),end="")
    
def transmit_word(word):
    words=[]
    k=""
    for r in range(len(word)):
        words.append(word[r])
    for t in range(len(word)):
        if t==len(word)-1:      
            k+=transmitt_letter(words[t])
        else:
            k+=transmitt_letter(words[t])+"|"
    return print(k,end="")
