def code_for(letter):
    letters={'a':'.-','b':'-...','c':'-.-.','d':'-..','e':'.','f':'..-.','g':'--.','h':'....','i':'..','j':'.---','k':'-.-','l':'.-..','m':'--','n':'-.','o':'---','p':'.--.','q':'--.-','r':'.-.','s':'...','t':'-','u':'..-','v':'...-','w':'.--','x':'-..-','y':'-.--','z':'--..',' ':''}
    return letters[letter.lower()]

def transmit_letter(letter):
    code=code_for(letter)
    code1=''
    for i in code:
        code1+=i+'^'
    print(code1[:-1],end='')
    
def transmit_word(word):
    code2=''
    for i in word:
        code=code_for(i)
        code1=''
        for i in code:
            code1+=i+'^'        
        
        code2+=code1[:-1]+'|'
    print(code2[:-1],end='')

def default_response():
    return 'Curious, tell me more.'
     
def get_response(words):#debugging
    res_1=''
    keys=responses.keys()
    for word in words:
        for key in keys:
            if word.lower()==key:
                res_1=word.lower()
             
    if res_1!='':
        return responses[res_1]
    else:
        return default_response()
