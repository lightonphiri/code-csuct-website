def code_for(l):
    letter = {'a': '.-', 'b' :'-...','c': '-.-.', 'd':'-..', 'e':'.', 'f':'..-.', 'g':'--.', 'h':'....', 'i':'..', 'j':'.---','k':'-.-','l':'.-..', 'm':'--', 'n':'-.', 'o':'---', 'p':'.--.', 'q':'--.-', 'r':'.-.', 's':'...', 't':'-', 'u':'..-', 'v':'...-', 'w':'.--', 'x':'-..-','y':'-.--', 'z':'--..'}
    return letter[l] 

def transmit_letter(l):
    string = ""
    letter = code_for(l.lower())
    
    for i in range(len(letter)):
        string += letter[i]
        string += "^"
    
    print(string[:-1], end="")

def transmit_word(word):
    length = len(word)
    
    for i in range(length - 1):
        transmit_letter(word[i])
        print("|",end='')
    
    transmit_letter(word[length - 1])

def word_pause():
    print('||',end='')
