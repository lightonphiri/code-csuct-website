    if number in months:
        return months[number]

        return "Invalid Number"

    if is_leap_year(year):
        return days_in_month_leap_year_dict[months[month_number]]

        return days_in_month_dict[months[month_number]]

    day_number=5*fmod(year-1,4)
    day_number+=4*fmod(year-1,100)
    day_number+=6*fmod(year-1,400)
    day_number=fmod(day_number+1,7)
    return days[day_number]

    first_day=day_number[first_day_of_year(year)]

    else:
        for i in range(month_number):

    return day_number[days[total_days_past-31+first_day]]

def code_for(letter):
    morse_letter=''
    letter=letter.lower()
    for i in letter:
        morse_letter+=morse_code[i]
    return morse_letter

def transmit_letter(letter):
    morse_word=''
    morse_letter = code_for(letter)
    for code in morse_letter:
        if code == '.':
            beep(1000,100)
            morse_word+=code
        else:
            beep(1000,300)
            morse_word+=code
        sleep(1)
        morse_word+='^'
    print(morse_word, end='')
    return morse_word

def default_response():
    return 'Curious, tell me more.'
    
def get_response(words):
    for word in words:
        if word in responses:
            return responses[word]
    return default_response()

def print_welcome():
    print('Welcome to the automated technical support system.')
    print('Please describe your problem:')

def get_input():
    query = []
    query = input().lower()


    query = query.split(' ')
    return query

def main():
    print_welcome()
    while True:
        query = get_input()
        if query==['quit']: break
        response=get_response(query)
        print(response)
