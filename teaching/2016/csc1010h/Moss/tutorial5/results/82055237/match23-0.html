def transmit_letter(letter):
    alpha = '^'.join(code_for(str(letter)))
    return print(alpha,end='')

def transmit_word(word):
    mylist = []
    n = 0
    wd = ""
    for i in word:
        mylist.append(str('^'.join(code_for(str(i))))) 
        if n != len(word)-1:
            mylist.append(str('|'))
            n += 1

    for char in mylist:
        wd += char
    return print(wd,end='')

responses = {

    'windows':'Ah, I think I see your problem. What version?',
    'apple':'You do mean the computer kind?',
    'blue':'Ah, the blue screen of death. And then what happened?',

def default_response():
    return 'Curious, tell me more.'
    
def get_response(words):
    for i in range(1000):
        for word in words:
            word1 = word.lower()
            if word1 in responses:
                return responses[word1]
            elif not word1 in responses:
                continue
        if word1 == 'quit':
            return 'quit'
        return default_response()
 
    
def print_welcome():
    print('Welcome to the automated technical support system.')
    print('Please describe your problem:')


def get_input():
    query = input().split(' ')
    return query

def main():
    print_welcome()
    query = get_input()
    while True:
        response = get_response(query)
        if response == 'quit':
            break
        print(response)
        query = get_input()

main()
