def is_leap_year(year):
    if year % 4 == 0 and year % 100 != 0 or  year % 400 == 0 :
        return True
    else:
        return False
    


def month_name(mpume):
    dict = {1: "January", 2: "February", 3: "March", 4: "April", 5: "May", 6: "June", 7: "July", 8: "August", 9: "September", 10: "October", 11: "November", 12: "December"}
    return dict[mpume]

def days_in_month(month_no, year):
    if month_name(month_no) == "January" or month_name(month_no) == "March" or month_name(month_no) == "May" or month_name(month_no) == "July" or month_name(month_no) == "August" or month_name(month_no) == "October" or month_name(month_no) == "December" :
        return 31
    elif month_name(month_no) == "April" or month_name(month_no) == "June" or month_name(month_no) == "September" or month_name(month_no) == "November":
        return 30
    elif month_name(month_no) == "February" and is_leap_year(year) == True:
        return 29
    elif month_name(month_no) == "February" and is_leap_year(year) == False:
        return 28
def first_day_of_year(year):
    day = ((1+5*((year-1)%4) + 4*((year-1)%100) + 6*((year - 1)%400))% 7)
    return day
