def code_for(letter):
    letters=list('abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ')
    a=letters[:len(letters)//2]
    b=letters[len(letters)//2:]
    code=['.-','-...','-.-.','-..','.','..-.','--.','....','..','.---','-.-','.-..','--','-.','---','.--.','--.-','.-.','...','-','..-','...-','.--','-..-','-.--','--..']
    
  
    if letter in a:
        position = a.index(letter)
        element=code[position]
        return element
    elif letter in b:
        position = b.index(letter)
        element=code[position]
        return element

    x=code_for(letter)

def get_input():
    ques = input().lower().split(' ')
    return ques


def default_response():
    return 'Curious, tell me more.'
    
def get_response(ques):
    for word in ques:
        if word in responses:
            return responses[word]
    return default_response()
    

def print_welcome():
    print('Welcome to the automated technical support system.')
    print('Please describe your problem:')




def main():
    print_welcome()
    query=get_input()
    while query != ['quit']:
        print('',end='')
        response=get_response(query)
        print(response)
        query=get_input()
