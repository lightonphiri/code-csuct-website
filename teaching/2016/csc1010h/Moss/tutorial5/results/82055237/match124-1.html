responses={
    'bluetooth':'Have you tried mouthwash?',
    'windows':'Ah, I think I see your problem. What version?',
    'apple':'You do mean the computer kind?',
    'blue':'Ah, the blue screen of death. And then what happened?',
    'spam':'You should see if your mail client can filter messages.',
    'connection':'Contact Telkom.',
    'crashed':'Are the drivers up to date?',


    'hacked':'You should consider installing anti-virus software.'}


def default_response():
    return 'Curious, tell me more.'
    
def get_response(words):
    for word in words:
        if word.lower() in responses:                         
            return responses.get(word.lower())
    return default_response()
    

def print_welcome():
    print('Welcome to the automated technical support system.')
    print('Please describe your problem:')

def get_input():
    query = input().split(' ')
    return query

def main():
    print_welcome()
    query = get_input()
    while query!=['Quit'] and query!= ['quit'] and query!= ['qUIT'] and query!= ['QUIT'] and query!=['quiT']:
        response=get_response(query)
        print(response)
        query=get_input()
        
main()

#Algorithm to dubug in Wing IDE

#1. on the left hand side, along the line numbers, place a breakpoint (looks like a red dot or red dash) on the areas of your code you wish to debug.
#2. thereafter, look at the top of the screen at all you options.
#3. you should see a green start button, alongside this green start button, you should also see something that looks like a white box with a red dot inside, and also something that looks like a red bug.
#4. once you see this red bug, click on it.
#5. the dubugging has now started, and you should see a red line across one of your lines of code in your program.
#6. thereafter look to the bottom left hand side of your screen, and there should be two option boxes called 'Search' and 'Stack Data'. Click on the 'Stack Data' box.
#7. in the stack data box, you will see information come up, this information tells you about what a specific line of code at the red line and what it does.
#8. now look back at the top of the screen, and near the centre, you should see three arrows with black boxes underneath them. Click on the arrow pointing to the right to debug your next line of code.
#9. now look at the stack data box to see what the line of code does and if there are any errors in it.
#10. if no errors are found, proceed to the next line of code.
#11. repeat this process until all your errors are found, your code is correct, and your program is running correctly.
