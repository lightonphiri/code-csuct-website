def is_leap_year(year):
    if year == 1900:
        return False
    elif year % 4 == 0:
        return True
    elif year % 100 == 0 and year % 400 == 0:
        return True
    else:
        return False

def code_for(letter):
    letter_code = {'a':'.-','b':'-...','c':'-.-.','d':'-..','e':'.','f':'..-.',
                  'g':'--.','h':'....','i':'..','j':'.---','k':'-.-','l':'.-..',
                  'm':'--','n':'-.','o':'---','p':'.--.','q':'--.-','r':'.-.',
                  's':'...','t':'-','u':'..-','v':'...-','w':'.--','x':'-..-',
                  'y':'-.--','z':'--..'}
    
    return letter_code[letter.lower()]
      
def transmit_letter(letter):
    code = '^'.join(list(code_for(letter)))
  
    return code
    
def transmit_word(word):

def default_response():
    return 'Curious, tell me more.'
    
def get_response(words):
    for word in words:
        if word in responses:
            return responses[word]   

    return default_response() 
    


def print_welcome():
    print('Welcome to the automated technical support system.')
    print('Please describe your problem:')

def get_input():
    query = input().lower().split(' ')
    return query

def main():
    print_welcome()
    while True:
        query = get_input()
