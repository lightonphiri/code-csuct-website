import calendar_utils

def to_string(n, e):
        if n<1 or n>e:
                return '  '
        else:
                return '{:>2}'.format(n)
        
def row(i, s, e):
        start = i*7+s
        result=to_string(start, e)
        for column in range(start+1, start+7):
                result+=' '+to_string(column, e)  
        return result

def row_of_months(i, start_month, year):
        
        start_day=(calendar_utils.first_day_of_month(start_month, year)-1)%7
        days=calendar_utils.days_in_month(start_month, year)
        
        result=row(i, 1-start_day, days)

        for month in range(start_month+1, start_month+3):
                result+='   '
                start_day=(start_day+days)%7
                days=calendar_utils.days_in_month(month, year)
                result+=row(i, 1-start_day, days)

        return result

def is_leap_year(year):
    if (year%4)==0:
        new_num = str(year)
        y = year%400
        if new_num[2:4]=='00' and y !=0:
            return False
        return True
    else:
        return False

def first_day_of_year(year):
    day = (1 + 5*((year-1)%4) + 4*((year - 1)%100) + 6*((year - 1)%400))%7
    return day

def code_for(letter):
    dic={'a':'.-','b':'-...','c':'-.-.','d':'-..','e':'.','f':'..-.','g':'--.','h':'....','i':'..','j':'.---','k':'-.-','l':'.-..','m':'--','n':'-.','o':'---','p':'.--.','q':'--.-','r':'.-.','s':'...','t':'-','u':'..-','v':'...-','w':'.--','x':'-..-','y':'-.--','z':'--..'}
    
    return dic[letter.lower()]

def transmi_letter(letter):
    wode = code_for(letter)
    new_word=''
    word=[]
    for i in range(1,len(wode)):
        u = wode[i-1:i]
        word.append(u)
    for i in word:
        new_word += (i+'^')
    new_word=new_word+wode[-1:-2:-1]
    return new_word
    
def transmit_letter(letter):
    return print(transmi_letter(letter),end="")
def transmit_word(word):
    new_list=[]
    for i in range(len(word)):
        save=word[i]
        save=transmi_letter(save)
        new_list.append(save)
    wrd=''
    for i in new_list:
        wrd +=(i+'|')
    wrd=wrd[0:len(wrd)-1]
    return print(wrd,end='')
    

def word_pause():
    print('||',end='')
    sleep(0.1*7)
