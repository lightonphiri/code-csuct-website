def is_leap_year(year):
    
    if year%4==0 and year%100!=0 or year%400==0:
        is_leap_year = True
        return is_leap_year
    else:
        is_leap_year = False
        return is_leap_year

def days_in_month(month_number, year):
    monthLength = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
    monthLengthleap = [31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
    if is_leap_year(year):
        final = monthLengthleap[month_number-1]
    elif not is_leap_year(year):
        final = monthLength[month_number-1]
    return final
