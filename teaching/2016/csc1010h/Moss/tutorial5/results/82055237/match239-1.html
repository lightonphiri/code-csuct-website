def is_leap_year(year):
    if year % 4 == 0 and year % 100 != 0 or year % 400 == 0:
        return True
    else:
        return False     
def month_name(month_number):
    number = {1:'January',2:'February',3:'March',4:'April',5:'May',6:'June',7:'July',8:'August',9:'September',10:'October',11:'November',12:'December'}
    return number[month_number]

def days_in_month(month_number,year):
 
    if month_name(month_number) == 'January' or month_name(month_number) =='March' or month_name(month_number) =='May' or month_name(month_number) =='July' or month_name(month_number) =='August' or month_name(month_number) =='October' or month_name(month_number) =='December':
        return 31
    
    elif month_name(month_number) == 'April' or month_name(month_number) =='June' or month_name(month_number) =='September' or month_name(month_number) =='November':
        return 30
    
    elif month_name(month_number) == 'February' and is_leap_year(year) == True:
        return 29
    
    elif month_name(month_number) == 'February' and is_leap_year(year) == False:
        return 28
    
def first_day_of_year(year):
    first_of_jan = ((1+5*((year-1)%4)+4*((year-1)%100)+6*((year-1)%400))%7)
