def is_leap_year(year):
    if year == 1900:
        return "The year "+str(year)+" is not a leap year."
    if (year % 4 == 0 and year % 100 != 0 or year % 400 == 0) :
        return "The year "+str(year)+" is a leap year." 
    else:
        return "The year "+str(year)+" is not a leap year."

responses={
    'bluetooth':'Have you tried mouthwash?',
    'windows':'Ah, I think I see your problem. What version?',
    'apple':'You do mean the computer kind?',
    'blue':'Ah, the blue screen of death. And then what happened?',
    'spam':'You should see if your mail client can filter messages.',
    'connection':'Contact Telkom.',
    'crashed':'Are the drivers up to date?',
    'hacked':'You should consider installing anti-virus software.'}
    
def get_response(words):
    lower_case_query = []
    for ele in words:
        lower_case_query.append(ele.lower())
        
    if 'bluetooth' in lower_case_query:
        return responses['bluetooth']
    if 'windows' in lower_case_query:
        return responses['windows']
    if 'apple' in lower_case_query:
        return responses['apple']
    if 'blue' in lower_case_query:
        return responses['blue']
    if 'spam' in lower_case_query:
        return responses['spam']
    if 'connection' in lower_case_query:
        return responses['connection']
    if 'crashed' in lower_case_query:
        return responses['crashed']
    if 'hacked' in lower_case_query:
        return responses['hacked']
    else:
        return 'Curious, tell me more.'
    
def print_welcome():
    print('Welcome to the automated technical support system.')
    print('Please describe your problem:')

def get_input():
    query = input().split(' ')
    lowerquery =[]
    for i in query:
        lowerquery.append(i.lower())
    return lowerquery

def main():
    print_welcome()
    query = get_input()
    while query != ['quit']:
        response=get_response(query)
        print(response)
        query = get_input()
        
main()
