def is_leap_year(year):#checks whether the year is a leap year
    year_string = str(year)
    if 999 < year < 9999:#checks if the year put in is 4 digits
        if year_string[2:] == "00":#if it is the start of the century
            if year % 4 == 0 and year % 100 == 0 and year % 400 == 0: #for a year to be leap year, it has to be divisible by 4, 100 and 400
                return True #if it is divisible by 4, 100 and 400, then it is confirmed to be a leap year
            else:
                return False #if it is not divisible by either 4, 100 or 400, then it is confirmed to not be a leap year
        else:#if it is not the start of the century
            if year % 4 == 0:#for a year to be a leap year, it has to be divisible by 4
                return True#if it is divisible by 4, then it is confirmed to be a leap year
            else:
                return False#if it is not divisible by 4, then it is not a leap year
    else:
        pass #if it does not have 4 digits, then the programme must stop running

    b = code_for(letter)
    c = len(b)
    e = []
    for d in range(0, c):
        if b[d] == ".":
            e.append(".")

        elif b[d] == " ":
            e.append("^")
    for d in range(0, c):

    return e

def default_response():
    return "Curious, tell me more."
    
def get_response(words):
    for word in words:
        if word.lower() in responses:
            return responses[word.lower()]
    return default_response()
    

def print_welcome():
    print("Welcome to the automated technical support system.")
    print("Please describe your problem:")

    query = input().split(" ")
    return query

def main():
    print_welcome()
    query = get_input()
    while query != ["quit"] and query != ["Quit"] and query != ["QUIT"] and query != ["qUIT"]:
        response = get_response(query)
