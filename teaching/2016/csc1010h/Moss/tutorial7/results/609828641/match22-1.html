from time import*
from random import*

def name_output_file():
    return input('Enter the name of the output file:\n')

def subject_name():
    return input('Enter subject name (or press \'enter\' to quit):\n')

def tests_num():
    return int(input('Enter the number of tests:\n'))
    
def steady():
    print('Get ready...', flush=True)

def flush_input():
    try:
        import msvcrt


        while msvcrt.kbhit():
            msvcrt.getch()
    except ImportError:
        import sys, termios
        termios.tcflush(sys.stdin, termios.TCIOFLUSH)

def main():
    filename = name_output_file()      # name of the file to witch the results will be written.
    f = open(filename,'a')
    while True:
        sub_name = subject_name()
        if sub_name == '':
            break
        number = tests_num()
        total = 0
        for i in range(number):
            steady()
            sleep(randint(1,5))
            
            start_time = time()        # Calculating reaction time
            flush_input()
            reaction = input('Press \'enter\' NOW!')
            end_time = time()
            
            reaction_time = end_time - start_time # Reaction time
            total += reaction_time
            print('Reaction time: {:6f}'.format(reaction_time))
            
        average = total/2
        f.write('{0} {1} {2:6f}'.format(sub_name,number,reaction_time)+'\n')
        print('The average is {:6f}'.format(average))
    f.close()

main()
