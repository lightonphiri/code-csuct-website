def process_line(line):
    try:
        parts = line.split(' ')
        name = parts[0]
        trials = int(parts[1])
        average = float(parts[2])
    except ValueError as wrong:
        print(wrong)
       
    except ValueError as incorrect:
        print(incorrect)
        
    return (name, trials, average)

def get_max(entries):
    max = entries[0]
    for i in range(1, len(entries)):
        try:
            entries[i][2]
        except TypeError as something:
            print(something)
        else:
            if entries[i][2] <max[2]:
                max = entries[i]
    entries.remove(max)
    return max                 
def sort(entries):


    sorted_entries = []
    while len(entries)>0:
        sorted_entries.append(get_max(entries))
    return sorted_entries

def output(entries):
    print('Subject        | Average Time')
    print('************** | ************')
    count=0
    for entry in entries:
        count+=1
        print('{:14} | {:.6}'.format(entry[0], entry[2]))
