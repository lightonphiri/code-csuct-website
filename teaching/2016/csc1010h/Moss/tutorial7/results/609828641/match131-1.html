# leader_board.py



def process_line(line):
    try:
        parts = line.split(' ')
        name = parts[0]
        trials = int(parts[1])
        average = float(parts[2])
        
    except TypeError:
        print('The data in the file is not of correct format.')
        
    else:
        return (name, trials, average)

def get_max(entries):
    max = entries[0]
    for i in range(1, len(entries)):
        if entries[i][2] <max[2]:
            max = entries[i]
            
    entries.remove(max)
    return max

def sort(entries):
    sorted_entries = []
    while len(entries)>0:
        sorted_entries.append(get_max(entries))
    return sorted_entries

def output(entries):
    print('Subject        | Average Time')
    print('************** | ************')
    for entry in entries:
        print('{:14} | {:.6}'.format(entry[0], entry[2]))
              
def main():
    file_name = input('Enter the name of the file:\n')
    try:
        file_input = open(file_name, 'r')
    
    except IOError:
        print('File does not exist. Please enter a valid file name.')

    else:


        entries = []
        for line in file_input:
            entries.append(process_line(line))
        file_input.close()
            
        entries = sort(entries)
        output(entries)        
        
    
main()

from random import randint
import time

def flush_input():
    try:
        import msvcrt
        while msvcrt.kbhit():
            msvcrt.getch()
    except:
        import sys, termios
        termios.tcflush(sys.stdin, termios.TCIOFLUSH)

file = input('Enter the name of the output file:\n')
f = open(file,'w')
user = input('Enter subject name (or press "enter" to quit):\n')
f.write(user)

if user != '':
    testXs = eval(input('Enter the number of tests:\n'))
    rctnTimes = []
    
    while testXs != 0:
        testXs = testXs - 1
        print('Get ready...', flush = True)
        flush_input()
        time.sleep(randint(1,5))
        start_time = time.time()
        reaction = input("Press 'enter' NOW!")
        if reaction == '':
            elapsed_time = time.time() - start_time
            rctnTimes.append(round(elapsed_time,6))
            print('Reaction time:',round(elapsed_time,6))
    
    print('Test complete.')
    avgTime = 0
    for i in range(0,len(rctnTimes)):
        avgTime += rctnTimes[i]
    final = round(avgTime/len(rctnTimes),6)
    print('Average reaction time:',final)
    
    #f.write('Subject        | Average Time\n')
    #f.write('************** | ************\n')
    #f.write(user
    
    f.write('\n')
    f.write(str(final))
