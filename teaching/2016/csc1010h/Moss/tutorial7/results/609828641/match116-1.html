def process_line(line):
    parts = line.split(' ')
    name = parts[0]
    trials = int(parts[1])
    average = float(parts[2])
    return (name, trials, average)

def get_max(entries):
    max = entries[0]
    for i in range(1, len(entries)):
        if entries[i][2] <max[2]:
            max = entries[i]
            
    entries.remove(max)
    return max

def sort(entries):
    sorted_entries = []
    while len(entries)>0:
        sorted_entries.append(get_max(entries))
    return sorted_entries

import time #importing the time module
import random #importing the random module
def reaction():
    filename = input("Enter the name of the output file:\n")
    f = open(filename,"w") #opening the filename and allowing to write
    f.write("Subject        | Average Time") # writes this in the filename
    f.write("\n")
    f.write("************** | ************")

    while True:
        totalvalue=0 #counter
        thesubject=input("Enter subject name (or press 'enter' to quit):\n")
        
        if thesubject == "quit" :
            break
        else:
            totalvalue=0
            testvalue = eval(input("Enter the number of tests:\n"))
            for i in range(testvalue):
                print('Get ready...', flush=True)
                time.sleep(random.randint(1,5))
                timebefore = time.time()
                input("Press 'enter' Now!")
                timeafter = time.time()
                total_time=timeafter-timebefore
                totalvalue+=total_time
                print('{}{:.6}'.format("Reaction time: ",total_time))
            average=totalvalue/testvalue
            print("Test complete.")
            print('{}{:.6}\n'.format("Average reaction: ",average))
            print('{:14} | {:.6}'.format(thesubject,average),file=f)    
            
            
    f.close()
        
reaction()
