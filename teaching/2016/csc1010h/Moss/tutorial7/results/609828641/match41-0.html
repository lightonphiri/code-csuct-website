# leader_board.py

def process_line(line):
    parts = line.split(' ')
    name = parts[0]
    trials = int(parts[1])
    average = float(parts[2])
    return (name, trials, average)

def get_max(entries):
    max = entries[0]
    for i in range(1, len(entries)):
        if entries[i][2] <max[2]:
            max = entries[i]
            
    entries.remove(max)
    return max

def sort(entries):
    sorted_entries = []
    while len(entries)>0:
        sorted_entries.append(get_max(entries))
    return sorted_entries

def output(entries):
    print('Subject        | Average Time')
    print('************** | ************')
    for entry in entries:
        print('{:14} | {:.6}'.format(entry[0], entry[2]))
              
def main():
    while True:
        try:
            file_name = input('Enter the name of the file:\n')
            file_input = open(file_name, 'r')
            
            entries = []
            for line in file_input:
                entries.append(process_line(line))
            file_input.close()
            
            entries = sort(entries)
            output(entries)
            break
        except ValueError:
            print("Line 6, ValueError ")

            print("Correct Format")
        except IOError:
            print("Line 35, IOError")
            print("file_input = open(file_name, 'r')")
            print("Invalid File")
main()

import time
import random
output=input("Enter the name of the output file:\n")
otp = open(output,"w")
print('Subject        | Average Time',file=otp)
print('************** | ************',file=otp)
while True:   
    subject_name=("Enter subject name (or press 'enter' to quit):\n")


    if subject_name =="":
        break
    test_num=int(input("Enter the number of tests:\n"))
    sum_test = 0
    for i in range(test_num):
        print("Get ready...",flush=True)
        time.sleep(random.randint(1,5))


        time1=time.time()
        input("Press 'enter' NOW!")
        time2=time.time()
        total_time =time2-time1
        sum_test+=total_time
        print("{}{:.6f}".format("Reaction time:",total_time))
    average = sum_test/test_num
    print("Tests complete.")
    print("{}{:.6f}".format("Average reaction time:",average))
    print("{:14} | {:.6f}".format(subject_name,average),file=otp)
otp.close()
