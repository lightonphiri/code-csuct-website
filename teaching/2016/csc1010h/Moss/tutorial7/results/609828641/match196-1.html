def get_max(entries):
    max = entries[0]
    for i in range(1, len(entries)):
        if entries[i][2] <max[2]:
            max = entries[i]
            
    entries.remove(max)
    return max

def sort(entries):
    sorted_entries = []
    while len(entries)>0:
        sorted_entries.append(get_max(entries))
    return sorted_entries

    while True:
        
        subject=input('Enter subject name (or press \'enter\' to quit):\n')
        if subject=='':
            break
        number_of_tests=int(input('Enter the number of tests:\n'))
        for test in range(number_of_tests):

            sleep(randint(1,5)) 
            time_1=time()

            time_2=time()
            reaction_time=(time_2-time_1)
            reaction+=reaction_time
            print('Reaction time:',round((reaction_time),6))
        average=round((reaction/number_of_tests),6)
        print('Tests complete.')
        print('Average reaction time:',str(average)+'\n')
