# leader_board.py

def process_line(line):
    parts = line.split(' ')
    name = parts[0]
    trials = int(parts[1])
    average = float(parts[2])
    return (name, trials, average)

def get_max(entries):
    max = entries[0]
    for i in range(1, len(entries)):
        if entries[i][2] <max[2]:
            max = entries[i]
            
    entries.remove(max)
    return max

def sort(entries):
    sorted_entries = []
    while len(entries)>0:
        sorted_entries.append(get_max(entries))
    return sorted_entries

def output(entries):
    print('Subject        | Average Time')


    print('************** | ************')
    for entry in entries:
        print('{:14} | {:.6}'.format(entry[0], entry[2]))
              
def main():
    file_name = input('Enter the name of the file:\n')
    try:
        file_input = open(file_name, 'r')
    except IOError as IO_err:
        print('There is IOError occured.',IO_err)
    else:
        break   
    entries = []
    for line in file_input:
        try:
            entries.append(process_line(line))
        except ValueError as Val_err:
            print('There is ValueError occured.',Val_err)
        except IndexError as Ind_err:
            print('There is IndexError occured.',Ind_err)
    file_input.close()
    
    entries = sort(entries)
    output(entries)
    
main()

from time import*
import random

def main():
    
    file_name = input('Enter the name of the output file:\n')
    f = open(file_name,'w')
    
    while True:
        subject = input("Enter subject name (or press 'enter' to quit):\n")
        if  subject == '':
            break
        
        tests = int(input('Enter the number of tests:\n'))
        s = 0
        for i in range(tests):
            print('Get ready...',flush = True)
            sleep(random.randint(1,5))
            start_time = time()
            n = input("Press 'enter' NOW!")
            end_time = time()
            r = end_time - start_time
            s += r
            average = s / tests
            print('Reaction time: {:6f}'.format(r))
        print('Tests complete.')
        print('Average reaction time: {:6f}\n'.format(average))       
       
    f.close()     
main()
