# leader_board.py

def process_line(line):
    parts = line.split(' ')
    name = parts[0]
    trials = int(parts[1])
    average = float(parts[2])
    return (name, trials, average)

def get_max(entries):
    max = entries[0]
    for i in range(1, len(entries)):
        if entries[i][2] <max[2]:
            max = entries[i]
            
    entries.remove(max)
    return max

def sort(entries):
    sorted_entries = []
    while len(entries)>0:
        sorted_entries.append(get_max(entries))
    return sorted_entries

def output(entries):
    print('Subject        | Average Time')
    print('************** | ************')
    for entry in entries:
        print('{:14} | {:.6}'.format(entry[0], entry[2]))
              
def main():

        try:
            file_name = input('Enter the name of the file:\n')

import time
import random



out_put = input("Enter the name of the output file:\n")
fil = open(out_put,"w")
print('Subject        | Average Time',file=fil)
print('************** | ************',file=fil)
while True:
    subject = input("Enter subject name (or press 'enter' to quit):\n")
    if subject=="":
        break
    number_test = int(input("Enter the number of tests:\n"))
    sum_a=0
    for i in range(number_test):
        print('Get ready...', flush=True)
        time.sleep(random.randint(1,5))
        initial = time.time()
        input("Press 'enter' Now!")
        final = time.time()
        total_time=final-initial
        sum_a+=total_time
        print('{}{:.6}'.format("Reaction time: ",total_time))
    average=sum_a/number_test
    print("Test complete.")
    print('{}{:.6}\n'.format("Average reaction: ",average))
    print('{:14} | {:.6}'.format(subject,average),file=fil)
fil.close()
