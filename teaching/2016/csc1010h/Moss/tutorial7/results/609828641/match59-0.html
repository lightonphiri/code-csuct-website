def process_line(line):
    try:
        parts = line.split(' ')
        #print(parts)
        name = parts[0]
        trials = int(parts[1])
        average = float(parts[2])
    except ValueError as t:
        print('ValueError occured',t)
    else:    
        return (name, trials, average)

def get_max(entries):
    max = entries[0]
    print(entries)
    for i in range(1, len(entries)):


        try:
            if entries[i][2] <max[2]:
                max = entries[i]
        except TypeError as h:
            print('This error occured',h)
    entries.remove(max)
    return max

def sort(entries):
    sorted_entries = []
    while len(entries)>0:
        sorted_entries.append(get_max(entries))
    return sorted_entries
