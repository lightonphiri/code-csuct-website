    new_list=[]
    for word in msg:
        if word in dic:
            new_list.append(str(int(dic[word][0])+1)+"-"+str(int(dic[word][1])+1))
        else:
            new_list.append(word)

def process_line(line):
    parts = line.split(' ')
    try:
        name = parts[0]
        trials = int(parts[1])
        average = float(parts[2])
    except ValueError as VE: 
        print(str(VE)+"\n"+str(line))
    except IndexError as IE:
        print(str(VE)+"\n"+str(line))
    except TypeError as TE:
        print(str(TE)+"\n"+str(line))
    else: 
        print(name, trials, average)
        return (name, trials, average)

def get_max(entries):
    count=0
    try:
        max = entries[0]
        for i in range(1, len(entries)):
            count+=1
            if entries[i][2] <max[2]:
                max = entries[i]
    except TypeError as TE:
        print(str(TE)+"in line:"+str(count)+"\n")
            
    entries.remove(max)
    return max

def sort(entries):
    sorted_entries = []
    while len(entries)>0:
        sorted_entries.append(get_max(entries))
    return sorted_entries

def output(entries):

        print(str(TE)+"\n"+str(entries))
