def main():
    print('***** Long Jump Information System *****')
    print('Please enter the names of competitors. (Press return when done.)')
    x=0
    competitors=[]
    maximum={}
  
    while True :
        x+=1
        competitor=input('Competitor no. '+str(x)+':\n')
        if competitor == '': break 
        competitors.append(competitor)
       
        
    print('Please enter the distances for each competitor.') 
    
    list1=[]
    
    for i in range(len(competitors)):
        name = competitors[i]
        print('Competitor '+str(name)+':')
         
              
        for a in range(1,4):
            attempts=input('Attempt '+str(a)+':\n')
            #if attempts.isalpha(): continue
            list1.append(attempts)

            
        get_max = 0
        for elem in list1:
            if not elem.isalpha():
                
                elem = float(elem)
                if elem>get_max:
                    get_max = elem
                
            
        list1=[]    
        maximum[name]=float(get_max)
        get_max = 0
                
   
    
    
    keep_track=0
    
    winner=[]
    #print('Competitors\t\tLongest Distance')
    print()
    print("{0:<25}{1:<25}".format("Competitor","Longest Distance"))
    for w in maximum:
        if maximum[w]==0:
            
            #print(str(w)+' {0:2}'.format("No jump"))
            print("{0:<25}{1:<27}".format(str(w),"No jump"))
        #print('Competitors\t\tLongest Distance') 
        elif maximum[w]>keep_track:
            keep_track=maximum[w]
            winner.append(w)
            #print(str(w)+' {0:2.2f}'.format(maximum[w]))
            print("{0:<25}{1:<27.2f}".format(str(w),maximum[w]))
            
            
        else:
            print("{0:<25}{1:<27.2f}".format(str(w),maximum[w]))
            #print(str(w)+' {0:2.2f}'.format(maximum[w]))
            
            
    print()   
    print('The winner is',winner[-1],'with a jump of','{0:2.2f}.'.format(float(keep_track)))
        
main()
