def main():
    print('***** Long Jump Information System *****')
    print('Please enter the names of competitors. (Press return when done.)')
    
    
    competitors = []
    a=0
    while True:
        a +=1
        print('Competitor no. '+str(a)+':')
        competitor = input('')
        if competitor =='':break
        else:
            competitors.append(competitor)
            
    distances = []
    print('Please enter the distances for each competitor.')
    b = len(competitors)
    c = 0
    for i in range(b):
        print('Competitor',str(competitors[c])+':')
        c+= 1
        for i in range(3):
            print('Attempt '+str(i+1)+':')
            attempt = input('')
            if attempt != 'foul':
                distance = float(attempt)
                distances.append(distance)
            else:
                zero = 0
                distances.append(zero)
    
    
    print('Competitor               Longest Distance')

    d = 0
    winner = {}
    
    for i in range(b):
        name= competitors[i]
        longest = max(distances[d],distances[d+1],distances[d+2])
        winner[name]= longest
        if longest == 0:
            print('{0:<10}               {1}'.format(name,'No jump'))
        else:
            print('{0:<10}               {1:.2f}'.format(name,longest))
        d += 3
        b +=1
    
    distances.sort()
    
    
    if winner[name]== distances[-1]:
        
        print('The winner is',name,'with a jump of {0:.2f}.'.format(winner[name]))
main()
