def main():
  import random
  print('Guess the mystery 4 letter code. (No letter occurs more than once.)')   
  difficulty=int(input('Choose difficulty level (4 to 26):\n'))
  letters=list('abcdefghijklmnopqrstuvwxyz')[0:difficulty]
  print('The letters in the code are chosen from: {}.'.format(letters))
  secrete_code=[]#empty list
  #The program randomly arranges the letters in 4 and then puts them in 
  #the empty list
  for i in range(4):
    rand_index=random.randint(0, len(letters)-1)
    letter=letters.pop(rand_index)
    secrete_code.append(letter)
    secrete_codestring=''
  for i in secrete_code:
    secrete_codestring+=i
  #print(secrete_codestring)
  
  chances=10
  while True:
    print('You have',chances,'guesses remaining.')
   
    guess=input('Enter your guess:\n')
    if guess==secrete_codestring:
      print('Correct!')
      break
    else:
      position=0
      niks_count=0
      naks_count=0
      naks=''
      niks=''
      for letter in guess:
        if letter ==secrete_code[position]:
          niks_count+=1
        else:
          for j in secrete_code:
            if j == letter:
              naks_count+=1
        position+=1
      if naks_count==1:
        naks='Nak'

      else:
          naks='Naks'
          y='are'

        niks='Nik'

      else:
        niks='Niks'
        s='are'


      print('There '+s+" "+str(niks_count)+" "+niks+' and '+str(naks_count)+" "+naks+'.')
      if chances==1:
        print('Sorry, you lose.\nThe secret number was',secrete_codestring+".")
        break

        chances-=1
