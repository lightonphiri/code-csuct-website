def main():
    print("***** Long Jump Information System *****")
    print("Please enter the names of competitors. (Press return when done.)")
    uluhlu = []
    maxima= []
    dwelling = {}
    Largest = []
    e = 1
    while True:
        jumper = input("Competitor no."+" "+str(e)+":\n")
        if jumper =='':
            break
        else:
            uluhlu.append(jumper) 
        e += 1        
    print("Please enter the distances for each competitor.")
    
    for person in uluhlu:
        print("Competitor",person+":")
        for trial in range(1,4):
            distance = input("Attempt"+" "+str(trial)+":\n")
            if distance.title() != "Foul":
                score = float(distance)
                maxima.append(score)
            else:
                score = 0
                maxima.append(score)
        if max(maxima) != 0:   
            dwelling[person] = max(maxima)
            Largest.append(max(maxima))
            maxima = []
        else:
            dwelling[person] = "No jump"
    
    print("\n{0:<25}{1:<25}".format("Competitor","Longest Distance"))
    
    for contender in uluhlu:
        if type(dwelling[contender]) != str:
            print("{0:<25}{1:<15.2f}".format(contender,dwelling[contender]))
        else:
            print("{0:<25}{1:<25}".format(contender,dwelling[contender]))
        Win = list(dwelling.keys())[list(dwelling.values()).index(max(Largest))]
    print("")                                                               #Getting Key corresponding with the maximum value from the 3 player's best scores
    print("The winner is",Win,"with a jump of","{:.2f}".format(max(Largest))+".")
        

main()
