def main():
    print('{:*^40}'.format(' Long Jump Information System '))
    attempts=3

    # Obtain competitor names
    competitors=[]    
    print('Please enter the names of competitors. (Press return when done.)')
    while True:
        name=input('Competitor no. {}:\n'.format(len(competitors)+1))
        if name=='':
            break
        else:
            competitors.append(name)
    
    # Obtain distances jumped by competitors
    print('Please enter the distances for each competitor.')
    distances={}

    for name in competitors:
        print('Competitor {}:'.format(name))
        
        distances[name]=-1
        for attempt in range(1, attempts+1):        
            distance=input('Attempt {}:\n'.format(attempt))
            if not distance.upper()=='FOUL' and float(distance)>distances[name]:
                    distances[name]=float(distance)
    
    # Print final results
    print()
    print('{0:25}{1:25}'.format('Competitor', 'Longest Distance'))

    
    winner=competitors[0]
    distance=distances[winner]
    if (distance>-1):
        print('{0:25}{1:<15.2f}'.format(winner, distance))
    else:
        print('{0:25}{1:25}'.format(winner, 'No jump'))

    for n in range(1, len(competitors)):
        name=competitors[n]
        distance=distances[name]
        if (distance>-1):
            print('{0:25}{1:<15.2f}'.format(name, distance))
        else:
            print('{0:25}{1:25}'.format(name, 'No jump'))
            
        if distance>distances[winner]:
            winner=name
    
    print()
    print('The winner is {0} with a jump of {1:.2f}.'.format(winner, distances[winner]))
    
     
main()

import random

def main():
    print('Guess the mystery 4 letter code. (No letter occurs more than once.)')
    # Generate a secret code
    difficulty=int(input('Choose difficulty level (4 to 26):\n'))
    letters=list('abcdefghijklmnopqrstuvwxyz')[0: difficulty]
    print('The letters in the code are chosen from: {}.'.format(letters))
    secret_code=[]


    for i in range(4):
        rand_index=random.randint(0, len(letters)-1)
        letter=letters.pop(rand_index)
        secret_code.append(letter)

    #print(secret_code)
    
    guesses=10
    while guesses>0:
        print('You have {0} guesses remaining.'.format(guesses))
        guess=input("Enter your guess:\n")
        if (len(guess)!=4 or not guess.isalpha()):
            print('Your guess should be a 4 letter sequence.')
        else:
            niks=0
            naks=0
            for i in range(4):
                if guess[i]==secret_code[i]:
                    niks+=1
                elif guess[i] in secret_code:
                    naks+=1
              
            
            if (niks==4):
                print('Correct!')
                break
            else:
                guesses-=1
                if niks==1:
                    nik_label='There is 1 Nik and '
                else:
                    nik_label='There are '+str(niks)+' Niks and '
                if naks==1:
                    nak_label='1 Nak.'
                else:
                    nak_label=str(naks)+' Naks.'
                    
                print(nik_label+nak_label)

    if guesses==0:
        print('Sorry, you lose.')
        reveal='The secret number was ';
