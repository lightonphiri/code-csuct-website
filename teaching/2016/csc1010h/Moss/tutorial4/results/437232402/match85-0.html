wheel=2

def main():
    begin_poly()
    speed(8)
    fillcolor("magenta")
    shape("circle")

        pensize(20)
        rt(180);pu();fd(190);pd()
        pencolor("yellow");circle(140,360);pu();home();rt(180);pu;fd(-150);pd();
        pencolor("green");circle(140,360);pu();home();lt(180);fd(500);lt(90) ;fd(-10);pd()

        pencolor("black");circle(140,360);pu();home();fd(460);lt(90);fd(10);pd();
