import random
def main():
    print("Guess the mystery 4 letter code. (No letter occurs more than once.)")
    difficulty=int(input('Choose difficulty level (4 to 26):\n'))
    letters=list('abcdefghijklmnopqrstuvwxyz')[0: difficulty]
    print('The letters in the code are chosen from: {}.'.format(letters))
    secret_code=[]


    for i in range(4):
        rand_index=random.randint(0, len(letters)-1)
        letter=letters.pop(rand_index)
        secret_code.append(letter)
    counter = 10
    while counter:
        print("You have",counter,"guesses remaining.")
        guess = input("Enter your guess:\n")
        niks = 0
        naks = 0
        for k in range(4):
            if secret_code[k]==guess[k]:
                niks += 1
            elif guess[k] in secret_code:
                naks += 1
        if niks == 4:
            print("Correct!")
            break
        a = "are"
        b = "Niks"
        c = "Naks."
        if niks == 1:
            a = "is"
            b = "Nik"
        if naks == 1:
            c = "Nak."
        print("There",a,niks,b,"and",naks,c)
        counter -= 1
    if not counter:
        print("Sorry, you lose.")
        print("The secret number was ",end="")
        for i in secret_code:
            print(i,end="")
        print(".")
if __name__=="__main__":
    main()
