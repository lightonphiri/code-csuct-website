print('***** Long Jump Information System *****')
print('Please enter the names of competitors. (Press enter when done.)')
competitors=[]
n=1

while True:
    c=input('Competitor no. {}:\n'.format(n))
    n+=1
    if len(c)>2:
        competitors.append(c)
    else:
        break

distance={}
print('Please enter the distances for each competitor.')
for com in competitors:
    print('Competitor {}:'.format(com))
    a=input('Attempt 1:\n')
    b=input('Attempt 2:\n')
    c=input('Attempt 3:\n')
    distance={}
    if b>a and b>c:
        distance=b
    elif a>b and a>c:
        distance=a
    elif c>b and c>a:
        distance=c
    else:
        distance="foul"
    
    m='Competitor'
    z='Longest Distance'
    print('\n{0:20}{1:<20}'.format(m,z))
    print('{0:20}{1:<20}'.format(competitors[0],distance)) 
    
print('\nThe winner is {} with a jump of {}'.format(competitors[0],distance))

import random

print('Guess the mystery 4 letter code. (No letter occurs more than once.)')
difficulty=int(input('Choose difficulty level (4 to 26):\n'))     
letters=list('abcdefghijklmnopqrstuvwxyz')[0: difficulty]     
print('The letters in the code are chosen from: {}.'.format(letters))     
secret_code=[]     
for i in range(4):         
    rand_index=random.randint(0, len(letters)-1)         
    letter=letters.pop(rand_index)         
    secret_code.append(letter) 
    
nik=0
nak=0
for g in range(10,0,-1):
    print('You have {} guesses remaining.'.format(g))
    guess=input('Enter your guess:\n')
    for n in range(4):
        if guess[n]==secret_code[n]:
            nik+=1
        else:
            continue
    for k in range(4-nik):
        for l in guess:
            if l==secret_code[k]:
                nak+=1
            else:
                continue
    
    if nik==4:
        print('Correct!')
    else:
        print('There are {} Niks and {} Naks.'.format(nik,nak))
    nik=0
    nak=0
