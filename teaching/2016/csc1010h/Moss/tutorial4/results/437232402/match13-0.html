def main():
    print('***** Long Jump Information System *****')
    print('Please enter the names of competitors. (Press return when done.)')
    no = 0
    
    names = []
    


    while True:
        no +=1
        print('Competitor no. '+str(no)+':')
        name = input('')
        if name =='':break
        else:
            names.append(name)
            
    distances = []
    print('Please enter the distances for each competitor.')
    num = len(names)
    name_no = 0
    for i in range(num):
        print('Competitor '+names[name_no]+':')
        name_no += 1
        for i in range(3):
            print('Attempt '+str(i+1)+':')
            attempt = input('')
            if attempt != 'foul':
                distance = float(attempt)
                distances.append(distance)
            else:
                zero = 0
                distances.append(zero)
    
    
    print('Competitor               Longest Distance')
    numb =num


    n = 0
    winner = {}
    
    for i in range(numb):
        competitor= names[i]
        longest = max(distances[n],distances[n+1],distances[n+2])
        winner[competitor]= longest
        if longest == 0:
            print('{0}{1:>30}'.format(competitor,'No jump'))
        else:
            print('{0:<10}               {1:.2f}'.format(competitor,longest))
        n += 3
        num +=1
    
    distances.sort()
    print('')
    if winner[competitor]== distances[-1]:
        print('The winner is',competitor,'with a jump of',str(winner[competitor])+'.')
        
    
       
main()
