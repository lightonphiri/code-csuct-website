def main():
    comp = []
    numbers = []
    dictionary = {}
    a = 0
    print('***** Long Jump Information System *****')
    print('Please enter the names of competitors. (Press return when done.)')
    
    while True:
        a += 1
        competitors = input('Competitor no. ' + str(a) + ':\n')
        if competitors =='':break
        comp.append(competitors)
    print('Please enter the distances for each competitor.')
  
    for i in comp:
        print('Competitor',i+":") 
        
        for j in range(1,4):
            att = input('Attempt '+str(j)+':\n') 
            
            if att.title() == 'Foul' :pass
            
            else:     
                numbers.append(eval(att))
                longest = max(numbers)
                dictionary[i] = longest
                
        if numbers == []:
            dictionary[i] = "No jump"
        numbers.clear()    
    print()            
    print('Competitor {0:>30}'.format('Longest Distance'))
    
    for s in comp:
        if type(dictionary[s]) !=str:
            print('{0:<25}{1:.2f}'.format(s,(float(dictionary[s]))))
        else:
            print('{0:<25}{1}'.format(s,dictionary[s]))
        
    for z in  dictionary:
        numbers.append(dictionary[z])
    w = max(numbers)
    for e in dictionary:
        if dictionary[e] == w:
            print()
            print('The winner is {0} with a jump of {1:.2f}.'.format(e,float(w)))     
            
main()

    letters = list('abcdefghijklmnopqrstuvwxyz')[0: difficulty]
    print('The letters in the code are chosen from: {}.'.format(letters))
    secret_code=[]
    secret_codes =''
    


    for i in range(4):
        
        rand_index = random.randint(0, len(letters)-1)
        letter = letters.pop(rand_index)
        secret_code.append(letter)
    for d in secret_code:
        secret_codes += d         
        
    
    for b in range(position):
        print('You have',position,'guesses remaining.')
        guess = input('Enter your guess:\n')
