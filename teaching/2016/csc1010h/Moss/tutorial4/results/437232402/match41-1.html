def main():
    print('***** Long Jump Information System *****')
    print('Please enter the names of competitors. (Press return when done.)')
    competitors={} # empty dict to store the scores for the competitors..
    list_names=[] # empty list to store the competitors
    a=1
    while True:
        competitor_names=input('Competitor no.'+' '+str(a)+':\n') # promts the user for a times
        a+=1
        if competitor_names=="": # breaking the loop
            break 
        list_names.append(competitor_names) # appends the names of competitors
    print('Please enter the distances for each competitor.')
    for t in (list_names): # lopps over the competitors in a list
        attempts = []    # empty to append the attempts of each competitor
        print('Competitor'+' '+t+':')
        for k in range(1,4): # loops over the no. of attempts
            attempt = input("Attempt "+str(k)+":\n") # promts the user for the attempts
            if attempt.isalpha(): # check if the attempt is alpha
                attempt=str(0) #convert the attempt to a string if alpha
            attempts.append(attempt) #  append it to the empty list in each loop
        b=max(attempts) # finds the maximum in the list
        competitors[t]=b # assigns max values to each competitor
    high= max(competitors.values()) # fins the abs max 
    print()
    print("Competitor               Longest Distance         ")
    for l in range(len(list_names)): # loops over the length of the list containing the competitors
        if competitors[list_names[l]] == "0": #check if the max is a foul then prints out No jump
            print('{0:25}{1:<25}'.format(list_names[l],"No jump"))
        else:
            print('{0:25}{1:<15.2f}'.format(list_names[l],float(competitors[list_names[l]]))) # print out the max values for each person
    for i in competitors: #loops over the dict
        if competitors[i] == high: # checks if competitor has the abs max value
            print()
            print('The winner is {0} with a jump of {1:.2f}.'.format(i,float(high))) # prints the winner
