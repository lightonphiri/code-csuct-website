def main():
    comp_no = 0 # initialising competitor no.
    names = []  # Empty list to store Competitor's names
    distances = [] # Empty list to store Competitor's distances
    name_max_distance = {} # Empty dictionary to store Competitor's longest distance and their name
    
    print('{:*^40}'.format(' Long Jump Information System ')) # print out Heading of the program
    print('Please enter the names of competitors. (Press return when done.)') 
    while True: # Continually prompts the user to enter Competitor's names
        comp_no += 1
        name = input('Competitor no. ' + str(comp_no) + ':\n')
        if name == '':break # if the user press <Enter> the while loop terminates
        names.append(name) # Storing Competitor's names to the list called names
    
    print('Please enter the distances for each competitor.') 
    for name in names: # For loop iterating within the list names 
        print('Competitor '+ name +':')
        maximum = 'No jump' # variable maximum is assigned to the string No jump    
        name_max_distance[name] = maximum # adding Competitor's name and its Longest distance       
        for attempts in range(3): # For loop to promt the user to enter Distances three times
            distance = input('Attempt '+str(attempts + 1)+ ':\n')
            if distance == 'foul':
                pass
            else:
                dist_cast = eval(distance)
                distances.append(dist_cast)
                maximum = max(distances)
                name_max_distance[name] = maximum # adding Competitor's name and its Longest distance
                     
        distances.clear() # cleaning the list named distances
    print()    
    print('{0:25}{1:25}'.format('Competitor','Longest Distance')) 
    for na_mes in names: # printing out a table of names and longest distances
        if type(name_max_distance[na_mes]) == str:
            print('{0:25}{1:15}'.format(na_mes,name_max_distance[na_mes]))
        else:
            print('{0:25}{1:<15.2f}'.format(na_mes,name_max_distance[na_mes]))        
    for key in name_max_distance:
        if type(name_max_distance[key]) == int or type(name_max_distance[key]) == float:
            distances.append(name_max_distance[key])
    print()
    winner = max(distances)
    for key in name_max_distance: # Telling who the winner is and by what distance
            if name_max_distance[key] == winner:
                print('The winner is '+ key + ' with a jump of {:2.2f}.'.format(winner))
