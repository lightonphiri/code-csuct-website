def main():
    print("***** Long Jump Information System *****")
    print("Please enter the names of competitors. (Press return when done.)")
    num=0
    names,distances,max_dist,highest_score=[],[],[],[]
    scores={}
    while True:
        num+=1
        competitors=input("Competitor no. "+str(num)+":\n")
        if competitors=="":
            break
        else:
            names.append(competitors)
    print("Please enter the distances for each competitor.")
    for competitor in names:
        print("Competitor "+competitor+":")
        for attempt in range(3):
            Attmpts=input("Attempt "+str(attempt+1)+":\n")
            if Attmpts.isalpha():
                scores[competitor]="No jump"
            else:
                distances.append(Attmpts)
                scores[competitor]=max(distances)
                
                
        distances.clear()
    print()
    print("Competitor               Longest Distance")
    for contender in names:
        if scores[contender]=="No jump":
            print("{0:<25}{1:<5}".format(contender,scores[contender]))
        else:
            print("{0:<25}{1:<5.2f}".format(contender,float(scores[contender])))
            
        max_dist.append(scores[contender])
    print()
    for highest in max_dist:
        if highest!="No jump" :
            highest_score.append(highest)        
    for winner in names:
        if max(highest_score)==scores[winner]:
            max_score=float(max(highest_score))
            
            print("The winner is",winner,"with a jump of {0:2.2f}.".format(max_score))

    secret_code = []

    niks = []

    letters = alphabets[0:difficulty]
    


    for i in range(4):

        code = letters.pop(rand_index)
        secret_code.append(code)
        
    guesses = 10

        guesses -= 1

                    niks.append(i)

        elif nikz !=1 and nakz == 1:
            print("There are {} Niks and {} Nak.".format(nikz, nakz))
