def main():
    print('***** Long Jump Information System *****')
    print('Please enter the names of competitors. (Press return when done.)')
    comp_number=1
    competitors=[]
    while True:
        competitor=input('Competitor no. '+ str(comp_number)+':\n')
        if competitor!='':
            competitors.append(competitor)
            comp_number+=1
        else: break
    print('Please enter the distances for each competitor.')
    highest=[]

    for compt in competitors:
        Foul = 0
        n = 0
        print('Competitor '+compt+':')
        for attempt in range(1,4):
            Att=input('Attempt '+str(attempt)+':\n')
            if Att=='foul':
                Foul +=1
                if Foul==3:
                    high = 'No jump'


            elif Att!='foul':

                n+=1
                if n ==1:
                    high = float(Att)
                else:
                    if float(Att) > high:
                        high = float(Att)
        
        highest.append(high)
           
    print('\n{0:<25}'.format('Competitor')+'Longest Distance')
    second=0
    for comp in competitors:
        positions=highest[second]
        if positions=='No jump':
            print('\n{0:<25}{1:}'.format(comp,positions))
        else:
            print('{0:<25}{1:.2f}'.format(comp,positions))
        second+=1
    yaay = 0
    numbers=[]
    for top in highest:
        if top!='No jump':
            numbers.append(top)
    biggest=max(numbers)    
    winner=(competitors[(highest.index(biggest))])
    
    
    print("\nThe winner is {0:} with a jump of {1:.2f}.".format(winner,biggest))

import random
def main():
    letters=list('abcdefghijklmnopqrstuvwxyz')
    print("Guess the mystery 4 letter code. (No letter occurs more than once.)")
    difficulty=int(input('Choose difficulty level (4 to 26):\n'))
    updated_letters=letters[0:difficulty]
    print('The letters in the code are chosen from: {}.'.format(updated_letters))
    secret_code=[]
    for i in range(4):
        rand_index=random.randint(0, len(updated_letters)-1)
        letter=updated_letters.pop(rand_index)
        secret_code.append(letter)
    

    
    left=10
    while True:
        
        print('You have '+str(left)+' guesses remaining.')
        guess=input('Enter your guess:\n')
        variables=list(guess)
        niks=0
        naks=0 
        if secret_code==variables:
            print('Correct!')
            break
        else:
            for variable in variables:
                position=variables.index(variable)
                if variables[position]==secret_code[position]:
                    niks+=1
                else:
                    if variable in secret_code:
                        naks+=1
                        
        if naks==1:
            if niks==1:
                print('There is '+str(niks)+' Nik and '+str(naks)+' Nak.')
            else:
                print('There are '+str(niks)+' Niks and '+str(naks)+' Nak.')
        else:
            if niks==1:
                print('There is '+str(niks)+' Nik and '+str(naks)+' Naks.')
            else:
                print('There are '+str(niks)+' Niks and '+str(naks)+' Naks.')
     
        left-=1
        
        secretcode=''
        if left<1:
            print('Sorry, you lose.')
            for item in secret_code:
                secretcode+=item
                
            print("The secret number was",secretcode+".")
            break
