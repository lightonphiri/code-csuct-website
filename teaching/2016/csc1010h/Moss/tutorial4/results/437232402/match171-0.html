        a+=1

        competitors.append(com)
        if com=='':break

    print('Please enter the distances for each competitor.')
    
    for name in competitors:
        
        attempts=[]
        print('Competitor '+name+':')
        for i in range(1,4):
           
            att = input('Attempt '+str(i)+':\n')
            a = att[0]
            if a.isnumeric():
                attempts.append(float(att))
                
                      
        if attempts==[]:
            dis = 'No jump'
            Long.append(dis)
           
        else:
            dis = '{:.2f}'.format(max(attempts))
            Long.append(dis)

        dic[dis] = name
    print()

def main():

    print('The letters in the code are chosen from: {}.'.format(letters))
    secret_code=[]
    for i in range(difficulty):
        rand_index=random.randint(0, len(letters)-1)
        letter=letters.pop(rand_index)
        secret_code.append(letter)
    # To check things are working, uncomment this print line to see the

                lst.append('Niks')

                lst.append('Naks')
