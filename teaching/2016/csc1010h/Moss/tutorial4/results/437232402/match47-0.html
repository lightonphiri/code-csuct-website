from collections import OrderedDict

dictionary = {}
names = OrderedDict(dictionary)
maximum_distance = [0]
maximum_jumper = []
competitor_number = 1


print("***** Long Jump Information System *****")
print("Please enter the names of competitors. (Press return when done.)")

while True:
    competitors = input("Competitor no. {}:\n".format(competitor_number))
    if competitors == '':
        break
    names[competitors]=0
    competitor_number += 1 
    
print("Please enter the distances for each competitor.")
  
for distance in names:
    print('Competitor {}:'.format(distance))
    for attempt in range(1,4):
        answer = input('Attempt {}: \n'.format(attempt))
        if answer == 'foul':
            continue
        elif float(answer) > names[distance]:
            names[distance] = float(answer)  
    
print("\n{0:25}Longest Distance".format("Competitor"))
for s in names:
    if names[s] == 0:
        print("{0:25}No jump".format(s))
        continue
    print("{0:25}{1:.2f}".format(s, names[s]))
    if names[s] > maximum_distance[0]:
        maximum_distance[0] = names[s]
        maximum_jumper = s
        
print("\nThe winner is {} with a jump of {:.2f}.".format(maximum_jumper, maximum_distance[0]))

    secret_code = []

        rand_index = random.randint(0,len(letters)-1)
        letter = letters.pop(rand_index)
        secret_code.append(letter)

        print('You have {} guesses remaining.'.format(guess))
        answer = input('Enter your guess:\n')

            print('Correct!')

                naks +=1
