import random

def main():
    print("Guess the mystery 4 letter code. (No letter occurs more than once.)") 
    difficulty=int(input('Choose difficulty level (4 to 26):\n')) 
    letters=list('abcdefghijklmnopqrstuvwxyz')[0: difficulty] 
    print('The letters in the code are chosen from: {}.'.format(letters))
    print("You have 10 guesses remaining.")
    secret_code=[]
    for i in range(4):
        rand_index=random.randint(0, len(letters)-1) 
        letter=letters.pop(rand_index) 
        secret_code.append(letter)
        
    guess = input("Enter your guess:\n")
    empty = []
    for i in guess:
        empty.append(i)
    count = 0
    
    while empty != secret_code:
        Niks = 0 
        Naks = 0
        n = 0
        for i in empty:
            if i in secret_code :
                if empty[n] == secret_code[n]:
                    Niks += 1
                else:
                    Naks += 1
            n += 1
        count += 1
        if Naks>1 or Naks==0:
            if Niks>1 or Niks==0:
                print("There are", Niks,"Niks and", Naks,"Naks.")
            else:
                print("There is", Niks,"Nik and", Naks,"Naks.")
        else:
            print("There are", Niks,"Niks and", Naks,"Nak.")
        if count == 10:break
        
        print("You have", 10 - count,"guesses remaining.")
        empty = []
        guess = input("Enter your guess:\n")
        for i in guess:
            empty.append(i)
             
    if empty == secret_code:
        print("Correct!")
    else:
        code=""
        for i in secret_code:
            code+=i
        print("Sorry, you lose.\nThe secret number was",code+".")
        
if __name__=="__main__":
    main()
