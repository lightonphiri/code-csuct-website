def main():
    print("***** Long Jump Information System *****")
    print("Please enter the names of competitors. (Press return when done.)")#basics assking the user for input
    competitor = []  #creating a list for the competitors
    name_no = 1
    while True:
        name = input("Competitor no. {}:\n".format(name_no))
        if name == "":break #terminate the while loop
        name_no += 1
        competitor.append(name)
        
       
    print("Please enter the distances for each competitor.")
    maxmum_tries = 0
    maxmum_numbers =[]
    overall_winner = {}
    for i in competitor:
        print("Competitor",i+":")
        for x in range(3):
            trials = (input("Attempt {}:\n".format(x+1)))
            if trials == "":break
            trials = trials.lower()
            if trials == "foul":
                maxmum_numbers.append(0)
                overall_winner[i] =max( maxmum_numbers)
            else:
                trials = eval(trials)
                maxmum_numbers.append(trials)          
                overall_winner[i] = max(maxmum_numbers)
                
                if trials > maxmum_tries:
                    maxmum_tries = trials
        maxmum_numbers =[]
    y=0
    z=""
    print()
    print("{0:<25}{1:<15}".format("Competitor","Longest Distance         "))
    for i in range(len(competitor)):
        if overall_winner[competitor[i]]== 0:
            print("{0:<25}{1:<15}".format(competitor[i],"No jump"))
        else:
            print("{0:<25}{1:<15.2f}".format(competitor[i],float(overall_winner[competitor[i]])))
            y+=1
        if overall_winner[competitor[i]]==maxmum_tries:
            z=competitor[i]
    print("\nThe winner is {0} with a jump of {1:.2f}.".format(z,float(maxmum_tries))) 
    
main()

import random

def main():
    print("Guess the mystery 4 letter code.(No letter occurs more than once.)")
    
    no_of_attempts = 10
    difficulty=int(input('Choose difficulty level (4 to 26):\n'))
    letters=list('abcdefghijklmnopqrstuvwxyz')[0: difficulty]
    print('The letters in the code are chosen from: {}.'.format(letters))
    secret_code=[]
    for i in range(4):
        rand_index=random.randint(0, len(letters)-1)
        letter=letters.pop(rand_index)
        secret_code.append(letter)  
        #print(secret_code)  this line was making my problem not work it was a bug


    niks = 0
    naks = 0

        no_guesses = input("You have {} guesses remaining.\n Enter your guess:\n".format(no_of_attempts))
