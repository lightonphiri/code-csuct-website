def main():
    print("***** Long Jump Information System *****")
    print("Please enter the names of competitors. (Press return when done.)")
    competitors = []
    count = 0
    position = -1
    while True:
        count+=1
        competitor = input("Competitor no. %d:\n" %count)
        if competitor == "": break
        competitors.append(competitor)
    print("Please enter the distances for each competitor.")
    a = len(competitors) 
    d={}
    for i in range (a):
        position +=1 
        print("Competitor",competitors[position]+":")
        l=[]
        for attempts in range (3):
            attempts = input("Attempt %d: \n" %(attempts+1))
            if attempts.isalpha():
                attempts = '0.00'
            attempts = eval(attempts)
            l.append(attempts)
        d[competitors[position]]=max(l)
    print("\nCompetitor               Longest Distance ")
    q = []
    g={}
    for e in d:
        g[d[e]]=e
        q.append(float(d[e]))
    for z in competitors:
        if d[z] == 0:
            d[z]='No jump'
            print('{0:25}{1:}'.format(z,d[z]))
        else:
            print("{0:25}{1:.2f}".format(z,d[z]))
    winner = max(q)
    print("\nThe winner is",g[max(q)],"with a jump of {:.2f}.".format(winner))
