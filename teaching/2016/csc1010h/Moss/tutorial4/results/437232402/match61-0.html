competitors = []
data = {}

print('***** Long Jump Information System *****\nPlease enter the names of competitors. (Press return when done.)')

#get a series of names
print('Competitor no.',n,end = ':\n')
x = input()

while x != '':
    n = n + 1
    competitors.append(x)

    print('Competitor no.',n,end = ':\n')
    x = input()
    
#get distances for each name and store maximum
print('Please enter the distances for each competitor.')
for N in range(len(competitors)):
    print('Competitor',competitors[N],':')

    a = input('Attempt 1:\n')
    b = input('Attempt 2:\n')
    c = input('Attempt 3:\n')

    maximum = max (max (a, b), c)

    if a and b and c == 'foul':
        maximum = 'No jump'
    elif a == 'foul':
        maximum = max(b, c)
    elif b == 'foul':
        maximum = max(a, c)
    elif c == 'foul':
        maximum = max(a, b)
    elif a and b == 'foul':
        maximum = c
    elif b and c == 'foul':
        maximum = a
    elif a and c == 'foul':
        maximum = b
    data[(str(competitors[N]))] = (maximum)

#print out data in correct format
print('Competitor','Longest Distance',sep = ' ' * 15)

for key in data:
    print('{0:<24}'.format(key),(data[key]))
