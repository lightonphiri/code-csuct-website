import turtle
def main():
    turtle.bgcolor('black')
    turtle.pencolor('white')
    turtle.circle(100)
    turtle.tilt(30)
    turtle.left(90)
    turtle.penup()
    turtle.forward(100)
    turtle.pendown()
    turtle.fillcolor('green')
    turtle.begin_fill()
    for i in range(20):
        turtle.forward(i * 10)
        turtle.right(144)
    turtle.end_fill()
    turtle.right(180)
    turtle.penup()
