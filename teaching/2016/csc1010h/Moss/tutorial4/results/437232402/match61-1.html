import turtle
turtle.bgcolor("grey")
def leaf(t, r, corner):

    for i in range(2):
        t.circle(r,corner)
        t.left(180-corner)

def plant(t, n, r, corner):

    for i in range(n):
        leaf(t, r, corner)
        t.left(360.0/n)

def motion(t, length):

    t.pu()
    t.fd(length)
    t.pd()

turtle.hideturtle()
a = turtle.Pen()

a.speed(0)

a.color("red")
motion(a, -100)
