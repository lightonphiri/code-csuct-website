def main():
    min_inches = int(input('Enter the minimum number of inches (not less than 0):\n'))
    max_inches = int(input('Enter the maximum number of inches (not greater than 11):\n'))
    
    print('Inches:', end=' ')
    for i in range(min_inches,max_inches+1):
        if len(str(i))==1:
            print('  ',i, end=' ')
        else:
            print(" ",i, end=' ')
    
    print('\n'+'Metres:', end=' ')
    for i in range(min_inches,max_inches+1):
        m = i/3.2804
        n = str('{0:.2f}'.format(m*0.08333))
        print(n, end=' ')
        
main()

def main():
    time = input('Enter a time ([h]h:mm [am|pm]):\n')
    
    if time.endswith('am') or time.endswith('pm'):
        hour = time[0:1]
        hour_1 = time[0:2]
        t = len(time)
        minu = time[-6:-3:1]
        
        if time.endswith('am') and t==7:
            print('0'+str(hour)+str(minu))
        elif time.endswith('am') and time.startswith('12'):
            print('00'+str(minu))
        elif time.endswith('am') and t==8:
            print(hour_1+minu)
            
        elif time.endswith('pm') and t==7:
            print(str(int(hour)+12)+str(minu))
        elif time.endswith('pm') and time.startswith('12'):
            print('12'+str(minu))    
        elif time.endswith('pm') and t==8:
            print(str(int(hour_1)+12)+str(minu))
    else:
        hours = time[:2]
        minutes = time[2:]
        hour_2 = time[1]
        
        if int(hours)==00:
            print('12'+minutes,'am')
        elif hours<'10':
            print(hour_2+minutes,'am')
        elif 10<=int(hours)<=11:
            print(hours+minutes,'am')
        elif int(hours)==12:
            print('12'+minutes,'pm')  
        elif hours>'12':
            print(str(int(hours)-12)+minutes,'pm')
