    hours = time[:2]
    minutes = time[3:5]
    if not hours.isdigit():
        hours = time[:1]
        minutes = time[2:4]
        
    hours = int(hours)
    if hours >= 12:
        hours = hours - 12

        hours = hours
        
    if time[5:7] == "pm" or time[6:8] == "pm":
        hours = hours + 12
        
    hours = str(hours)
    
    if len(hours) == 1:
        print("0",hours,":",minutes,sep="")
    else:
        print(hours,":",minutes,sep="")
        
else:
    hours = time[0:2]
    minutes = time[3:5]
    if minutes[:1] == "00":
        minutes = minutes[1:]
    
    hours = int(hours)
    if hours > 12:
        hours = hours-12
        postfix = "pm"
    elif hours < 12:
        hours = hours
        postfix = "am"
    elif hours == 12:
        postfix = "pm"
    elif len(hours) == 1:
        print("0",hours,":",minutes,sep="")
    else:
        print(hours,":",minutes,sep="")
    print(hours,":",minutes," ",postfix,sep="")

gvm = eval(input("What is the gross vehicle mass (in kg)?\n"))
if gvm <= 3500:
    gvm = "B"
elif 3500<gvm<=16000:
    gvm = "C1"
else:
    gvm = "C"
    
trailer = input("Does the vehicle have a trailer?\n")
if trailer == "y":
    trailer_weight = eval(input("What is gross vehicle mass of the trailer (in kg)?\n"))
    if trailer_weight > 750:
        trailer_weight = "E"
        print("This vehicle is class ",trailer_weight,gvm,".",sep="")
    elif trailer_weight <= 750:
        trailer_weight=""
        print("This vehicle is class ",trailer_weight,gvm,".",sep="")
elif trailer == "n":
    articulated = input("Is the vehicle articulated?\n")
    if articulated == "y":
        articulated = "E"
        print("This vehicle is class ",articulated,gvm,".",sep="")
    elif articulated == "n":
        articulated = ""
        print("This vehicle is class ",articulated,gvm,".",sep="")
