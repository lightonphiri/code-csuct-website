def main():
    #input    
    time_old = input("Enter a time ([h]h:mm [am|pm]): \n")
    
    #12 hour time to 24 hour time
    if time_old[-1] == "m":
        #if midnight
        if time_old[0:2] == "12" and time_old[6:] == "pm":
            print("00" + time_old[2:6])    
            
        #if in morning
        elif (time_old[1] != "0" or time_old[1] != "1" or time_old[1] != "2") and time_old[5:] == "am":
            print("0" + time_old[0:4])
            
        elif (time_old[1] == "0" or time_old[1] == "1" or time_old[1] == "2") and time_old[6:] == "am":
            print(time_old[0:5])
            
            #if in afternoon / evening
        elif (time_old[1] != "0" or time_old[1] != "1" or time_old[1] != "2") and time_old[5:] == "pm":
            time_new_1 = int(time_old[0:1]) + 12
            time_new_string_1 = str(time_new_1)
            print(time_new_string_1 + time_old[1:4])
            
        elif (time_old[1] == "0" or time_old[1] == "1" or time_old[1] == "2") and time_old[6:] == "pm":
            time_new_2 = int(time_old[0:2]) + 12
            time_new_string_2 = str(time_new_2)
            print(time_new_string_2 + time_old[2:5])        
    
    #24 hour time to 12 hour time
    elif time_old[-1] != "m":            
        time_old_integer = int(time_old[0:2])
            
        #calculations
        time_new_string = str(time_old_integer - 12)
        
        #process
        #if midnight
        if time_old[0:2] == "00":
            print("12" + time_old[2:] + " " + "am")
            
        #if in morning    
        elif time_old_integer < 10:
            print(time_old[1:] + " " + "am")
            
        elif 10 <= time_old_integer <= 12:
            print(time_old + " " + "am")
            
        #if in afternoon / evening
        elif 13 <= time_old_integer <= 24:
            print(time_new_string + time_old[2:] + " " + "pm")
