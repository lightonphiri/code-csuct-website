def main():
    s = int(input("Enter the minimum number of feet (not less than 0):\n"))
    e = int(input("Enter the maximum number of feet (not more than 99):\n"))
    for i in range(s,e+1):
        m = (i*0.3048)
        print("{0:>3}{1}{2:^4}{3:<4.2f}m".format(i,"'","|",m))
