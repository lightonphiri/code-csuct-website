petals = int(input("Enter the number of petals (must be an even number): "))
stem = int(input("Enter the legnth of the stem: "))
petal_type = input("Enter the petal type (triangle, square, or pentagon): ")
length_side = int(input("Enter the length of side:"))
stem_color = input("Enter the stem colour: ")
fir_color = input("Enter the first colour: ")
seco_color = input("Enter the second colour: ")
if petal_type == "triangle":
    sid = 3
elif petal_type == "square":
    sid = 4
else:
    sid = 5
def draw_shape(size=100, sides=3):
    angle= (180 - 360/sides)
    right(angle/2)
    for i in range(sides):
        forward(size)
        if i<sides-1:
            left(180- angle)
    right(angle/2)
def draw_stem(no_of_stems, s_stem, n_sides, n_size, c_stem, n_c1, n_c2):
    angle = 180- 360/no_of_stems
    r_color = color()
    color(c_stem)
    for i in range(no_of_stems):
        forward(s_stem)
        color(n_c1 if i%2 else n_c2)
        draw_shape(n_size, n_sides)
        color(c_stem)
        forward(s_stem)
        right(angle)
    color(*r_color)
draw_stem(petals,stem,sid,length_side, stem_color, fir_color,seco_color)
