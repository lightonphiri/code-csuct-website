def main():

    maximum_inches = int(input("Enter the maximum number of inches (not greater than 11):\n"))
    
    if minimum_inches >= 0 and maximum_inches <= 11:
        print("Inches:", end = " ")
        for inches in range(minimum_inches, maximum_inches+1):
            print("{:4}".format(inches), end = " ")
        print("\nMetres:", end = " ")
        for metres in range (minimum_inches, maximum_inches+1):
            inches_to_meters = (metres * 0.0254)
            print("{:4.2f}".format(inches_to_meters), end = " ")

# Converting 12 hour to 24 hour and 24 hour to 12 hour
# Craig Jarvis
# 11 May 2016

def main():
    time = input("Enter a time ([h]h:mm [am|pm]):\n")
    if time[-1] == "m": # checking for 12 hour time
        am_pm = time[-1:-3:-1]
        colon = time.find(":")
        if am_pm == "ma": # checking reversed index of am
            if time[:2] == "12": # checking for 12am
                print("00:{}".format(time[colon+1 : colon+3]))
            elif time[0] == "1" and time[2] == ":": # checking for 10am and 11am
                print("{}:{}".format(time[:2], time[colon+1 : colon+3]))
            else: # checking for 0am-9am
                print("0{}:{}".format(time[:1], time[colon+1 : colon+3]))
        elif am_pm == "mp": # checking reversed index of pm
            if time[:2] == "12": # checking for 12pm
                print("12:{}".format(time[colon+1 : colon+3]))
            elif time[0] == "1" and time[2] == ":": # checking for 10pm and 11pm
                print("{}:{}".format(int(time[:2])+12, time[colon+1 : colon+3]))
            else: # checking for 0pm-9pm
                print("{}:{}".format(int(time[:1])+12, time[colon+1 : colon+3])) 
    else: # checking for 24 hour time
        first = int(time[:2]) # firsts = first 2 digits of time
        
        if 0 <= first <= 11:
            if first == 00:
                print("12:{} am".format(time[3:]))
            else:
                print("{}:{} am".format(first, time[3:]))
        
        elif 12 <= first <= 23:
            if first == 12:
                print("12:{} pm".format(time[3:]))
            else:
                print("{}:{} pm".format(first-12, time[3:]))
