import turtle
def main():
    turtle.speed(1)
    a=int(input('Enter the number of petals (must be an even number): '))
    stem_length=int(input('Enter the length of stem: '))
    petal_type=input('Enter the petal type (triangle, square, or pentagon): ')
    c=360/a
    d=90-c
    n=0
    length=int(input('Enter the length of side: '))
    stem_colour=input('Enter the stem colour: ')
    first_petal_colour=input('Enter the first petal colour: ')
    second_petal_colour=input('Enter the second petal colour: ')
    for i in range(a):
        n+=1
        turtle.pencolor(stem_colour)
        turtle.right(n*c)
        turtle.forward(stem_length)
        if petal_type=='triangle':
            turtle.right(d)
            for i in range(2):
                if n%2==0:
                    turtle.pencolor(second_petal_colour)
                else:
                    turtle.pencolor(first_petal_colour)
                turtle.forward(length)
                turtle.left(120)
            turtle.forward(length)turtle.pencolor(stem_colour)
            turtle.home()
        elif petal_type=='square':
            if n%2==0:
                turtle.pencolor(second_petal_colour)
            else:
                turtle.pencolor(first_petal_colour)
            for i in range(4):
                turtle.forward(length)
                turtle.left(90)                
            turtle.pencolor(stem_colour)
            turtle.home()                
        elif petal_type=='pentagon':
            if n%2==0:
                turtle.pencolor(second_petal_colour)
            else:
                turtle.pencolor(first_petal_colour)            
            turtle.left(54)
            turtle.forward(length)
            turtle.right(72)
            turtle.forward(length)
            turtle.right(72)
            turtle.forward(length)
            turtle.right(72)
            turtle.forward(length)
            turtle.right(72)
            turtle.forward(length)
            turtle.pencolor(stem_colour)
            turtle.home()
    turtle.pencolor(stem_colour)
    turtle.home()    
    turtle.exitonclick()
