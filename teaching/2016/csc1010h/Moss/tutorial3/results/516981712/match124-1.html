def main():
    time=input("Enter a time in 24 hour clock (hh:mm):\n")
    t=int(time[0:2])
    t1=int(time[-2])
    t2=int(time[-1])
    
    if t==12:
        print("12:",t1,t2," pm",sep="")
        
    elif t>12:
        t=t-12
        print(t,":",t1,t2," pm",sep="")
        
    elif t==00:
        print("12:",t1,t2," am",sep="")
        
    else:
        print(t,":",t1,t2," am",sep="")
