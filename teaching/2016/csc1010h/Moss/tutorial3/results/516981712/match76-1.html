from turtle import*
n=int(input('Enter the number of petals (must be an even number): '))
length=int(input('Enter the length of stem: '))
shape=input('Enter the petal type (triangle, square, or pentagon): ')
lenside=int(input('Enter the length of side: '))
scolor=input('Enter the stem colour: ')
fcolor=input('Enter the first petal colour: ')
secolor=input('Enter the second petal colour: ')

def square():
    left(45)
    for i in range(4):
        forward(lenside)
        right(360/4)
    left(180-45)

def pent():
    left(60)
    for i in range(5):
        forward(lenside)
        right(360/5)
    left(180-60)

def tri():
    
    left(30)
    for i in range(3):
        forward(lenside)
        right(360/3)
    left(180-30)
    
def stem():
    k=0
    for i in range(n):
        color(scolor)
        k=k+1
        forward(length)
        
        if k%2==0:
            
            color(fcolor)
        else:
            color(secolor)
        
        if shape=='triangle':
            tri()
        elif shape=='pentagon':
            pent()
        else:
            square()
        penup()
            
        forward(length)
        pendown()
        right(180-360/n)
