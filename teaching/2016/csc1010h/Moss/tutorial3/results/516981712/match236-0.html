from turtle import*
petals = int(input("Enter the number of petals (must be an even number): "))
stem = int(input("Enter the length of stem: "))
shape = input("Enter the petal type (triangle, square, or pentagon): ")
length = int(input("Enter the length of side: "))
stem_col = input("Enter the stem colour: ")
petal_col_1 = input("Enter the first petal colour: ")
petal_col_2 = input("Enter the second petal colour: ")
pensize(10)
speed(0)
def square():
    left(45)
    for i in range(4):
        forward(length)
        right(360/4)
    left(180-45)
    
def triangle():
    left(30)
    for i in range(3):
        forward(length)
        right(360/3)
    left(180-30)
    
def pentagon():
    left(54)
    for i in range(5):
        forward(length)
        right(360/5)   
    left(180-54)

def draw_shape(val):
    if val=="square":
        square()
    elif val=="triangle":
        triangle()
    else:
        pentagon()
    
def main2():
    angle = 180- 360/petals
    petal = True 
    for i in range(petals):
        color(stem_col)
        forward(stem)
        if petal:
            color(petal_col_1)
        else:
            color(petal_col_2)
        draw_shape(shape)
        color(stem_col)
        forward(stem)
        right(angle)
        petal = not petal
        
main2()
exitonclick()
