length_side = int(input('Enter the length of side:'))
stem_colour = input('Enter the stem colour:')
first_petal = input('Enter the first petal colour:')
second_petal = input('Enter the second petal colour:')

turtle.speed(0)

if petal_type == 'pentagon':
    for u in range(int(petals/2)):
        turtle.pendown()
        turtle.color(stem_colour)
        turtle.forward(length_stem)
        turtle.left(54)
        for i in range(5):
            turtle.color(first_petal)
            turtle.forward(length_side)
            turtle.right(72)
        turtle.penup()
        turtle.right(54)
        turtle.back(length_stem)
        turtle.right(int(360/petals))
        
        turtle.pendown()

        turtle.forward(length_stem)

            turtle.right(72)
