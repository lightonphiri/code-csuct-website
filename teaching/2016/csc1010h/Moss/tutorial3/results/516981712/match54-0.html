from turtle import*
def main():
    petalnum=int(input('Enter the number of petals (must be an even number): '))


    lengthstem=int(input('Enter the length of stem: '))
    petal=input('Enter the petal type (triangle, square, pentagon): ')
    side=int(input('Enter the length of side: '))
    stem=input('Enter the stem colour: ')
    petal1=input('Enter the first petal colour: ')
    petal2=input('Enter the second petal colour: ')    
    
    if petal=='triangle':
        for i in range(petalnum):
            pencolor(stem)
            forward(lengthstem)
            if i%2==0:
                    pencolor(petal2)
            else:
                    pencolor(petal1)                
            right(30)#turn for petal
            forward(side)
            left(120)
            forward(side)
            left(120)
            forward(side)
            right(30)
            penup()
            forward(lengthstem)
            pendown()
            right(180-360/petalnum)#turn for stem        
        
            
    elif petal=='square':
        for i in range(petalnum):
            pencolor(stem)
            forward(lengthstem)
            if i%2==0:
                    pencolor(petal2)
            else:
                    pencolor(petal1)                
            right(90)
            forward(side)
            left(90)
            forward(side)
            left(90)
            forward(side)
            left(90)
            forward(side)
            right(0)
            penup()
            forward(lengthstem)
            pendown()
            right(180-360/petalnum)        
        
                
    else:
        for i in range(petalnum):
                pencolor(stem)
                forward(lengthstem)
                if i%2==0:
                    pencolor(petal2)
                else:
                    pencolor(petal1)        
                right(54)
                forward(side)

                right(54)
                penup()
                forward(lengthstem)
                pendown()
                right(180-360/petalnum)        
       
    exitonclick()

def inches():


    min_inche=int(input('Enter the minimum number of inches (not less than 0):\n'))
    max_inche=int(input('Enter the maximum number of inches (not greater than 11):\n'))
    print('Inches:',end=' ') 
    for i in range(min_inche,max_inche+1):
        print('  {:>2}'.format(i),end=' ')
    print()    
    print('Metres:',end=' ')    
    for i in range(min_inche,max_inche+1):
        metres=i*0.0254
        print('{:.2f}'.format(metres),end=' ')

def main():
    #24 hour time to 12 hour time
    time=input('Enter a time ([h]h:mm [am|pm]):\n')
    end=time[-1:]
    if end!='m':
        hh=time[0:2]
        mm=time[3:5]
        if hh=='00':
            print('12'+':'+str(mm),'am')
        if '00'<hh<'10':
            print(time[1:],'am')
        if '09'<hh<'12':
            print(time,'am')
        if hh>'12':
            newhh=int(hh)-12
            print(str(newhh)+':'+str(mm),'pm')
        if hh=='12':
            print(time,'pm')
            
    #12 hour to 24 hour time        
    elif end=='m':
        if len(time)==8: # checks the length of time
            hh=time[0:2]
            mm=time[3:5]
            am_pm=time[6:]
            #convertion of time
            if hh=='12' and am_pm=='am':
                print(str(int(hh)-12)+'0'+':'+str(mm))#prints the results
            elif hh=='12' and am_pm=='pm':
                print(time[0:5])
            elif '09'<hh<'12'and am_pm=='am':
                print(time[0:5])                 #prints the results
            elif '09'<hh<'12' and am_pm=='pm':
                print(str(int(hh)+12)+time[2:5]) #prints the results
                
        elif len(time)==7:  #checks the length of time
            hh=time[0]
            mm=time[2:4]
            am_pm=time[5:]
            #convertion of time
            if am_pm=='am':
                print('0'+str(time[0:4]))
            elif am_pm=='pm':
                print(str(int(hh)+12)+time[1:4])
                
main()
