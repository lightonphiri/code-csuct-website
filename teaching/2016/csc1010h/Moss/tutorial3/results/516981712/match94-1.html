def main():
    time=input("Enter a time in 24 hour clock (hh:mm):\n")
    t=time[0:2]
    t=int(t)
    if t>00 and t<12:
        print(str(t)+time[2::],"am")
    elif t==12:
        print(str(t)+time[2::]+" pm")
    elif t==00:
        print(str(t+12)+time[2::]+" am")
    else:
        print(str(abs(t-12))+time[2::]+" pm")
