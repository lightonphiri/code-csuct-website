import turtle

petals_number = int(input('Enter the number of petals (must be an even number): '))
stem_length = int(input('Enter the length of stem: '))
petal_type = str(input('Enter the petal type (triangle, square, or pentagon): '))
side_length = int(input('Enter the length of side: '))
stem_colour = str(input('Enter the stem colour: '))
first_petal_colour = str(input('Enter the first petal colour: '))
second_petal_colour = str(input('Enter the second petal colour: '))

if petal_type =='triangle':
    turtle.pencolor(stem_colour)
    for i in range(petals_number//2):
        turtle.forward(stem_length)
        turtle.left(30)
        turtle.pencolor(first_petal_colour)
        for i in range(3):
            turtle.forward(side_length)
            turtle.right(360/3)
        turtle.pencolor(stem_colour)
        turtle.right(30)
        turtle.backward(stem_length)
        turtle.right(360/petals_number)
        
        turtle.forward(stem_length)
        turtle.left(30)
        turtle.pencolor(second_petal_colour)
        for i in range(3):
            turtle.forward(side_length)
            turtle.right(360/3)
        turtle.pencolor(stem_colour)
        turtle.right(30)
        turtle.backward(stem_length)
        turtle.right(360/petals_number)  
             
elif petal_type =='square':
    turtle.pencolor(stem_colour)

        turtle.pencolor(stem_colour)
