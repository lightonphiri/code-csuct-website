def square(rotAngle,size,firstColor,stemColor,secondColor,stemLength,stemNumber):
    
    #k =Turtle()

        color(stemColor)
        fd(stemLength)
        if i%2==0:
            color(firstColor)

            color(secondColor)
        lt(45)
        fd(size)

        fd(size)
        rt(90)
        fd(size)
        rt(90)
        fd(size)
        color(stemColor)
        home()

    exitonclick
