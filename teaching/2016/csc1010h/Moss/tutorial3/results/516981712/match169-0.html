def main():
    petal = int(input('Enter the number of petals (must be an even number): '))
    stem = int(input('Enter the length of stem: '))
    typ = input('Enter the petal type (triangle, square, or pentagon): ')
    side = int(input('Enter the length of side: '))
    Stem_col = input('Enter the stem colour: ')
    first_P = input('Enter the first petal colour: ')
    sec_P = input('Enter the second petal colour: ')

    import turtle
    
    turn1 = 360 / petal
    
    for i in range(1,petal+1):
        turtle.pencolor(Stem_col)
        turn = turn1 * i
        turtle.right(turn)
        turtle.fd(stem)
        det = i % 2
        if det != 0:
            turtle.pencolor(first_P)
        else:
            turtle.pencolor(sec_P)
        if typ == 'triangle':
            turtle.left(30)


            turtle.fd(side)
            turtle.right(120)
            turtle.fd(side)
            turtle.right(120)
            turtle.fd(side)
            turtle.penup()
            turtle.home()
            turtle.pendown()
        
        elif typ == 'square':
            turtle.left(90)
            turtle.fd(side//2)
            turtle.right(90)
            turtle.fd(side)
            turtle.right(90)
            turtle.fd(side)
            turtle.right(90)
            turtle.fd(side)
            turtle.right(90)
            turtle.fd(side//2)
            turtle.penup()
            turtle.home()
            turtle.pendown()

            turtle.left(54)

    print('Inches:', end='')
    for i in range(mini,maxi+1):
        print('{0:>5}'.format(i), end='')
        if i == maxi:
            print("")

    print('Metres:', end="")
    for i in range (mini, maxi+1):
        metre = round(0.0254 * i,2)
        print('{0:>5.2f}'.format(metre),end="")
