        if time[0:2]=='12' and time[6:]=='am':
            print('00'+':'+time[3:5])
        elif 1 <= int(time[0:1]) <= 11 and time[5:]=='am':
            print('0'+time[0:5])
        elif 1 <= int(time[0:1]) <= 11 and time[5:] =='pm':
            pm=int(time[0:1])+12
            print(str(pm) +':'+str(time[2:5]))
        else:
            print(time[0:5])
    else:
        if time[0:2]== "00" :
            print('12:'+time[3:]+' am')

        elif 13<= int(time[0:2]) <= 23 :
            pm=int(time[0:2])-12
            print(str(pm)+':'+time[3:]+' pm')
        elif time[0:2]=='12':
            print(time+' pm')
        else:
            print(time[1:]+' am')
