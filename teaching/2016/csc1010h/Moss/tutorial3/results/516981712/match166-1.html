def to_twelve_hr(nako):
    

    #nako  = input("Enter a time in 24 hour clock (hh:mm):\n")
    if int(nako[nako.find(':')+1:])<=59:
        if nako[:nako.find(":")] == '00':
            d = '12:' + nako[nako.find(":")+1:] + ' am'
        elif nako[:nako.find(":")] == '12':
            d = '12:' + nako[nako.find(":")+1:] + ' pm'
        elif int(nako[:nako.find(":")])>12:
            d = str(int(nako[:nako.find(":")])-12) + nako[nako.find(':'):]+' pm'
        elif int(nako[:nako.find(":")])<12:
            d = str(int(nako[:nako.find(":")]))+nako[nako.find(":"):]+' am'
        print(d)
