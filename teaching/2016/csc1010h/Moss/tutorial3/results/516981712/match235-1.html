classify=''
gvm=int(input("What is the gross vehicle mass (in kg)?\n"))
if gvm<= 3500:
    classify='B'
elif 3500<gvm<=16000:
    classify='C1'
else:
    classify='C'

trailer=input('Does the vehicle have a trailer?\n')
if trailer=='n':
    art=input('Is the vehicle articulated?\n')
    if art=='y':
        classify='E'+classify
    else:
        classify=classify
else:
    gvm_trail=int(input('What is gross vehicle mass of the trailer (in kg)?\n'))
    if gvm_trail>750:
        classify='E'+classify
    else:
        classify=classify
        
print('This vehicle is class {}.'.format(classify))
