b=int(input("Enter the length of stem:""\n"))
c=input("Enter the petal type (triangle, square, or pentagon):""\n")
d=int(input("Enter the length of side:""\n"))
e=input("Enter the stem color:""\n")
f=input("Enter the first petal color:""\n")
g=input("Enter the second petal color:""\n")
odd=int(0.5*a)
z=int(360/odd)
	 
def singlepentapetal(clr1):
    turtle.pencolor(e)
    turtle.forward(b)
    turtle.left(54)    
    turtle.pencolor(clr1)    
    for i in range(5):        
        turtle.forward(d)
        turtle.right(72)
    turtle.pencolor(e)
    turtle.left(126)
    turtle.forward(b)    
    turtle.right(180)    

def singlesqrpetal(clr3):
    turtle.pencolor(e)
    turtle.forward(b)
    turtle.left(45)
    turtle.pencolor(clr3)
    for i in range(4):
        turtle.forward(d)
        turtle.right(90)    
    turtle.left(135)
    turtle.pencolor(e)
    turtle.forward(b)
    turtle.right(180)    

def singletripetal(clr2):
    turtle.pencolor(e)
    turtle.forward(b)
    turtle.left(30)
    turtle.pencolor(clr2)
    for i in range(3):
        turtle.forward(d)
        turtle.right(120)
    turtle.pencolor(e)
    turtle.left(150)    
    turtle.forward(b)
    turtle.right(180)

def main_1():

    hours, minutes = int(time[0:2]), int(time[3:5])
    if 12<hours<=23:
        hours=hours-12
        if minutes==00:
            print(hours,":","00"," pm",sep="")  
            hours=hours-12
        elif minutes>0 and minutes<=59:
            print(hours,":",minutes," pm",sep="")
    elif 0<=hours<12:
        if minutes==00 and hours==00:
            print("12",":","00"," am",sep="")  
        if minutes<10 and hours==00:
            print("12",":","0",minutes," am",sep="")        
        elif hours==0 and 10<=minutes<=59:
            print("00",":",minutes," am",sep="")
        elif hours>0 and minutes>0 and minutes<=59:
            print(hours,":",minutes," am",sep="")
        elif hours>0 and minutes==00:
            print(hours,":","00"," am",sep="")
    elif hours==12 and minutes==00:
        print(hours,":","00"," pm",sep="")
    elif hours==12 and 10<minutes<=59:
        print(hours,":",minutes," pm",sep="")
    elif hours==12 and minutes<10:
        print(hours,":","0",minutes," pm",sep="")

def main():
    gvm=int(input("What is the gross vehicle mass (in kg)?""\n"))
    trailer=input("Does the vehicle have a trailer?""\n")
    if gvm<=3500:
        if trailer=="y":
            gvm_trailer=int(input("What is gross vehicle mass of the trailer (in kg)?""\n"))
            if gvm_trailer>750:
                print("This vehicle is class EB.")
            elif gvm_trailer<=750:
                print("This vehicle is class B.")
        elif trailer=="n":
            articulation=input("Is the vehicle articulated?""\n") 
            if articulation=="y":
                print("This vehicle is class EB.")
            if articulation=="n":
                print("This vehicle is class B.")                
    if 3500<gvm<=16000:
        if trailer=="y":
            gvm_trailer=int(input("What is gross vehicle mass of the trailer (in kg)?""\n"))
            if gvm_trailer>750:
                print("This vehicle is class EC1.")
            elif gvm_trailer<=750:
                print("This vehicle is class C1.")
        elif trailer=="n":
            articulation=input("Is the vehicle articulated?""\n") 
            if articulation=="y":
                print("This vehicle is class EC1.")        
            if articulation=="n":
                print("This vehicle is class C1.")
                
    if gvm>16000:
        if trailer=="y":
            gvm_trailer=int(input("What is gross vehicle mass of the trailer (in kg)?""\n"))
            if gvm_trailer>750:
                print("This vehicle is class EC.")
            elif gvm_trailer<=750:
                print("This vehicle is class C.")
        elif trailer=="n":
            articulation=input("Is the vehicle articulated?""\n") 
            if articulation=="y":
