stem_colour=turtle.textinput('stem','Enter the stem colour: ')
petal_colour1=turtle.textinput('colour1','Enter the first petal colour: ')
pedal_colour2=turtle.textinput('colour2','Enter the second petal colour: ')

  a=4
  b=45
  c=2
  d=135
elif pedal_type=='triangle':
  a=3
  c=4
  d=150

  a=5

  d=120
