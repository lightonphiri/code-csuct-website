# Program that converts inches to metres
# Wendy Ngcobo
# 10 April 2016

def main():
    # variables that store user's input,minimum number not less than 0 and max number not greater than 11
    min_num=int(input('Enter the minimum number of inches (not less than 0):\n'))
    max_num=int(input('Enter the maximum number of inches (not greater than 11):\n'))
    # empty strings 
    metres=''  
    inches=''
    for i in range(min_num,max_num+1):  # loop over the numbers between minimum and maximum
        metres_num=i*0.0254   # converts meters from inches
        # aligning inches and meters with caption
        inches=inches+'{0:4}{1:<1}'.format(i,' ') 
        metres=metres+'{0:1.2f}{1}'.format(metres_num,' ')
    print('Inches:',inches) # display output
    print('Metres:',metres)
    print()
main()
