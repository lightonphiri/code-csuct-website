import turtle
def main():
    number_of_polygons=int(input('Enter the number of petals (must be an even number): '))
    length_stem=int(input('Enter the length of stem: '))
    petal=input('Enter the petal type (triangle, square, or pentagon): ')
    petal_side=int(input('Enter the length of side: '))
    stem_colour=input('Enter the stem colour: ')
    first_colour=input('Enter the first petal colour: ')
    second_colour=input('Enter the second petal colour: ')
    

    if petal=='triangle':
        for i in range(number_of_polygons):
            turtle.color(stem_colour)
            turtle.forward(length_stem)
            turtle.left(30)
        
            for j in range(3):
                if i%2==0:
                    turtle.color(first_colour)
                    turtle.forward(petal_side)
                    turtle.right(120)
                    
                else:
                    turtle.color(second_colour)
                    turtle.forward(petal_side)
                    turtle.right(120)
            turtle.left(150)
            turtle.penup()
            turtle.forward(length_stem)
            turtle.pendown()
            turtle.left((180)-(360/number_of_polygons))
            
            
            
    elif petal=='square':
        for i in range(number_of_polygons):
            turtle.color(stem_colour)
            turtle.forward(length_stem)
            turtle.left(45)
            for j in range(4):
                if i%2==0:
                    turtle.color(first_colour)
                    turtle.forward(petal_side)
                    turtle.right(90)
                else:
                    turtle.color(second_colour)
                    turtle.forward(petal_side)
                    turtle.right(90)
                    
            turtle.left(135)
            turtle.penup()
            turtle.forward(length_stem)
            turtle.pendown()
            turtle.left((180-(360/number_of_polygons)))            
            
    elif petal=='pentagon':
        for i in range(number_of_polygons):
            turtle.color(stem_colour)
            turtle.forward(length_stem)
            turtle.left(54)
            for j in range(5):
                if i%2==0:
                    turtle.color(first_colour)
                    turtle.forward(petal_side)
                    turtle.right(72)
                else:
                    turtle.color(second_colour)
                    turtle.forward(petal_side)
                    turtle.right(72)
                
            turtle.left(126)
            turtle.penup()
            turtle.forward(length_stem)
            turtle.pendown()
            turtle.left((180)-(360/number_of_polygons))        
    
    turtle.exitonclick()

def main():
    inches=''
    meters=''    
    min_inches=int(input('Enter the minimum number of inches (not less than 0):\n'))
    max_inches=int(input('Enter the maximum number of inches (not greater than 11):\n'))
    for inch in range(min_inches,max_inches+1):
        inches+='{0:5}'.format(inch)
        meter=round(0.0254*inch,2)
        meterr=str(meter)
        if len(meterr)==3:
            
            met=meterr+'0'
        else:
            met=meterr
            
        meters+='{0:>5}'.format(met)
    print('Inches:'+inches)
    print('Metres:'+meters)

def main():
    #get the grosss vehicle mass
    gvm=int(input('What is the gross vehicle mass (in kg)?\n'))
    # ask the user whether the vehicle has a trailer
    trailer=input('Does the vehicle have a trailer?\n')
    if trailer=='y':
        #prompt the user to enter the trailer's mass
        mass=int(input('What is gross vehicle mass of the trailer (in kg)?\n'))
        # check whether the mass of the trailer is greater or less than750 kg
        if mass>750:
            if gvm<=3500:
                print('This vehicle is class EB.')
            elif gvm>3500 and gvm<=16000:
                print('This vehicle is class EC1.')
            else:
                print('This vehicle is class EC.')
        else:
            if gvm<=3500:
                print('This vehicle is class B.')
            elif gvm>3500 and gvm<=16000:
                print('This vehicle is class C1.')
            else:
                print('This vehicle is class C.')
    elif trailer=='n':
        #if the vehicle does not have a trailer ask the user whether the vehicle is articulated
        articulated=input('Is the vehicle articulated?\n')
        if articulated=='y':
            if gvm<=3500:
                print('This vehicle is class EB.')
            elif gvm>3500 and gvm<=16000:
                print('This vehicle is class EC1.')
            else:
                print('This vehicle is class EC.')
        else:
            if gvm<=3500:
                print('This vehicle is class B.')
            elif gvm>3500 and gvm<=16000:
                print('This vehicle is class C1.')
            else:
                print('This vehicle is class C.')
