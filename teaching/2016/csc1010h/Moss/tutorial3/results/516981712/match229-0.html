def main():
    GVM = int(input('What is the gross vehicle mass (in kg)?\n'))
    if GVM <= 3500 or 3500<GVM<=16000 or GVM > 16000:
        if GVM <= 3500:
            response = 'B'
        elif 3500<GVM<=16000:
            response = 'C1'
        elif GVM>16000:
            response = 'C'
        trail = input('Does the vehicle have a trailer?')
        if trail == 'y':
            gvm = int(input('\nWhat is gross vehicle mass of the trailer (in kg)?\n'))
            if gvm > 750:
                print('This vehicle is class E'+str(response),end='.')
            elif gvm < 750:
                print('This vehicle is class',response,end='.')
        elif trail == 'n':
            art = input('\nIs the vehicle articulated?\n')
            if art == 'y':
                print('This vehicle is class E'+str(response),end='.')
            elif art == 'n':
                print('This vehicle is class',response,end='.')
main()
