# GVM CAlculator program



gvm_vehicle = eval (input ("What is the gross vehicle mass (in kg)?\n"))

if gvm_vehicle <= 3500:
    class2 = "B"
elif gvm_vehicle > 3500 and gvm_vehicle <= 16000:
    class2 = "C1"
else:
    class2 = "C"
    
trailor = input ("Does the vehicle have a trailer?\n")

if trailor == "y":
    gvm_trailor = eval ( input ("What is gross vehicle mass of the trailer (in kg)?\n"))
    if gvm_trailor>750:
        class1 = "E"
    else:
        class1 = ""
        
else:
    art = input ("Is the vehicle articulated?\n")
    if art == "y":
        class1 = "E"
    else:
        class1 = ""
        
clas = class1 + class2
print ("This vehicle is class "+clas+'.')
