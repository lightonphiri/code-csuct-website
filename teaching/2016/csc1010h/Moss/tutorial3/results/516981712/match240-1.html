def main():
    x = int(input('Enter the minimum number of inches (not less than 0):\n'))
    y = int(input('Enter the maximum number of inches (not greater than 11):\n'))
    print("Inches:",end=" ")
    for inch in range(x,y+1):
        print('{0:>4}'.format(inch),end=' ')
        
    print("\nMetres:",end=' ')
    for m in range(x,y+1):
        metre=round(m/39.370,3)
        print('{0:0.2f}'.format(metre),end=' ')
