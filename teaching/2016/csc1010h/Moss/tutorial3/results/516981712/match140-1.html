from turtle import *

def draw_shape(number_of_sides=3, size=100):
    angle=360/number_of_sides
    left(90-angle/2)
    for i in range(number_of_sides):
        forward(size)
        if i+1 < number_of_sides:
            right(angle)
    left(90-angle/2)

def draw_stem(s_size=100, n_stems=6, n_sides=4, n_size=100 , color1='red', color2='blue', stem_c='green'):
    angle = 360/n_stems
    cur_c=color()
    color(stem_c)
    for i in range(n_stems):
        forward(s_size)
        color((color1,color2)[i%2])
        draw_shape(n_sides, n_size)
        color(stem_c)
        forward(s_size)
        left((angle + 180)%360)
    color(*cur_c)
draw_stem()
