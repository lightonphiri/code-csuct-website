def main():
    # prompt user, minimum inches
    minimum_inches=int(input("Enter the minimum number of inches (not less than 0):\n"))
    
    # prompt user, maximum inches
    maximum_inches=int(input("Enter the maximum number of inches (not greater than 11):\n"))
    
    # convert to metres
    print("Inches:", end='')
    for i in range(minimum_inches,maximum_inches+1):
        print("{0:5}".format(i), end='')
        continue
    print("\nMetres:", end='')
    for i in range(minimum_inches,maximum_inches+1):
        metres=i/39.3701
        print("{0:5.2f}".format(metres), end='')

def main():
    
    # prompt user, enter GVM
    vehicle_gvm=int(input("What is the gross vehicle mass (in kg)?\n"))
    
    # prompt user, does it have trailer
    trailer=input("Does the vehicle have a trailer?\n")
    if trailer=='y':
        # prompt user, enter GVM of trailer
        trailer_gvm=int(input("What is gross vehicle mass of the trailer (in kg)?\n"))


        if vehicle_gvm<=3500 and trailer_gvm>750:
            print("This vehicle is class EB.")
        elif 3500<vehicle_gvm<=16000 and trailer_gvm>750:
            print("This vehicle is class EC1.")
        elif vehicle_gvm>16000 and trailer_gvm>750:
            print("This vehicle is class EC.")
        elif vehicle_gvm<=3500 and trailer_gvm<=750:


            print("This vehicle is class B.")
        elif 3500<vehicle_gvm<=16000 and trailer_gvm<=750:
            print("This vehicle is class C1.")
        elif vehicle_gvm>16000 and trailer_gvm<=750:
            print("This vehicle is class C.")
            
    elif trailer=='n':
        # prompt user, is vehicle articulated
        articulated=input("Is the vehicle articulated?\n")
        if articulated=='y':
            if vehicle_gvm<=3500:
                print("This vehicle is class EB.")
            elif 3500<vehicle_gvm<=16000:
