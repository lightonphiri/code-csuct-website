import string

def palindrome (s):
   if len(s) == 1: # case: a
      return s.isalpha()
   elif len(s) == 2 and s[0].lower() == s[1].lower(): # case: aa
      return True
   elif len(s) == 0:
      return False
   elif s[0] in string.punctuation or s[0] in string.whitespace:
      return palindrome(s[1:])
   elif s[-1] in string.punctuation or s[-1] in string.whitespace:
      return palindrome(s[:-1])
   if len(s)%2 != 0:   
      if s[0].lower() == s[-1].lower():
         return palindrome (s[1:-1])
      else:
         return False
   elif len(s)%2 ==0:
      if s[0].lower() == s[-1].lower():
         return palindrome (s[1:-1])
      else:
         return False
   return False

def query(pattern, word):

        return True
    if len(pattern) != len(word):
        return False

def match(pattern, word):
    if pattern == '' and word == '':
        return True
    
    if len(pattern) == 0 and len(word) > 0:
        return False
    elif len(pattern) > 0 and len(word) == 0:
        if pattern[0] == '*':
            return match(pattern[1:], word)
        else:
            return False
    
    if pattern[0] == word[0]: #case: 'abc' , 'acd' 
        return match(pattern[1:], word[1:])
    
    if pattern[0] == '?': #case: ?werty
        if word == '':
            return False
        else:
            return match(pattern[1:], word[1:])
    
    if pattern[0] == '*':
        if len(pattern) == 1:
            return True
        if (len(pattern) > 1) and (pattern[1] in ("*","?")): #case: ?*?*?
            return match(pattern[1:],word)
        starPos = pattern[1:].find("*") + 1
        questionMarkPos = pattern[1:].find("?") + 1
        end = starPos
        if (questionMarkPos > starPos) and (starPos == 0):
            end = questionMarkPos
        if end == 0:
            end = len(pattern)
        cutoff = word.find(pattern[1:end])
        return match(pattern[1:],word[cutoff::])    
    
    return False
