import string

def palindrome (s):
   if len(s)==0 or len(s) == 1:
      return True
   elif s[0] in string.punctuation or s[0] in string.whitespace:
      return palindrome(s[:-1])
   elif s[-1] in string.punctuation or s[-1] in string.whitespace:
      return palindrome(s[:-1])
   elif s[0].lower() == s[-1].lower():
      return palindrome (s[1:-1])
   else:
      return False

def equals(word_one, word_two): #step 1
    if len(word_one) == len(word_two): #base case 1
        if word_one[0] == word_two[0]: #base case 2
            return equals(word_one[1:], word_two[1:])
        elif word_two == " " and word_one == " ": #base case 3
            return True
        else:
            return False

 
def query(pattern,word): #step 2 
    if len(pattern) == len(word):
        if pattern=="":
            return True # base case
        elif word[0]==pattern[0]:
            return query(pattern[1:],word[1:])
        elif pattern[0]=="?":
            return query(pattern[1:],word[1:])
        else:
            return False
