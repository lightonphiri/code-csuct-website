import string
def palindrome (s):
   if len(s) == 1:
      return s.isalpha()
   elif s=='':
      return True
   elif s[0] in string.punctuation or s[0] in string.whitespace:
      return palindrome(s[1:])


   elif s[-1] in string.punctuation or s[-1] in string.whitespace:
      return palindrome(s[0:-1])
   elif s[0].lower() == s[-1].lower():
      return palindrome (s[1:-1])
   else:
      return False

def equals(word_one,word_two):
    if len(word_one)==0 and len(word_two)==0:
        return True
    elif len(word_one)!=len(word_two):
        return False
    else:
        if word_one[0].lower()!=word_two[0].lower():
            return False
        else:
            return equals(word_one[1:],word_two[1:])
        
def query(pattern,word):
    if len(pattern)==0 and len(word)==0:
        return True
    elif len(pattern)!=len(word):
        return False
    
    else:
        if pattern[0].lower()!=word[0].lower() and pattern[0]!='?':
            return False
        elif pattern[0].lower()==word[0].lower() or pattern[0]=='?': 
            return query(pattern[1:],word[1:])
        
def match(pattern,word):
    if pattern=='' and word=='':
        return True
    elif pattern=='*' and word=='':
        return True
    elif pattern[0] and word=='':
            return False    
    else:      


        if pattern[0]==word[0] or pattern[0]=='?':
            return match(pattern[1:],word[1:])
        elif pattern[0]=='*' and pattern.count('*')==1:
            index=len(pattern)
            index2=len(word)-index+1
            return match(pattern[1:],word[index2:])
        elif pattern[0]=='*' and pattern.count('*')>1:
            return match(pattern[1:],word[word.find(pattern[1:2]):])        
        elif (pattern[0]!='?'or pattern[0]!='*') and (pattern[0]!=word[0]):
            return False
