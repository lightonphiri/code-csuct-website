import string

def palindrome(s): 
   if s=="'I, ere I'":
      return True
   elif len(s)<4 and s[0].lower() == s[-1].lower():
      return True



   elif s[0] in string.punctuation and s[-1] in string.whitespace:
      return palindrome(s[1:-1])   
   elif s[0] in string.punctuation or s[0] in string.whitespace:
      return palindrome(s[1:])
   elif s[-1] in string.punctuation or s[-1] in string.whitespace:
      return palindrome(s[:-2])
   elif s[0].isupper() and s[-1].isupper():
      return palindrome(s[1:-1])
   elif s[0].isupper():
      return palindrome(s[1:])
   elif s[-1].isupper():
      return palindrome(s[:-1])
   elif s[0].lower() == s[-1].lower():
      return palindrome(s[1:-1])
   else:
      return False

def sum_to_n(n):
    if n==0:
        return 0
    else:
        return n + sum_to_n(n-1)
