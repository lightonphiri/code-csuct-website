def equals(x,y):
    if len(x)==len(y):
        if x=='' and y=='':
            return True
        elif x[0]==y[0]:
            return equals(x[1:],y[1:])
        else:
            return False
    else:
        return False




def query(pattern, word):
    if pattern==word=='':
        return True
    elif pattern[0] == '?' and word[0].isalpha():
        return query(pattern[1:],word[1:])
    elif pattern[0] == word[0]:
        return query(pattern[1:],word[1:])
    elif len(pattern)!=len(word):
        return False
    elif pattern[0] != word[0]:
        return False

    if pattern==word=='':
        return True
    elif word=='' and len(pattern)>0:
        return False
    elif pattern =='' and len(word)>0:

    elif '*' in pattern and len(pattern)>len(word):

def population(n):
    if n<3:
        return 2
    else:
        return population(n-1)+population(n-2)

def sum_to_n(n):
    if n == 0:
        return 0
    else:
        return n + sum_to_n(n-1)
