def match(word_one, word_two):
    # length of words
    one=len(word_one)
    two=len(word_two)
    # find *
    first=word_one.find("*")
    second=word_two.find("*")
    # letters after *
    if first+second>=-1:
        if first>-1:
            if first==int(len(word_one))-1:
                return True
            word_one=word_one[first+1]
            rest=-int(len(word_one))
            word_two=word_two[rest:]
        elif second>-1:
            if second==int(len(word_two))-1:
                return True
            word_two=word_two[second+1:]
            rest=-int(len(word_one))
            word_one=word_one[rest:]

        return False
    elif len(word_one)==0 and len(word_two)==0:
        return True
    elif word_one[0]==word_two[0]:

    elif word_one[0]=="?":
        return match(word_one[1:], word_two[1:])

        return match(word_one[1:], word_two[1:])
