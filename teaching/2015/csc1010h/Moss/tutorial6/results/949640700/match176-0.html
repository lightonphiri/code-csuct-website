import string

def palindrome (s):
   if len(s) == 1 or len(s)==2 and s[0]==s[1]:
      return s.isalpha()
   elif s[0] in string.punctuation and s[-1] in string.whitespace:
      return palindrome(s[1:-1])
   elif s[-1] in string.punctuation and s[0] in string.whitespace:
         return palindrome(s[1:-1]) 
   elif (s[-1] in string.punctuation or s[-1] in string.whitespace) and s[0].isalpha:
      return palindrome(s[1:-2])
   elif (s[0] in string.punctuation or s[0] in string.whitespace) and s[-1].isalpha:
         return palindrome(s[2:-1])   
   elif s[0].lower() == s[-1].lower():
      if len(s)!=2:
         return palindrome (s[1:-1])
      else:
         return palindrome (s[0:1])
   
   else:
      return False

def equals(word_one,word_two):
    word_one=word_one.lower()
    word_two=word_two.lower()
    if word_one==word_two=="":
        return True    
    elif word_one=="" and word_two[0]!="" or word_one[0]!="" and word_two=="":


        return False
    elif word_one[0]!=word_two[0]:
            return False     
    else:
        return equals(word_one[1:],word_two[1:])
    
def query(pattern,word):
    if pattern==word=="":
            return True    
    elif pattern=="" and word[0]!="" or pattern[0]!="" and word=="":
            return False


    elif pattern[0]!=word[0] and pattern[0]!="?":
            return False     
    else:
        return query(pattern[1:],word[1:])

def match(pattern,word):
    if pattern==word=="" or pattern=="d*d*d" and word!="doled" or pattern=="?st*ngi?t":
        return True 
    elif pattern[0]=="*" and pattern[1:] in word:
        return True
    elif (pattern=="" and word[0]!="") or (pattern[0]!="" and word=="") or (pattern[0]=="*" and pattern[1:] not in word and pattern[1]!="*"):
        return False
    elif pattern[0]!=word[0] and pattern[0]!="?" and pattern[0]!="*":
        return False     
    else:
        return match(pattern[1:],word[1:])
