import string

def palindrome(s):
    if s==" ":
        return True
    if s=="":
        return True
    if len(s)==1:
        return s.isalpha()
    elif s[0] in string.punctuation or s[0] in string.whitespace:
        return palindrome(s[1:])


    elif s[-1] in string.punctuation or s[-1] in string.whitespace:
        return palindrome(s[0:-1])
    elif s[0].lower() == s[-1].lower():
        return palindrome(s[1:-1])
    else:
        return False

def equals(word_one,word_two):
    if len(word_one)!=len(word_two):
        return False
    if word_one == "" and word_two == "":
        return True
    elif word_one[0].lower() == word_two[0].lower():
        return equals(word_one[1:],word_two[1:])
    else:
        return False
    
def query(pattern,word):
    equals(pattern,word)
    if len(pattern)!=len(word):
        return False
    if pattern[0] =="?" and len(word) ==1:
        return True
    if pattern[-1] == "?" or len(word)==1:
        return query(pattern[1:],word[1:])
    if pattern[0]=="?" and len(word)!=1:
        return query(pattern[:-1],word[:-1])
    elif pattern[0]!="?" and len(word)!=1:
        return query(pattern[1:-1],word[1:-1])
    else:
        return False
    
def match(pattern,word):
    if len(pattern)==0 and len(word)>=0:
            return True
    elif pattern[0]=='*' and len(word)>0:
        return match(pattern[1:],word[:])
    elif pattern[0]==word[0]:
        return match(pattern[1:],word[1:])
    elif pattern[-2:]==word[-2:] and pattern[-1]==word[-1]:
        return True
    elif pattern[0]=='*' and pattern[-1]==word[-1]:
        return True
    elif len(pattern)==len(word) and pattern[0]!=word[0]:
        return False
    elif pattern[0]==word[-1] and len(pattern)<len(word):
        return True
    return False

def sum_to_n(number):

        return 0
    else:
        return number + sum_to_n(number-1)
