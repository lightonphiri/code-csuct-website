import string

def palindrome (s):
   if len(s) == 1: #check the length of the string
      return s.isalpha()
   elif len(s)==2: #check the length of the string
      if s[0].lower() == s[1].lower():
         return True
      else:
         return False
   elif s[0] in string.punctuation or s[0] in string.whitespace: #check if beginning of s has punctuation or whitespace
      return palindrome(s[1:])
   elif s[-1] in string.punctuation or s[-1] in string.whitespace: #check if end o s has punctuation or whitespace
      return palindrome(s[:-1])
   elif s[0].lower() == s[-1].lower():
      return palindrome (s[1:-1])
   else:
      return False

def equals(word_one, word_two):
    if word_one=="" and word_two =="": #simplest base case
        return True
    elif (word_one=="" and word_two !="") or (word_one !="" and word_two =="") : #Check if any of the words contain anything while the other does not
        return False
    elif word_one != "" and word_two != "": #checks if each letter in each word matches
        if word_one[0] == word_two[0]:
            word_one = word_one[1:]
            word_two = word_two[1:]
            return equals(word_one, word_two)
        else:
            return False
        
    
def query(pattern,word):
    if pattern=="" and word =="": #simplest base case
        return True
    elif (pattern=="" and word !="") or (pattern !="" and word =="") : #checks if pattern or word both have something in them
        return False
    elif pattern != "" and word != "": #checks if pattern and word match
        if (pattern[0] == word[0]) or  (pattern[0] == "?"):
            pattern = pattern[1:]
            word = word[1:]
            return query(pattern, word)
        else:
            return False

def sum_to_n(n):
    if n==1: #simplest base case
        return 1
    else:
        return n + sum_to_n(n-1) # recursive calling of function
