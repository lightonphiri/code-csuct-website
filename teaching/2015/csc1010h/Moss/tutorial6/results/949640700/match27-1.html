def equals(word_one, word_two):
    if len(word_one)==len(word_two):
        if word_one=="" and word_two=="":
            return True
        elif word_one[0]==word_two[0]:
            return equals(word_one[1:], word_two[1:])
        else:
            return False
    else:
        return False

def match(pattern, word):
    if pattern =="" and word=="":
        return True
    elif pattern == "*" and len(word) > 0:
        return True
    if word=="":
        if pattern=="*" or pattern == "**":
            return True
        else:
            return False
    elif pattern[0] == "*":
        pattern = pattern[1:]
        word = word[word.find(pattern[0]):]
        pattern = pattern[pattern.find(word[0]):]
        return match(pattern, word)
    elif pattern[0] == "?":
        pattern = pattern[pattern.find("?")+1:]
        word = word[word.find(pattern[0]):]
        return match(pattern, word)
    elif pattern[0] == word[0]:
        return match(pattern[1:], word[1:])
    else:
        return False
