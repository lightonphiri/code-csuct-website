import string
def palindrome (s):
    if len(s) == 1:
        return s.isalpha()
    if len(s) == 0:
        return True
    elif s[0] in string.punctuation or s[0] in string.whitespace:
        return palindrome(s[1:])
    elif s[-1] in string.punctuation or s[-1] in string.whitespace:
        return palindrome(s[:-1])


    elif s[0].lower() == s[-1].lower():
        return palindrome (s[1:-1])
    else:
        return False

def equals(s,l):
    if len(s) == 0 and len(l) == 0:
        return True
    
    elif len(s)!=len(l) :
        return False
    
    if s[-1] == l[-1] :
        return equals(s[:-1],l[:-1])
    
    else :
        return False

def query(p,w):
    if len(p) == 0 and len(w) == 0:
        return True
    
    elif len(p)!=len(w) :
        return False
    


    elif p[0] == w[0] or p[0]=="?" :
        return query(p[1:],w[1:])
    
    else :
        return False

def population(n):
    if n == 0 :
        return 0
    elif n == 1:
        return 2
    else:
        return population(n-1) + population(n-2)

    if p == 0:
        return 0
    else:
        return p + sum_to_n(p-1)
