def equals(word_one, word_two):
    if word_one=='' and word_two!='':
        return False
    elif word_one=='' and word_two=='':
        return True
        
    elif word_one[0].lower()==word_two[0].lower():
        return equals(word_one[1:],word_two[1:])
    else:
        return False
    return equals(word_one, word_two)

def query(pattern, word):
    if len(pattern)!=len(word):
        return False
    elif pattern=='' and word=='':
        return True
    elif pattern[0]=='?' and len(pattern)==len(word):
        return query(pattern[1:],word[1:])
    elif pattern[0]==word[0]:
        return query(pattern[1:], word[1:])

    else:
        return False
    return query(pattern,word)
