def equals(word_one,word_two):
    if(len(word_one) == 0 and len(word_two) == 0):
        return True
    elif(len(word_one) == 0 or len(word_two) == 0):
        return False
    elif word_one[0] == word_two[0]:
        return equals(word_one[1:len(word_one)],word_two[1:len(word_two)])
    else:
        return False

def query(pattern,word):
    if(len(pattern) == 0 and len(word) == 0):
        return True
    elif(len(pattern) == 0 or len(word) == 0):
        return False
    elif pattern[0] == word[0] or pattern[0] == "?":
        return query(pattern[1:len(pattern)],word[1:len(word)])
    else:
        return False
