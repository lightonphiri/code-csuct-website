def equals(word_one, word_two):
    word_one=word_one.lower() #convert string to lowercase
    word_two=word_two.lower()
    
    if len(word_one)!=len(word_two): #check length of strings
        return False
    elif word_one == "" and word_two=="": #set base case 
        return True
    elif word_one[0]==word_two[0]:
        return equals(word_one[1:],word_two[1:])
    else:
        return False
