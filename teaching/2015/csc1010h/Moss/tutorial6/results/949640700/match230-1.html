import string

def palindrome (s):
   if s ==" ":
      return True
   if len(s) == 1:
      return s.isalpha()
   elif len(s)==0:
      return True
   elif s[0] in string.punctuation or s[0] in string.whitespace or s[0]==' ' :
      return (palindrome(s[1:]))
   elif s[-1] in string.punctuation or s[-1] in string.whitespace or s[-1]==' ':
      return palindrome(s[:-1])
   elif s[0].lower() == s[-1].lower():
      return palindrome (s[1:-1])
   else:
      return False

def equals(word_one,word_two):
    
    if word_one[0]!=word_two[0] or len(word_one)!=len(word_two):
        return False
    elif word_one[0]==word_two[0] and len(word_one)==len(word_two) and len(word_one)==1 and len(word_two)==1:
        return True
    else:
        return equals(word_one[1:], word_two[1:])



def query(pattern,word):
    if len (pattern)==0:
        return True
    elif pattern[0]=="?":
        return query(pattern[1:],word[1:])
    elif pattern[0]==word[0] and len(pattern) == len(word) or pattern[0]=="?":
        return True


    elif len(pattern)!=len(word) or pattern[0]!=word[0]:
        return False
    else:
        return query(pattern[1:],word[1:])



def match (pattern,word):
    if len(pattern)==0:
        return True    
    elif pattern[0]=="*" or pattern[0]=="?":
        return match(pattern[1:],word)
    elif pattern[0] in word and len(pattern)==1  or pattern in word :
        return True
    elif pattern.find("?")<=0 and pattern.find("*")<=0 and pattern not in word or pattern[0] not in word or pattern.count(pattern[0])>word.count(pattern[0]):
        return False
    else:
        return match(pattern[1:],word)
