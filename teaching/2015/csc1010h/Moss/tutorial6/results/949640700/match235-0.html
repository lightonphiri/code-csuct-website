import string

def palindrome (s):
   s=s.lower()
   if len(s) == 1 or s=="":
      return True
   elif len(s)==2:
      if s[-1]==s[0]:
         return True
      else:
         return False
   elif s[-1] in string.punctuation or s[-1] in string.whitespace:
      return palindrome(s[:-1])
   elif s[0] in string.punctuation or s[0] in string.whitespace:
      return palindrome(s[1:])
   elif s[0].lower() == s[-1].lower():
      return palindrome (s[1:-1])
   else:
      return False

def equals(word_one, word_two):
    word_one=word_one.lower()
    word_two=word_two.lower()
    if word_one == '' and word_two =='':
        return True
    elif word_one == '' and word_two != '':
        return False
    elif word_one != '' and word_two == '':
        return False
    elif len(word_one)!=len(word_two):
        return False
    elif word_one[0]==word_two[0] and word_one[-1]==word_two[-1]:
        return equals(word_one[1:-1],word_two[1:-1])
    else:
        return False

def sum_to_n(n):
    if n == 1:
        return 1
    else:
        return n + sum_to_n(n-1)
