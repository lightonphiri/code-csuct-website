      return palindrome(s[:-1])

def equals(word_one,word_two):
    if len(word_one)==len(word_two):
        if len(word_one)==0:
            return True
        elif word_one[0]==word_two[0]:
            return equals(word_one[1:],word_two[1:])

            return False

        return False
