import string
def palindrome (s):
   
    if len(s) == 1 or len(s)==0: #checking if len of s is 1 or is 0
        return True #must return true if condition above is met it must not retunr s.isalpha
    elif s[0] in string.punctuation or s[0] in string.whitespace: #check is s[0] is punctuation or whitespace
        return palindrome(s[1:])
    elif s[-1] in string.punctuation or s[-1] in string.whitespace: #checks is s[-1] is punctuation or whitespace
        return palindrome(s[:-1])
    elif s[0].lower() == s[-1].lower(): #makes first letter and last letter of s lower case
        return palindrome (s[1:-1]) #recursive part where first and last letters od s are sliced out
    else:
        return False

        return False

def sum_to_n(n):
    if n==0: #base case
        return 0
    else:
        return n + sum_to_n(n-1) #recursive par that sums up numbers from 1 to n
