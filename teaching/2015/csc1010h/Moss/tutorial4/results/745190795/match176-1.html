def LongJump():
    print('***** Long Jump Information System *****')
    print('Please enter the names of competitors. (Press enter when done.)')    
    count=1
    C=str(count)
    competitors=input('Competitor no.'+C+':\n')
    names=[]
    names.append(competitors)
    Dic={}
    while competitors!="":
        count+=1
        C=str(count)
        competitors=input('Competitor no. '+C+':\n')
        names.append(competitors)
    names.remove("")
    print('Please enter the distances for each competitor.')
    for i in names:
        L_i=[]
        print('Competitor:',i)
        A1=(input('Attempt 1:\n'))
        
        if isinstance(eval(A1),float):
        
            L_i.append((A1))
        A2=(input('Attempt 2:\n'))
        if isinstance(eval(A2),float):
            L_i.append((A2))
        A3=(input('Attempt 3:\n'))
        if isinstance(eval(A3),float):
            L_i.append((A3))
        if L_i==[]:
            L_i.append('No jump')
    
        Dic[i]=max(L_i)
    
    print(L_i)      
    print('{0:25}{1}'.format('Competitor','Longest Distance'))
    for j in Dic:
        print('{0:25}{1}'.format(j,Dic[j]))
    inv={v:k for k,v in Dic.items() }
    Max=max(inv)
    Max2=str(Max)
    
    print('The winner is',inv[Max],'with a jump of',Max2+'.')
LongJump()
