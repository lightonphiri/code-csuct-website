import turtle
def main():
    turtle.goto(0,0)
    cube_size=turtle.numinput("Size","Enter the size of the sides:")
    turtle.setx(cube_size)
    turtle.sety(cube_size)
    turtle.left(90)
    turtle.goto(0,0)
    turtle.sety(cube_size)
    turtle.setx(cube_size)
    turtle.seth(45)
    turtle.forward(cube_size)
    turtle.seth(180)
    side=cube_size+((cube_size**2)/2)**0.5
    turtle.goto(side,0)
    turtle.goto(0,0)
    turtle.goto(0,cube_size)
    turtle.circle(cube_size)
    turtle.exitonclick()
main()
