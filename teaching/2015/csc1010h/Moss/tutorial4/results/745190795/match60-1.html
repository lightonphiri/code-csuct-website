        # End list with enter
        if competitor == "":
            del comp_list[-1]
            break
    # distance  
    print("Please enter the distances for each competitor.")
    distance = []
    personal_best = {}
    for name in comp_list:
        print('Competitor ' + name + ':')
        for i in range(1,4):
            att_no = str(i)
            attempt = input("Attempt " + att_no + ":\n")
            # no jump and foul
            if attempt == 'foul':
                distance = distance.append(attempt)
            else:
                attempt = float(attempt)
                distance.append(attempt)
                highest = distance[0]
                for i in distance:
                    if i > highest:
                        highest = i
            for j in distance:
                if j == 'foul':
                    distance.remove("foul")
            if distance == []:
                personal_best[name]= "No jump"
        personal_best[name] = highest
        distance = []
        
    # Make table of the personal bests
    rank_as = sorted(personal_best, key=personal_best.__getitem__)   # Ranking of the best ascending

import random

def main():
    print("Guess the mystery 4 letter code. (No letter occurs more than once.)")
    # Random Generator
    difficulty=int(input('Choose difficulty level (4 to 26):\n'))
    letters=list('abcdefghijklmnopqrstuvwxyz')[0: difficulty]
    print('The letters in the code are chosen from: {}.'.format(letters))
    secret_code=[]
    for i in range(4):
        rand_index=random.randint(0, len(letters)-1)
        letter=letters.pop(rand_index)
        secret_code.append(letter)

    # NikNaks Game
    j = 11
    while True:
        j -= 1
        if j != 0:
            print("You have", j, "guesses remaining.")
        else:
            print("Sorry, you lose.")
            print("The secret number was " + secret_code[0] + secret_code[1] + secret_code[2] + secret_code[3] + ".")
            break
        # guess input
        guess = list(input(("Enter your guess:\n")))
        if guess == secret_code:
            print("Correct!")
            break
        else:
            naks = []
            niks = []            
            for i in guess:
                for e in secret_code:
                    if i == e and guess.index(i) != secret_code.index(e):
                        naks.append(i)
                    elif i == e and guess.index(i) == secret_code.index(e):
                        niks.append(i)
            no_naks = len(naks)
            no_niks = len(niks)   


            if no_niks == 1 and no_naks == 1:
                print("There is", no_niks, "Nik and", no_naks, "Nak.")
            elif no_niks == 1 and no_naks != 1:
                print("There is", no_niks, "Nik and", no_naks, "Naks.")
            elif no_niks != 1 and no_naks != 1:
                print("There are", no_niks, "Niks and", no_naks, "Naks.")
