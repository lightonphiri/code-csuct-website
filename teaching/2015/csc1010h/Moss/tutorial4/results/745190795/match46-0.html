import random

def main():
    print("Guess the mystery 4 letter code. (No letter occurs more than once.)")
    difficulty=int(input("Choose difficulty level (4 to 26):\n"))
    letters=list("abcdefghijklmnopqrstuvwxyz")[0: difficulty]
    print("The letters in the code are chosen from: {}.".format(letters))
    secret_code=[]
    for i in range(4):
        rand_index=random.randint(0, len(letters)-1)
        letter=letters.pop(rand_index)
        secret_code.append(letter)
    
    for r in range(10,0,-1):
        print("You have",r, "guesses remaining.")
        guess=input("Enter your guess:\n")
        if list(guess)==secret_code:
            print("Correct!")
        


        else:
            
            count=0
            count_2=0
            for i in range(4):
                if secret_code[i]==guess[i]:
                    count+=1
                    
                elif secret_code[i]!=guess[i] and guess[i] in secret_code:
                 
                    count_2+=1
                
             
             


            if count==1 and count_2==1:
                print("There is",count,"Nik and",count_2,"Nak.")
                
            elif count==1 and count_2!=1:
                print("There is",count,"Nik and",count_2,"Naks.")
                 
            elif count!=1 and count_2!=1:
                print("There are",count,"Niks and",count_2,"Naks.")
            elif count!=1 and count_2==1:
                print("There are",count,"Niks and",count_2,"Nak.")
                
        
    
            
        if list(guess)==secret_code:
            break
        
        elif list(guess)!=secret_code and r==1:
            print("Sorry, you lose.")
            x=""
            for l in secret_code:
                x+=l
            print("The secret number was "+str(x)+".")
            
            
            
if __name__=='__main__':
    main()
