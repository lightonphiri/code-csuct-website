def names () :
    print("{:*^40}".format(' Long Jump Information System '))
    print("Please enter the names of competitors. (Press return when done.)")
    
    n  = 0  
    _player_names_ = []
    while True :
        n+=1
        print ("Competitor no. {}:".format(n))
        _name_ = input("")
        _player_names_.append(_name_)     
        if not _name_ :   
            _player_names_.remove("")
            break 
                       
    print("Please enter the distances for each competitor.")    
    player_highest ,highest , sort = {} , [] , []
    
  
    for player in _player_names_ :
        z=0
        print ("Competitor {}:".format(player))
        attempts = ['']
        attempts.remove('')
        for attempt in range(3) :
            z +=1    
            print("Attempt {}:".format(z))
            
            _attempt_ = input("")
            
            
            
            if _attempt_.isalpha() : 

                attempts.append('No Jump')
            else :
                d = float(_attempt_)
                attempts.append('{0:.2f}'.format(d)) 
                sort.append('{0:.2f}'.format(d))
                
                
        order = sorted(sort)       
        player_highest[max(attempts)]=player
        highest.append(max(attempts))
        
    #s_high = sorted(highest)
    #print(order)    
    #print(player_highest)
     
  #  if max(highest) == 0 : 
   #     max(highest) == "No jump"
   
    print("")
    print("{0:<25}{1}".format("Competitor","Longest Distance"))
    count = -1
    #print(attempts)
    for i in highest:
        count+=1
        print("{:25}{:}".format(player_highest[i],i))
              
    print("The winner is {} with a jump of {}.".format(player_highest[order[-1]],order[-1]))
    print("")
