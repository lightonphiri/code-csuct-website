import random
def main():    
    print('Guess the mystery 4 letter code. (No letter occurs more than once.)')
    difficulty=int(input('Choose difficulty level (4 to 26):\n'))
    letters=list('abcdefghijklmnopqrstuvwxyz')[0: difficulty]
    print('The letters in the code are chosen from: {}.'.format(letters))
    secret_code=[]
    for i in range(4):
        rand_index=random.randint(0, len(letters)-1)
        letter=letters.pop(rand_index)
        secret_code.append(letter)    
    
    
    guess=0
    while True:
        niks=0
        naks=0
        


        print('You have',10-guess,'guesses remaining.')
        raai=input('Enter your guess:\n')        
        for i in range(4):
           
            if secret_code[i]==raai[i]:
                niks+=1
            elif secret_code[i] in raai:
                naks+=1
                
        guess+=1
        if niks==4:
            break
        if niks==1 and naks!=1:
            print("There is ",niks," Nik and ",naks," Naks",".",sep="")
        elif niks!=1 and naks ==1:
            print("There are ",niks," Niks and ",naks," Nak",".",sep="")
        elif niks!=1 and naks !=1:
            print("There are ",niks," Niks and ",naks," Naks",".",sep="")
        elif niks==1 and naks ==1:
            print("There is ",niks," Nik and ",naks," Nak",".",sep="")
            
            
        
        
        if niks==4:
            break
        if guess==10:
            break
    if niks==4:
        print('Correct!')

        print('Sorry, you lose.')
        c=''
