def main():
    
    from collections import OrderedDict
    universe=OrderedDict()
    print("***** Long Jump Information System *****")
    print("Please enter the names of competitors. (Press return when done.)")
    com=1
    while True:
        
        print("Competitor no.",str(com)+":")
        compete=input()
        if compete=="":
            break
        else:
            universe[compete]=""
            com=int(com)
            com+=1
    print("Please enter the distances for each competitor.")
    for i in universe:
        print("Competitor",i+":")
        for j in range(1):
            attempt1=input("Attempt 1:\n")
            attempt2=input("Attempt 2:\n")
            attempt3=input("Attempt 3:\n")
            universe[i]=attempt1,attempt2,attempt3
    print()
    winnerlist=[]
    print("Competitor","             ","Longest Distance")
    for i in universe:
        newkid=[]
        if universe[i][0]==universe[i][1]==universe[i][2]=="foul":
            print("{0:<24}".format(i),"No jump")
        else:
            for d in universe[i]:
                if not d.isalpha():
                    newkid.append(d)
                    highest=max(newkid)
                    highest="{0:4.2f}".format(float(highest))
                    winnerlist.append(highest)  
            print("{0:<24}".format(i),highest)
    
    winner=max(winnerlist)
    for e in universe:
        doctor=[]
        for v in universe[e]:
            if not v.isalpha():
                v="{0:4.2f}".format(float(v))
                doctor.append(v)
            else:
                pass
        if doctor!=[]:
            if winner==max(doctor):
                print()
                print("The winner is",e,"with a jump of",winner+".")
        else:
            pass
main()
