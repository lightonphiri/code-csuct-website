def main():
    number_of_compititor=1
    name=[]
    atmpt=[]
    diction={}
    score=0
    winner=''
    print('***** Long Jump Information System *****')
    print('Please enter the names of competitors. (Press return when done.)')
    while True:
        print("Competitor no.",str(number_of_compititor)+":")
        compititor=input('')
        if compititor!="":
            number_of_compititor+=1
            name.append(compititor)
            continue
        else:
            print('Please enter the distances for each competitor.')
            for i in name:
                print('Competitor '+str(i)+":")
                for p in range(3):
                    print('Attempt', str(p+1)+':')
                    attempt=input("")
                    if attempt=="foul":
                        attempt="0"
                        atmpt.append(attempt)
                        diction[i]=max(atmpt)
                    else:
                        atmpt.append(attempt)
                        diction[i]=max(atmpt)
                    if float(attempt)>float(score):
                        score=attempt
                atmpt=[]
            print()
            print("Competitor               Longest Distance         ")
            L=len(name)
            for t in range(L):
                #print("{0:25}{1:<15.2f}".format(name[t],float(diction[i])))
                if diction[name[t]]==score:
                    winner=name[t]
                if diction[name[t]]==str(0):
                    print("{0:25}{1:<25}".format(name[t],"No jump"))
                else:
                    print("{0:25}{1:<15.2f}".format(name[t],float(diction[name[t]])))
            print()       
            print("The winner is {0} with a jump of {1:.2f}.".format(winner,float(score)))
            break
                
            
main()

import random
def main():
    print('Guess the mystery 4 letter code. (No letter occurs more than once.)')
    difficulty=int(input('Choose difficulty level (4 to 26):\n'))
    letters=list('abcdefghijklmnopqrstuvwxyz')[0: difficulty]
    print('The letters in the code are chosen from: {}.'.format(letters))
    secret_code=[]
    p=""
    niks=0
    naks=0
    for i in range(4):
        rand_index=random.randint(0, len(letters)-1)
        letter=letters.pop(rand_index)
        secret_code.append(letter)
    attempt=10
    while attempt>=1:
        print("You have",attempt,"guesses remaining.")
        guess=input("Enter your guess:\n")
        for p in range(4):
            if guess[p]==secret_code[p]:
                niks+=1
            elif guess[p] in secret_code:
                naks+=1
        if niks==4:
            print('Correct!')
            break
        else:
            attempt-=1
        if niks==1 and naks==1:
                print("There is",niks, "Nik and",naks,"Nak.")
        elif niks==1 and naks>1:
                print("There is",niks, "Nik and",naks,"Naks.")

                print("There are",niks, "Niks and",naks,"Nak.")
        elif niks>1 and naks>1:
                print("There are",niks, "Niks and",naks,"Naks.")

        niks=0
