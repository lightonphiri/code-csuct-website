        break

for i in names:
    print('Competitor',i+':')
    current=0
    
    for p in range(1,4):
        attempt=input('Attempt '+str(p)+': \n')
        
        if attempt.isalpha():
            attempt=current
        elif float(attempt)>=float(current):
            current=eval(attempt)
            
    score.append(current)

modified_score=[]
for i in score:    
    if i==0:
        modified_score.append('No jump')
    else:
        modified_score.append('{0:.2f}'.format(i))

        difficulty=int(input('Choose difficulty level (4 to 26):\n'))
        letters=list('abcdefghijklmnopqrstuvwxyz')[0: difficulty]
        print('The letters in the code are chosen from: {}.'.format(letters))
        secret_code=[]


        for i in range(4):
                rand_index=random.randint(0, len(letters)-1)
                letter=letters.pop(rand_index)
                secret_code.append(letter)
        
        #print(secret_code)
        for i in range(10,-1,-1):
                if i==0:
                        print('Sorry, you lose.') 
                        print('The secret number was',secret_code[0]+secret_code[1]+secret_code[2]+secret_code[3]+'.')
                        break                
                
                else:
                        print('You have',i,'guesses remaining.')
                        guess=input('Enter your guess: \n')                        
                        nik=0
                        nak=0
                        for p in range(4):
                                if guess[p]==secret_code[p]:
                                        nik+=1
                                
                                elif guess[p] in secret_code:
                                        nak+=1
