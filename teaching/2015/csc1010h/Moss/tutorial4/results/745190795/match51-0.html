import random
def main():
    remaining=10

    print("Guess the mystery 4 letter code. (No letter occurs more than once.)")
    difficulty=int(input('Choose difficulty level (4 to 26):\n'))
    letters=list('abcdefghijklmnopqrstuvwxyz')[0: difficulty]
    print('The letters in the code are chosen from: {}.'.format(letters))
    code=[]
    for i in range(4):
        rand_index=random.randint(0, len(letters)-1)
        letter=letters.pop(rand_index)
        code.append(letter)

    one=code[0]
    two=code[1]
    three=code[2]
    four=code[3]
    while remaining!=0:
        nik=0
        nak=0    
        
        if remaining!=0:
            print("You have", remaining, "guesses remaining.")
            
        guess=input("Enter your guess:\n")
    

        guess_one=guess[0]
        guess_two=guess[1]
        guess_three=guess[2]
        guess_four=guess[3]

        guess_list=[]

        guess_list.append(guess_one)
        guess_list.append(guess_two)
        guess_list.append(guess_three)
        guess_list.append(guess_four)

        count_guess1=guess_list.count(one)
        count_guess2=guess_list.count(two)
        count_guess3=guess_list.count(three)
        count_guess4=guess_list.count(four)
        nak=count_guess1+count_guess2+count_guess3+count_guess4

        if guess_one==one:
            nak-=1
            nik+=1
        if guess_two==two:
            nak-=1
            nik+=1 
        if guess_three==three:
            nak-=1
            nik+=1
        if guess_four==four:
            nak-=1
            nik+=1

        elif nak==1:
            print("There are", nik, "Niks and", nak, "Nak.")
        elif nik==1 and nak==1:
            print("There is", nik, "Nik and", nak, "Nak.")
