while True:
        print("Competitor no.",str(n)+":")
        name=input('')

                competitors.append(name)
                n+=1
        else:
                print("Please enter the distances for each competitor.")
                for i in range(len(competitors)):
                        print("Competitor {}:".format(competitors[i]))
                        #score2=[]
                        score1=[]
                        for t in range(3):
                                print("Attempt",str(t+1)+":")
                                attempt=input()

                                if attempt=="foul":
                                        score1.append("0")
                                else:
                                        score1.append(attempt)
                        score2.append(max(score1))
                        
                for i in range(len(competitors)):
                        d[competitors[i]]=score2[i]
                print()
                print("Competitor               Longest Distance")
                winner=''
                for u in range(len(competitors)):
                        if d[competitors[u]]== max(score2):
                                winner=competitors[u]
                        if d[competitors[u]]=="0":
                                print("{0:25}{1}".format(competitors[u],"No jump"))
                        else:
                                print("{0:25}{1:.2f}".format(competitors[u],float(d[competitors[u]])))
                print()
                print('The winner is {0} with a jump of {1:.2f}.'.format(winner,float(d[winner])))
                break

from turtle import *
size=250
color('green', 'yellow')
speed('fastest')
setpos(size/10,32)
start=pos()             # Note starting position
begin_fill()
while True:
    forward(size)
    left (170)
    if abs(pos()-start) < 1:    # have we come back to start?
        break
end_fill()

exitonclick()
