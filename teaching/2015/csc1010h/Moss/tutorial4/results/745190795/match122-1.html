import turtle
import random
red = 200
green = 85
blue = 47        
pen = 10
turtle.write("Burning Forest")
def tree(branchLen, t, red, green, blue, pen):
    turtle.forward(80)
    #turtle.write("Burning Forest")
    turtle.dot(15,"red")
    if branchLen > 3:
        pen = pen*0.8
        t.pensize(pen)
        red = red - 15
        green = green + 8
    if branchLen > 5:
        angle = random.randrange(10, 70)
        angleTwo = 0.50*angle
        sub = (0.8*(random.randrange(1,24)))
        t.forward(branchLen) 
        t.right(angleTwo)
        tree(branchLen-sub,t, red, green, blue, pen)
        t.left(angle)
        tree(branchLen-sub, t, red, green, blue, pen)
        t.right(angleTwo)
        t.backward(branchLen)

def main():
    t = turtle.Turtle()
    t.hideturtle()
    t.speed(0)
    myWindow = turtle.Screen()
    t.left(90)
    t.up()
    t.backward(170)
    t.down()
    tree(random.randrange(40,47),t,red,green,blue, pen)
    turtle.bgcolor('orange')
    n = turtle.Screen()
    n.bgcolor("orange")
    tt= turtle.Turtle()
    tt.shape("turtle")
    tt.color("yellow")
    tt.penup()               
    size = 20

        t.tilt(30)

    turtle.color("black", "red")

main()
