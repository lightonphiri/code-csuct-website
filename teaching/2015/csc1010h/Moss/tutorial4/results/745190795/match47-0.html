#Program to log the results of an athletics long jump contest
#Mogase Maloa
#MLXMOG001
#03 August 2015

def main():
    name=' Long Jump Information System '
    print("{:*^40}".format(name))
    print("Please enter the names of competitors. (Press return when done.)")
    count=1
    competitors=[]
    while True:
        competitor_name=input("Competitor no. "+str(count)+":\n")
        if competitor_name=="":
            break
        competitors.append(competitor_name)
        count+=1
    print("Please enter the distances for each competitor.")
    competitordistance={}
    for i in competitors:
        print("Competitor",i+":")
        distances=[]
        for j in range(3):
            distance=input("Attempt "+str(j+1)+":\n")
            if distance!='foul':
                distance2=float(distance)
                distances.append(distance2)
        
        
        if distances==[]:
            competitordistance[i]='No jump'
        else:
            maximum=max(distances)
            competitordistance[i]=maximum
    heading1="Competitor"
    heading2="Longest Distance"
    print("\n{:<25}".format(heading1)+"{}".format(heading2))
    for i in competitors:
        if competitordistance[i]!='No jump':
            print("{:<25}".format(i)+"{:.2f}".format(competitordistance[i]))
        else:
            print("{:<25}".format(i)+"{}".format(competitordistance[i]))
    distanceonly=[]
    for i in competitordistance:
        if competitordistance[i] !='No jump':
            distanceonly.append(competitordistance[i])
    maximum2=max(distanceonly)
    distancecompetitor={v:m for m,v in competitordistance.items()}


    print("\nThe winner is",distancecompetitor[maximum2],"with a jump of","{:.2f}".format(maximum2)+".")
main()

#Program to play a game of Niknaks
#Mogase Maloa

#03 August 2015

import random
def main():
    print("Guess the mystery 4 letter code. (No letter occurs more than once.)")
    difficulty=int(input("Choose difficulty level (4 to 26):\n"))
    letters=list('abcdefghijklmnopqrstuvwxyz')[0:difficulty]
    print('The letters in the code are chosen from: {}'.format(letters)+".")
    secret_code=[]
    for i in range(4):


        rand_index=random.randint(0,len(letters)-1)
        letter=letters.pop(rand_index)
        secret_code.append(letter)
    guesses=10
    while guesses>0:
        print('You have',guesses,'guesses remaining.')
        code=input("Enter your guess:\n")
        count=0
        num=0
        t=0
        for j in secret_code:
            for k in code:
                if j==k and secret_code.index(j)==t:
                    count+=1
                    
                else:
                    if j==k and secret_code.index(j)!=t:
                        num+=1
                t+=1
            t=0
                        
        if count==4 and num==0:
            
            print('Correct!')
            break


        elif count==1 and num!=1:
            print("There is",count,"Nik and",num,"Naks.")
        elif count!=1 and num==1:
            print("There are",count,"Niks and",num,"Nak.")
        elif count==1 and num==1:
            print("There is",count,"Nik and",num,"Nak.")
        else:
            print("There are",count,"Niks and",num,"Naks.")
        guesses-=1
    else:
        print("Sorry, you lose.")
        print("The secret number was ",end='')
        for k in secret_code:
            print(k,end='')
        print('.')
if __name__ == '__main__':
    main()

turtle.shape('square')
turtle.shapesize(1.00,1.00,1)
turtle.penup()
turtle.setpos(-100,90)
turtle.setheading(270)
turtle.pendown()
turtle.begin_fill()
for i in range(8):
    
    
    for i in range(4):
        
        turtle.dot(10,'red')
        turtle.forward(100)
        turtle.right(90)
    turtle.right(45)
turtle.setheading(0)
turtle.end_fill()
turtle.exitonclick()
