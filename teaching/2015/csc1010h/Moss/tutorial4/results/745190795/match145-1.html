    letters=list('abcdefghijklmnopqrstuvwxyz')[0: difficulty]
    print('The letters in the code are chosen from: {}.'.format(letters))
    secret_code=[]
    for i in range(4):
        rand_index=random.randint(0, len(letters)-1)
        letter=letters.pop(rand_index)
        secret_code.append(letter)
    p=10
    while 0<p<=10:
        print("You have",p,"guesses remaining.")
        p-=1
        user=input("Enter your guess:\n")
        nik=0
        nak=0
        if secret_code==list(user):
            print("Correct!")
            break
        else:
            for i in range(4):
                if secret_code[i]==user[i]:
                    nik+=1
                elif user[i] in secret_code:
                    nak+=1
