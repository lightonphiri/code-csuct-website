def main():
    print("***** Long Jump Information System *****")
    print("Please enter the names of competitors. (Press return when done.)")
    number_count = 0 
    names = []
    while True:
        number_count +=1
        jump_name = str(input("Competitor no. " + str(number_count) + ":" +"\n"))
        if jump_name == "":
            break
        else:
            names.append(jump_name)
    print("Please enter the distances for each competitor.")
    person_and_best = {}
    for person in names:
        print("Competitor " + person + ":")
        jump1 = input("Attempt 1:"+ "\n")
        jump2 = input("Attempt 2:"+ "\n")
        jump3 = input("Attempt 3:"+ "\n")
        if jump1.isalpha() and jump2.isalpha() and jump3.isalpha():
            maximum = "No jump"
          
        elif jump1.isalpha() and not jump2.isalpha() and not jump3.isalpha():
            if float(jump2)>float(jump3):
                maximum = '{0:.2f}'.format(float(jump2))
            else:
                maximum = '{0:.2f}'.format(float(jump3))
                
        elif jump2.isalpha() and not jump1.isalpha() and not jump3.isalpha():
            if float(jump1)>float(jump3):
                maximum = '{0:.2f}'.format(float(jump1))
            else:
                maximum = '{0:.2f}'.format(float(jump3)) 
                
        elif jump3.isalpha() and not jump2.isalpha() and not jump1.isalpha():
            if float(jump2)>float(jump1):
                maximum = '{0:.2f}'.format(float(jump2))
            else:
                maximum = '{0:.2f}'.format(float(jump1))           
            
        elif jump1.isalpha() and jump2.isalpha() and not jump3.isalpha():
            maximum = '{0:.2f}'.format(float(jump3))
            
        elif jump1.isalpha() and jump3.isalpha() and not jump2.isalpha():
            maximum = '{0:.2f}'.format(float(jump2))  
        
        elif jump2.isalpha() and jump3.isalpha() and not jump1.isalpha():
            maximum = '{0:.2f}'.format(float(jump1))
            
        elif not jump1.isalpha() and not jump2.isalpha() and not jump3.isalpha():
            maximum = max('{0:.2f}'.format(float(jump1)), '{0:.2f}'.format(float(jump2)), '{0:.2f}'.format(float(jump3)))
        
        person_and_best[person]= maximum 
    
    print()
        
    print('{0:24}'.format("Competitor"), '{0:25}'.format("Longest Distance"))
    for person in names:
        print('{0:24}'.format(person), '{0:<25}'.format(person_and_best[person]))
    
    
    highest_jumps = []
    highest_jumps_num = []
    for person in names:
        highest_jumps.append(person_and_best[person])
        length = len(highest_jumps)
        for i in range (length):
            if not str(highest_jumps[i]).isalpha():
                    highest_jumps_num.append(highest_jumps[i])
                    real_highest = max(highest_jumps_num)
        high_jumper = max(person_and_best, key=person_and_best.get)
           
                       
    print()
    print("The winner is", high_jumper, "with a jump of", str(real_highest) + '.')    
                
                      
    
main()

import random 

def main():
        print("Guess the mystery 4 letter code. (No letter occurs more than once.)")
        difficulty=int(input('Choose difficulty level (4 to 26):\n'))
        letters=list('abcdefghijklmnopqrstuvwxyz')[0: difficulty]
        print('The letters in the code are chosen from: {}.'.format(letters))
        secret_code=[]
        for i in range(4):
                rand_index=random.randint(0, len(letters)-1)
                letter=letters.pop(rand_index)
                secret_code.append(letter)
        guesses = 10
        while guesses>0:
                print("You have", guesses, "guesses remaining.")
                guess = list(input("Enter your guess: \n"))
                if guess == secret_code:
                        print("Correct!")
                        break
                else:
                        nik = 0
                        nak_ini = 0
                        for i in range(4):
                                        if secret_code[i] == guess[i]:
                                                        nik +=1
                        for i in range(4):
                                        nak_ini = nak_ini + guess.count(secret_code[i])
                        nak = nak_ini - nik
                        if nik == 1 and nak == 1:
                                print("There is", nik , "Nik and", nak , "Nak.")
                        elif nik == 1 and nak !=1:
                                print("There is", nik , "Nik and", nak , "Naks.")
                        elif nik != 1 and nak ==1:
                                print("There are", nik , "Niks and", nak , "Nak.")
                        elif nik != 1 and nak != 1:
                                print("There are", nik, "Niks and", nak, "Naks.")
                        guesses-=1    
        if guesses == 0:
                a = secret_code[0]
                b = secret_code[1]
                c = secret_code[2]
                d = secret_code[3]
                print("Sorry, you lose. \nThe secret number was" + " " + a+b+c+d + ".") 
        

if __name__ == '__main__':
        main()
