import random



def main():
    no_of_guesses=10

    print('Guess the mystery 4 letter code. (No letter occurs more than once.)')
    difficulty=int(input('Choose difficulty level (4 to 26):\n'))
    letters=list('abcdefghijklmnopqrstuvwxyz')[0: difficulty]
    print('The letters in the code are chosen from: {}.'.format(letters))
    secret_code=[]
    for i in range(4):
        rand_index=random.randint(0, len(letters)-1)
        letter=letters.pop(rand_index)
        secret_code.append(letter)

    while no_of_guesses>0:
        print('You have',no_of_guesses,'guesses remaining.')
        no_of_guesses-=1
        guess=input('Enter your guess:\n')
        guess=list(guess)
            
        Niks=0
        Naks=0    
        for j in range(4):
            alpha1=secret_code[j]
            alpha2=guess[j]
        
            if alpha1 == alpha2:
                Niks+=1
            elif alpha2 in secret_code and alpha1!=alpha2: 
                Naks+=1
            else: pass
        
        
        if Niks!=4:
            if no_of_guesses>0: 
                if Niks==1 and Naks!=1:
                    print('There is',Niks,'Nik and',Naks,'Naks.')
                elif Naks==1 and Niks!=1:
                    print('There are',Niks,'Niks and',Naks,'Nak.')
                elif Niks==1 and Naks==1:
                    print('There is',Niks,'Nik and',Naks,'Nak.')
                else:
                    print('There are',Niks,'Niks and',Naks,'Naks.')
            else:
                print('Sorry, you lose.'+'\n'+'The secret number was '+secret_code[0]+secret_code[1]+secret_code[2]+secret_code[3]+'.')
        else: 
            print('Correct!')
            break
        
if __name__ == '__main__':
    main()
