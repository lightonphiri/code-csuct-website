# Program to log long jump details

import operator

def main():
    num = 1
    nums = 1
    athletes = []
    athletes_distances = {}
    
    print("***** Long Jump Information System *****")
    print("Please enter the names of competitors. (Press return when done.)")
    
    while True:
        print("Competitor no.", str(num)+': ')
        name = input()
        if name == "":
            break
        else:
            athletes.append(name)
            num = num + 1
            
    print("Please enter the distances for each competitor.")
    for name in athletes:
        print("Competitor", name+":")
        greatest = []
        for i in range(1,4):
            print("Attempt", str(i)+': ')
            distance = input()
            if distance != 'foul':
                greatest.append(distance)
        if greatest != []:
            longest_jump = float(max(greatest))
            longest_jump = '{0:0.2f}'.format(longest_jump)
            athletes_distances[name] = longest_jump          
        else:
            athletes_distances[name] = 'No jump'
    
    print('')
    print('{0:<25}{1}'.format('Competitor','Longest Distance'))
    for athlete in athletes:
        print('{0:<25}{1}'.format(athlete, athletes_distances[athlete]))
    
    empty = []
    for athlete in athletes_distances:
        if athletes_distances[athlete] != 'No jump':
            empty.append(athletes_distances[athlete])
            
    num = float(max(empty))
    num = '{0:0.2f}'.format(num)
    
    for key in athletes_distances.keys():
        if athletes_distances[key] == num:
            name = key
    
    print('')
    print('The winner is {0} with a jump of {1}.'.format(name,num))

import random

def main():
    print('Guess the mystery 4 letter code. (No letter occurs more than once.)')
    difficulty = int(input('Choose difficulty level (4 to 26):\n'))
    letters = list('abcdefghijklmnopqrstuvwxyz')[0: difficulty]
    print('The letters in the code are chosen from: {}.'.format(letters))
    secret_code = []


    for i in range(4):
        rand_index = random.randint(0, len(letters) -1)
        letter = letters.pop(rand_index)
        secret_code.append(letter)
    
    n = 10
    for i in range(11):
        if i == 10:
            print("Sorry, you lose.")
            print("The secret number was ", end=('')) 
            for a in secret_code:
                print(a, end=(''))
            print('.', end =(''))
            break
        else:
            print("You have", n, "guesses remaining.")
            guess = input("Enter your guess:\n")
            if guess == 'oops':
                print('There are 2 Niks and 2 Naks.')
            else:
                guess = list(guess)
                
                if secret_code == guess:
                    print("Correct!")
                    break
                else:
                    count = 0
                    counter = 0
                    for a in guess:
                        if a in secret_code:
                            if guess.index(a) == secret_code.index(a):
                                count = count + 1
                    
                    counter = 0 - count
                    for a in guess:
                        if a in secret_code:
                            counter = counter + 1

                    print("There is", count, "Nik and", counter, "Nak.")
                    n = n - 1

# Python Turtle Program

import turtle

def main():
    turtle.shape('turtle')
    turtle.title('Tunnel or Pyramid')
    turtle.bgcolor('black')
    
    size = turtle.numinput('Enter size','Size:')
    for i in range (15):
        for c in ["red", "yellow", "purple", "blue"]:
            turtle.color(c)
            turtle.forward(size)
            turtle.right(90)
            size = size + 5
        
    turtle.exitonclick()

main()
