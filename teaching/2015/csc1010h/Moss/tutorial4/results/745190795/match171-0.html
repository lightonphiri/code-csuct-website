import turtle

turtle.shape('triangle')
turtle.fillcolor('pink')

for i in range(4):
    if i % 2==0:
        turtle.color("orange", "pink")
        turtle.begin_fill()
        turtle.circle(80)
        turtle.end_fill()         
    else:
        turtle.pencolor('red')


    turtle.color("green", "blue")
    turtle.begin_fill()
    turtle.circle(80)
    turtle.end_fill()     
   
    turtle.tilt(180)
    turtle.right(90)
    turtle.forward(100)
    turtle.circle(90,360)
    
turtle.exitonclick()
