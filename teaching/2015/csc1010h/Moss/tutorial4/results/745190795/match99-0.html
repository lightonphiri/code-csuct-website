def main():
    import turtle
    name = turtle.textinput("Name","Enter your name:")
    turtle.bgcolor("grey")
    turtle.penup()
    turtle.left(90)
    turtle.forward(250)
    turtle.pencolor("white")
    turtle.hideturtle()    
    turtle.write("Welcome to the humble abode of:", move=False, align="center", font=("Edwardian Script ITC", 32, "normal"))
    turtle.home()
    turtle.showturtle()
    
    count = 0
    turtle.pendown()
    turtle.pensize(3)
    for i in range(4):
        turtle.color("red", "black")
        turtle.begin_fill()
        turtle.circle(100)
        turtle.end_fill()
        angle = 360/4
        count+= angle
        turtle.right(count)        
    
    count = 0
    turtle.home()
    for i in range(4):
        for i in range(5):
            turtle.color("white")
            turtle.forward(100)
            turtle.right(72)
        angle = 360/4
        count+= angle
        turtle.right(count)
    turtle.penup()
    turtle.home()
    turtle.tilt(30)
    turtle.right(90)
    turtle.forward(260)
    turtle.pendown()
    turtle.pencolor("white")
    turtle.hideturtle()
    turtle.dot(10,"black")
    turtle.write(name, move=False, align="center", font=("Edwardian Script ITC", 32, "normal"))
        
    turtle.exitonclick()
main()
