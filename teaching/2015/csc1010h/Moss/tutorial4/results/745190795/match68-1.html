            longest.append(distance)                        # add value into the list

            longest=[]                                      # initialize longest to restart loop with new values

            print('{:<25}{}'.format(item,(display[item])))

import random
def main():
    code=''
    print("Guess the mystery 4 letter code. (No letter occurs more than once.)")
    difficulty=int(input('Choose difficulty level (4 to 26):\n'))
    letters=list('abcdefghijklmnopqrstuvwxyz')[0: difficulty]
    print('The letters in the code are chosen from: {}.'.format(letters))
    secret_code=[]
    for i in range(4):
        rand_index=random.randint(0, len(letters)-1)
        letter=letters.pop(rand_index)
        secret_code.append(letter)
    # To check things are working, uncomment this print line to see the
    # code that has been created.
    #print(secret_code)  
    
     
    guesses=10                    # number of guesses remaining
    niks=0                        # initialise a count variable, niks
    naks=0 
    t=0# initialise a count variable, naks
    while guesses>0:             #create a loop,only repeats 10 times
        print("You have {} guesses remaining.".format(guesses)) #print the number of guesses left
        guess=input("Enter your guess:\n")   #prompt user, enter guess
        for m in secret_code:                 # outer loop,each value in secret_code
            for n in guess:                   # inner loop, each value in guess
                if m==n and secret_code.index(m)==t:  # if there are similar letters and in same postion
                    niks+=1  # add one to niks
                elif m==n and secret_code.index(m)!=t:  # else if there are similar letters and not in the same postion
                    naks+=1  # add one to Naks
                #print(secret_code.index(m))
                #print(t)
                t+=1
                
            t=0
        
        if niks==4:          # if niks equals to 4
            print("Correct!")  #print correct then break
            break
        elif niks==1 and naks==1:
            print("There is {} Nik and {} Nak.".format(niks,naks))  #else print results, tthe number of niks and naks
        elif niks==1 and naks!=1:
            print("There is {} Nik and {} Naks.".format(niks,naks))  #else print results, tthe number of niks and naks
        elif niks!=1 and naks==1:
            print("There are {} Niks and {} Nak.".format(niks,naks))  #else print results, tthe number of niks and naks           
        else:
            print("There are {} Niks and {} Naks.".format(niks,naks))  #else print results, tthe number of niks and naks
        guesses-=1
        naks=0
        niks=0
        

    if guesses==0:
        for letter in secret_code:
            code+=letter
            
        print('Sorry, you lose.')
        print('The secret number was {}.'.format(code))
if __name__ == '__main__':
