def main():

    difficulty=int(input('Choose difficulty level (4 to 26):\n'))
    letters=list('abcdefghijklmnopqrstuvwxyz')[0: difficulty]
    print('The letters in the code are chosen from: {}.'.format(letters))
    secret_code=[]
    for i in range(4):


        rand_index=random.randint(0, len(letters)-1)
        letter=letters.pop(rand_index)
        secret_code.append(letter)
    #print(secret_code)
    niks=0
    naks=0
    string_secret_code=""
    for x in range(10,0,-1):
        print("You have",x,"guesses remaining.")
        answer=input("Enter your guess:\n")
        for n in range(4):
            string_secret_code+=secret_code[n]
        if list(answer)==secret_code:
            print("Correct!")
            break
        else:
            for z in range(4):
                a=string_secret_code.find(answer[z])
                if answer[z]==secret_code[z]:
                    niks+=1
                if answer[z]!=secret_code[z] and a>-1:
                    naks+=1
        if niks == 1:
            print('There is',niks,'Nik and',naks,'Naks.')
        elif naks == 1:
            print('There are',niks,'Niks and',naks,'Nak.')
        elif niks ==1 and naks == 1:
            print('There is',niks,'Nik and',naks,'Nak.')
        else:
            print("There are",niks,"Niks and",naks,"Naks.")
        niks=0
        naks=0
if __name__ == '__main__':     
    main()
