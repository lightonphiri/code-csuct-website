import turtle
import math
def circle(radius):    
    turtle.penup()
    # go to (0, radius)
    turtle.goto(0,radius)
    turtle.pendown() 
    colour = turtle.textinput('Colour Window','Enter colour:')
    turtle.color(colour)
    # number of times the y axis has been crossed with x having a a different sign
    crossedY = 0
    xSign = 1.0
    while crossedY <= 1:
        # move by 1/360 circumference
        turtle.forward(2*math.pi*radius/360.0)
        # rotate by one degree 
        turtle.right(1.0)
        # copysign function to get the sign of x coordinate 
        NewXsign = math.copysign(1, turtle.xcor())        
        if(NewXsign!= xSign):
            crossedY += 1
        xSign = NewXsign
       
    return  

circle(100)
turtle.penup()
turtle.home()
