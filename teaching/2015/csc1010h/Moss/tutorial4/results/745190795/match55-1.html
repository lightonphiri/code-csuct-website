import turtle

def main():
    turtle.shape('triangle')
    turtle.pensize(3)
    turtle.color("black", "magenta")
    turtle.begin_fill()
    turtle.circle(-40)
    turtle.end_fill()
    turtle.color("black")
    turtle.penup()
    turtle.left(90)
    turtle.back(40)
    turtle.right(180)
    turtle.pendown()
    turtle.stamp()
    turtle.penup()
    turtle.forward(100)
    turtle.left(90)
    turtle.shape('square')
    turtle.color("black", "blue")
    turtle.pendown()
    turtle.begin_fill()
    turtle.circle(-40)
    turtle.end_fill()
    turtle.color("black")
    turtle.penup()
    turtle.left(90)
    turtle.back(40)
    turtle.pendown()
    turtle.stamp()
    turtle.penup()
    turtle.forward(300)
    turtle.shape("turtle")
    turtle.right(90)
    turtle.exitonclick()
    
main()
