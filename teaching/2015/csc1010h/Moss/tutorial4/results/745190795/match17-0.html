def main():
    distance = {}
    nameList=[]
    hi =0
    count = 1
    while True:
        comp = input("Competitor no. {}:\n".format(count))
        
        if comp=="":break
        else:
            nameList.append(comp) 
        count+=1   
    print("Please enter the distances for each competitor.")   
    foul = 0
    Highest=0
    att=0
    for i in nameList:
               ##print(i)
        print("Competitor",i+":")
       
        for j in range(1,4):
            Attempt=input("Attempt {}:\n".format(j))
           
            if Attempt.startswith('f') or Attempt.startswith("F")==True: foul+=1
            else:
                att = eval(Attempt)
            if foul==3:
                Highest=0
                
                
            if Highest<att:
                Highest=att  
            distance[i]=Highest
    print('{0:<25} {1:<15}'.format("\nCompetitor","Longest Distance"))                
    for c in range(len(nameList)):
                if distance[nameList[c]]==0:
                    
                    print("{0:<24} {1:<14}".format(nameList[c],"No jump"))
                else:
                    
                    print("{0:<24} {1:<14.2f}".format(nameList[c],distance[nameList[c]]))
                    
                if distance[nameList[c]]>hi:
                    hi=distance[nameList[c]]
                    name = nameList[c]
    print("\nThe winner is {0} with a jump of {1:.2f}.".format(name,hi))                
       
main()
