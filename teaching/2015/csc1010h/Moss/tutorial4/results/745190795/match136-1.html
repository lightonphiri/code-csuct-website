def longjump():
    print("***** Long Jump Information System *****")
    print("Please enter the names of competitors. (Press return when done.)")
    compNum = 0
    name = input("Competitor no. 1:\n")
    names = []
    while name != "":
        compNum += 1
        names.append(name)
        name = input("Competitor no. {0}:\n".format(compNum+1))
    print("Please enter the distances for each competitor.")
    attempts = []
    for i in range(compNum):
        print("Competitor {0}:".format(names[i]))
        attempt = []
        for j in range(3):
            jump = input("Attempt {0}:\n".format(j+1))
            if jump == "foul":
                jump = 0
            attempt.append(float(jump))
        attempts.append(attempt)
    print("\nCompetitor               Longest Distance")
    bests = []
    for i, competitor in enumerate(names):
        distance = max(attempts[i])
        bests.append(distance)
        if distance == 0:
            distance = "No jump"
        else:
            distance = "{0:.2f}".format(distance)
        print("{0:24} {1}".format(competitor, distance))
    print("\nThe winner is {0} with a jump of {1:.2f}.".format(names[bests.index(max(bests))], max(bests)))

import random
import string

def main():
    print("Guess the mystery 4 letter code. (No letter occurs more than once.)")
    difficulty = int(input('Choose difficulty level (4 to 26):\n'))
    letters = list(string.ascii_lowercase)[0: difficulty]
    print('The letters in the code are chosen from: {}.'.format(letters))
    secret_code = []


    for i in range(4):
        rand_index = random.randint(0, len(letters)-1)
        letter = letters.pop(rand_index)
        secret_code.append(letter)
    for i in range(10,0,-1):
        niks = 0
        naks = 0
        print("You have {0} guesses remaining.".format(i))
        guess = input("Enter your guess:\n")
        if guess == "".join(secret_code):
            print("Correct!")
            break
        for idx, letter in enumerate(guess):
            if letter == secret_code[idx]:
                niks += 1
            elif letter in secret_code:
                naks += 1
        out1 = "Nik" if niks == 1 else "Niks"
        out2 = "Nak" if naks == 1 else "Naks"
        out3 = "is" if niks == 1 else "are"
        print("There {0} {1} {2} and {3} {4}.".format(out3, niks, out1, naks, out2))
    else:
        print("Sorry, you lose.")
        print("The secret number was {0}.".format("".join(secret_code)))

if __name__ == '__main__':
