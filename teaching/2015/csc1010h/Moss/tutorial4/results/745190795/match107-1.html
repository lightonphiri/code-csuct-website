        name_no += 1
        competitor.append(name)
        
       
    print("Please enter the distances for each competitor.")
    max_attempt = 0
    max_values =[]
    final_winner = {}
    for i in competitor:
        print("Competitor",i+":")
        for c in range(3):
            attempts = (input("Attempt {}:\n".format(c+1)))
            #if attempts == "":break
            attempts = attempts.lower()
            if attempts == "foul":
                max_values.append(0)
                final_winner[i] =max( max_values)
            else:
                attempts = eval(attempts)
                max_values.append(attempts)          
                final_winner[i] = max(max_values)
                
                if attempts > max_attempt:
                    max_attempt = attempts
        max_values =[]
    count=0
    w=""
    print()


    print("{0:<25}{1:<15}".format("Competitor","Longest Distance         "))

            print("{0:<25}{1:<15.2f}".format(competitor[i],float(final_winner[competitor[i]])))

import turtle
def draw_multicolor_square(t, sz):
    """Make turtle t draw a multi-color square of sz."""
    for i in ["red", "purple", "hotpink", "blue"]:
        t.color(i)
        t.forward(sz)
        t.left(90)

wn = turtle.Screen()        
wn.bgcolor("lightgreen")

tess = turtle.Turtle() 
tess.pensize(3)

size = 20                
for i in range(15):
    draw_multicolor_square(tess, size)
    size = size + 10        
    tess.forward(10)        
    tess.right(18)          

wn.mainloop()
