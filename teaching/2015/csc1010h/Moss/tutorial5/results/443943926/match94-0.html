def is_leap_year(year):
    if year%4!=0:       
        return False
    
    if year%100==0 and year%400!= 0:   
        return False
    
    if year%100==0 and year%400==0:     #if year evenly divisible by 100, is leap year when evenly divisible by 400 also
            return True    

    if year%4==0:                       #leap year if year is evenly divisible by 4   
        return True

def code_for(letter):
    letter= str(letter).lower()
    code={'a':'.-','b':'-...','c':'-.-.','d':'-..','e':'.',
          'f':'..-.','g':'--.','h':'....','i':'..','j':'.---',
          'k':'-.-','l':'.-..','m':'--','n':'-.','o':'---',
          'p':'.--.','q':'--.-','r':'.-.','s':'...','t':'-',
          'u':'..-','v':'...-','w':'.--','x':'-..-','y':'-.--','z':'--..'}
    
    return code[letter]

def transmit_letter(letter):
    letter_string=str(code_for(letter))
    count=0
    for x in letter_string:
        if x=='.':
            morse_utils.beep(1000,100)
                    
        elif x=='-':
            morse_utils.beep(1000,300)        
        
        print(x, end='')
        count+=1
        if len(letter_string)!=count:


            morse_utils.sleep(0.1)
            print('^', end='')

def transmit_word(word):
    count=0
    for y in word:
        transmit_letter(y)
        count+=1
        if len(word)!=count:

responses={
    'bluetooth':'Have you tried mouthwash?',
    'windows':'Ah, I think I see your problem. What version?',
    'apple':'You do mean the computer kind?',
    'blue':'Ah, the blue screen of death. And then what happened?',
    'spam':'You should see if your mail client can filter messages.',
    'connection':'Contact Telkom.',
    'crashed':'Are the drivers up to date?',
    'hacked':'You should consider installing anti-virus software.'}


def default_response():
    return 'Curious, tell me more.'
    
def get_response(words):
    for word in words:
        if word.lower() in responses:       
                                            # word.lower makes sure cases you type in dont matter
            return word.lower()
           
    return str(words).lower()
    

def print_welcome():
    print('Welcome to the automated technical support system.')
    print('Please describe your problem:')
    

def get_input():
    query = input().split(' ')
    return query

def main():
    print_welcome()
    query = get_input()
    while True:          
        response=get_response(query)
        if response in responses: print(responses[response])
        elif response=="['quit']":
            break
        else:
            print(default_response())
        query=get_input()
        
main()
