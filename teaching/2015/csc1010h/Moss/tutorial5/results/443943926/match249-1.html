import calendar_utils


def to_string(n, e):
        if n<1 or n>e:
                return '  '
        else:
                return '{:>2}'.format(n)
        
def row(i, s, e):
        start = i*7+s
        result=to_string(start, e)
        for column in range(start+1, start+7):
                result+=' '+to_string(column, e)  
        return result

def row_of_months(i, start_month, year):
        
        start_day=(calendar_utils.first_day_of_month(start_month, year)-1)%7
        days=calendar_utils.days_in_month(start_month, year)
        
        result=row(i, 1-start_day, days)

        for month in range(start_month+1, start_month+3):
                result+='   '
                start_day=(start_day+days)%7
                days=calendar_utils.days_in_month(month, year)
                result+=row(i, 1-start_day, days)

        return result

def print_year(year):
        print('{0:^69}'.format(year))

def is_leap_year(year):
    if year%4==0 and year%100!=0:
            if year%400==0:
                return True


    elif year%4!=0:
            return False   

def month_name(number):
    monthname = { 1: "January",2: "February",3:"March",4:"April",5:"May",6:"June",7:"July",8:"August",9:"September",10:"October",11:"November",12:"December"}
    return monthname.get(number)

def days_in_month(month_number, year):
    if(month_number==2):
        if( is_leap_year(year)):
            return 29
        else:
            return 28
    elif(month_number==1 or month_number==3 or month_number==5 or month_number==7 or month_number==8 or month_number==10 or month_number==12 ):
        return 31
    else:
        return 30
    
def first_day_of_year(year):
    a=(year-1)%4
    b=(year-1)%100
    c=(year-1)%400
    day=(5*a+4*b+6*c)%7
    return day

def first_day_of_month(month_number, year):
    first_jan= first_day_of_year(year)
    sum_days=first_jan
    for i in range(1,month_number+1):
        sum_days=sum_days+days_in_month(i, year)
    return sum_days%7
