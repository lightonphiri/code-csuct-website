def is_leap_year(year):
         if (year)%4==0 and (year)%100!=0:
                  return True
         elif year%4==0 and (year)%400==0:
                  return True
         else:
                  return False                 
 

def month_name(number):

def days_in_month(month_number, year):
         if month_number == 1 or month_number== 3 or month_number== 5 or month_number==7 or month_number==8 or month_number== 10 or month_number ==12:
                  return 31

         if not is_leap_year(year) and month_number==2:
                  return 28


         elif is_leap_year(year) and month_number==2:
                  return 29
         else:
                  return 30
          
           
           
                                      

def first_day_of_year(year):
         first_day_of_year=int((1+5*((year-1)%4)+4*((year-1)%100)+6*((year-1)%400))%7)
         return first_day_of_year
         
def first_day_of_month(month_number,year):
         total_days=1
         day1=['Sunday','Monday','Tuesday', 'Wednesday','Thursday','Friday','Saturday']
         day_calc={'Sunday':0,'Monday':1,'Tuesday':2, 'Wednesday':3,'Thursday':4,'Friday':5,'Saturday':6}
         for p in range(month_number-1):
                  
                  sum_days=days_in_month(p+1,year)
                  total_days+=sum_days
                  first_day = first_day_of_year(year)+(total_days)%7-7
                  return day_calc[day1[first_day]]

responses={
    'bluetooth':'Have you tried mouthwash?',
    'windows':'Ah, I think I see your problem. What version?',
    'apple':'You do mean the computer kind?',
    'blue':'Ah, the blue screen of death. And then what happened?',
    'spam':'You should see if your mail client can filter messages.',
    'connection':'Contact Telkom.',
    'crashed':'Are the drivers up to date?',
    'hacked':'You should consider installing anti-virus software.'}


def default_response():
    return 'Curious, tell me more.'
    
def get_response(words):
    for word in words:
        if word in responses:
            return responses[word]
    return default_response()
    

def print_welcome():
    print('Welcome to the automated technical support system.')
    print('Please describe your problem:')
    
def get_input():
    query = input().lower().split(' ')
    return query

def main():
    print_welcome()
    query = get_input()
    while True:
        if query == ['quit']:break
        response = get_response(query)
        print(response)
        query = get_input()

        
main()
