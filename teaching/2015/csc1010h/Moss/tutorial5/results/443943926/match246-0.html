def to_string(n, e):
        if n<1 or n>e:
                return '  '
        else:
                return '{:>2}'.format(n)
        
def row(i, s, e):
        start = i*7+s
        result=to_string(start, e)
        for column in range(start+1, start+7):
                result+=' '+to_string(column, e)  
        return result

                result+='   '
                start_day=(start_day+days)%7
                days=calendar_utils.days_in_month(month, year)
                result+=row(i, 1-start_day, days)

def is_leap_year(year):     
      
   if (year%4==0 and year%100!=0 ):
         return True  


   elif year%4==0 and year%400==0 and year%100==0:
         return True
   else:
         return False

   month={1:'January',2:'February',3:'March',4:'April',5:'May',6:'June',7:'July',8:'August',9:'September',10:'October',11:'November',12:'December'}
   return (month[month_number])

def first_day_of_year(year):
   day=1+5*((year-1)%4)
   day+=4*((year-1)%100)
   day+=6*((year-1)%400)
   day=day%7
   return day

def code_for(letter):

    alphabet={'a':'.-','b':'-...','c':'-.-.','d':'-..','e':'.','f':'..-.',
               'g':'--.','h':'....','i':'..','j':'.---','k':'-.-','l':'.-..',
               'm':'--','n':'-.','o':'---','p':'.--.','q':'--.-','r':'.-.',
               's':'...','t':'-','u':'..-','v':'...-','w':'.--','x':'-..-',
               'y':'-.--','z':'--..'}
    return alphabet[letter]


def transmit_letter(letter):
    sound_sequence=''
    
    for dot in code_for(letter):
        sound_sequence+=dot+'^'
    alpha=sound_sequence[:-1]
    
    for sound in alpha:
        print(sound,end='')
