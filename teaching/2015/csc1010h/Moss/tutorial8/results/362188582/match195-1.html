class Lecturer:
    
    def __init__(self, n='unknown', sn='unknown'):
        self.name = n
        self.staff_number = sn
        self.courses = []
    
    def __str__(self):
        s = self.name + ', ' + self.staff_number
        if len(self.courses)>0:
            s += ': '+str(self.courses[0])
            for c in self.courses[1:]:
                s += '; ' + str(c)
        return s+'.'
    
    def add_course(self, c):
        self.courses.append(c)
    
    def input(self):
        self.name = input('Enter name: ')
        self.staff_number = input('Enter staff number: ')

    def display(self):
        print('Name: ' + self.name)
        print('Staff number: ' + self.staff_number)

import sortsearch
from time import*

def main():
    size=int(input("Enter the size of the list:\n"))
    print("Choose the sort method:")
    print("1.Heap sort.")
    print("2.Bubble sort.")
    response=int(input())
    lst=sortsearch.generate_number_list(size)
    print("Before:",lst)
    start_time=clock()
    if response==1:
        sortedlst=sortsearch.heap_sort(lst)
    elif response==2:
        sortedlst=sortsearch.bubble_sort(lst)
    end_time=clock()
    time_taken=end_time-start_time
    print('After:',sortedlst)
    print('Time:',time_taken)
