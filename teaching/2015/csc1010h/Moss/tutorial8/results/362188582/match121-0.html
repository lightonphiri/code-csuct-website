class Lecturer:
    
    def __init__(self, n = "unknown", num = 00000000):
        self.name = n
        self.stuff_num = num
        self.courses = []
        
    def __str__(self):
        s = self.name + ', ' + self.stuff_num
        if len(self.courses)>0:
            s += ': '+str(self.courses[0])
            for c in self.courses[1:]:
                s += '; ' + str(c)
        return s+'.'
    
    def add_course (self, c):
        self.courses.append(c)
