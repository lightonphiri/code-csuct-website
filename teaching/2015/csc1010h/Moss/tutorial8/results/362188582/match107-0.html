class Lecturer:
    
    def __init__(self, n='unknown', sn='unknown'):
        self.name = n
        self.staff_number = sn
        self.course_lecturing = []
        
    def __str__(self):
        d = self.name + ', ' + self.staff_number
        if len(self.course_lecturing)>0:
            d += ': '+str(self.course_lecturing[0])
            for i in self.course_lecturing[1:]:
                d += '; ' + str(i)
        return d+'.'        
    
    def add_courselecturing(self,d):
        self.course_lecturing.append(d)
        
    def input(self):
        self.name = input('Enter name: ')
        self.staff_number = input('Enter staff number: ')
        
    def display(self):
        print('Name: ' + self.name)
        print('Staff number: ' + self.staff_number)

from time import*
from sortsearch import*
def main():
    list_size = int(input('Enter the size of the list:\n'))
    sort = int(input('Choose the sort method:\n1. Heap Sort.\n2. Bubble Sort.\n'))
    
    size = generate_number_list(list_size)
    if sort ==2: #Bubble sort
        print('Before:',size)
        start_time = clock()
        print('After:',bubble_sort(size)) 
        end_time = clock()
        time_taken = end_time -start_time
        print('Time:',time_taken)
    else: #Heap sort
        print('Before:',size)
        start_time = clock()
        print('After:',heap_sort(size)) 
        end_time = clock()
        time_taken = end_time -start_time
        print('Time:',time_taken)  
main()
