class Lecturer:
    
    def __init__(self, n = 'unknown', sn = 'unkown'):
        self.name = n
        self.staff_number = sn
        self.courses_lectured = []
        
    def __str__(self):
        s = self.name + ', ' + self.staff_number
        if len(self.courses_lectured) > 0:
            s += ': ' + str(self.courses_lectured[0])
            for c in self.courses_lectured[1:]:
                s += '; ' + str(c)
        return s + '.'
    
    def add_course(self, c):
        self.courses_lectured.append(c)
        
    def input(self):
        self.name = input('Enter name: ')
        self.staff_number = input('Enter staff number: ')
