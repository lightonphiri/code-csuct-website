from sortsearch import *
from time import *
list_length = eval(input("Enter the size of the list:\n"))
print("""Choose the sort method:
1. Heap Sort.
2. Bubble Sort.""")
generate_random_list = generate_number_list(list_length)
your_option = eval(input())
print("Before:",generate_random_list)
begin = time()
if your_option == 2:
    sorted_list = bubble_sort(generate_random_list)
elif your_option == 1:
    sorted_list = heap_sort(generate_random_list)
end = time()
print("After:",sorted_list)
print("Time:",end-begin)
