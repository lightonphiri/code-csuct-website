class Lecturer:
    def __init__(self,n = "unknown", s = "unknown"):
        self.name = n
        self.courses = []
        self.staff_number = s
    def __str__(self):
        s = self.name+','+self.staff_number
        for b in self.courses:
            s+=str(b)
        return s
    def add_course(self,b):
        self.courses.append(b)
    def input(self):
        self.name = input('Enter name:')
        self.staff_number = input('Enter staff number:')
        
    def display(self):
        
        print('Name:' + self.name)
        
        print('Staff Number:' + self.staff_number)
