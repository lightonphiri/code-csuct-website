class Lecturer:
    
    def __init__(self,name='unkwown',staff_number=0):
        self.name = name
        self.staff_number = staff_number
        self.lectured_courses = []
        
    def __str__(self):
            s = self.name + ', ' + str(self.staff_number)
            if len(self.lectured_courses)>0:
                s += ': '+str(self.lectured_courses[0])
                for c in self.lectured_courses[1:]:
                    s += '; ' + str(c)
            return s+'.'
        
    def add_course(self, c):
        self.lectured_courses.append(c)

        self.name = input('Enter name: ')
