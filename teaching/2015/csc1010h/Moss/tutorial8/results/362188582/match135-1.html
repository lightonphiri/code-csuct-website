class Lecturer:
    def __init__(self,n="unknown",num="unknown"):
        self.staff_name =n
        self.staff_no = num
        self.courses= []
        
    def add_lecturer_course(self,a):
        self.courses.append(a)

    def __str__(self):
        str_to_print = self.staff_name + ', ' + self.staff_no        
        if len(self.courses)>0:
            str_to_print += ': '+str(self.courses[0])
            for a in self.courses[1:]:
                str_to_print += '; ' + str(a)
            return str_to_print+'.'        

    
    def input(self):
        self.staff_name=input("Enter name: ")


        self.staff_no=input("Enter staff number: ")
        
    def desplay(self):
        print("Lecturer name: "+self.staff_name)
        print("Staff number: "+self.staff_no)
