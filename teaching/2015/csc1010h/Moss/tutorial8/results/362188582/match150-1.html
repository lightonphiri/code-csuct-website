            lecturers[l.staff_number] = l

            st.input()
            students[st.student_number] = st
        elif aa == 4:
            stunum = input("Enter student number: ")

                p = courses[coursecode]
                k.add_course(p)

            staffnum = input("Enter staff number: ")
