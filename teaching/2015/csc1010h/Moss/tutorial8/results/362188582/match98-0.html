class Course:

    def __init__(self, cc='unknown', d='unknown'):
        self.course_code = cc
        self.description = d

    def __str__(self):
        return self.course_code + ', ' + self.description

    def input(self):
        self.description = input('Enter description: ')
        self.course_code = input('Enter course code: ')
        
    def display(self):
        print('Course code: ' + self.course_code)

class Lecturer:
    
    def __init__(self,n='unknown',sfn='unknown'):
        self.name=n


        self.staff_number=sfn
        self.courses=[]
        
    def __str__(self):
        s=self.name +', ' + str(self.staff_number)
        if len(self.courses)>0:
            s += ': '+str(self.courses[0])
            for c in self.courses[1:]:
                s += '; ' + str(c)
        return s+'.'

method=int(input())

alist=sortsearch.generate_number_list(size)
print('Before:',alist)

if method==1:
    start=time()
    sortsearch.heap_sort(alist)
    stop=time()
    time_elapsed=stop-start
    print('After:',alist)
