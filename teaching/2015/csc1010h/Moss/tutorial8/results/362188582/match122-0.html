class Lecturer:
    def __init__ (self, name = 'unknown', staff_number = 0):
        self.name = name
        self.staff_number = staff_number
        self.courses = []
    def __str__(self):
            s = self.name + ', ' + self.staff_number
            if len(self.courses)>0:
                s += ': '+str(self.courses[0])
                for c in self.courses[1:]:
                    s += '; ' + str(c)
            return s+'.'
    def add_course (self, c):
        self.courses.append(c)
    def input(self):
            self.name = input('Enter name: ')
            self.staff_number = input('Enter staff number: ')
    def display(self):
        print('Name: ' + self.name)
        print('Staff number: ' + self.staff_number)

def main():
    import time 
    import sortsearch    
    list_size = eval(input('Enter the size of the list:\n'))
    print('Choose the sort method:')
    print('1. Heap Sort.')
    print('2. Bubble Sort.')
    sort_method = eval(input())
    if sort_method == 1:
        unsorted_list = sortsearch.generate_number_list(list_size)
        print('Before:',unsorted_list)
        time1 = time.clock()
        sorted_list = sortsearch.heap_sort(unsorted_list)
        print('After:',sorted_list)
        time2 = time.clock()
        final_time = time2 - time1
        print('Time:','{0:.9f}'.format(final_time))
    elif sort_method == 2:
        unsorted_list2 = sortsearch.generate_number_list(list_size)
        print('Before:',unsorted_list2)
        time3 = time.clock()
        sorted_list2 = sortsearch.bubble_sort(unsorted_list2)
        print('After:',sorted_list2)
        time4 = time.clock()
        final_time = time4 - time3
        print('Time:','{0:.9f}'.format(final_time))        
    else:
        print('Invalid input supplied. Try again!!')
        print('-----------------------------------')
        main()
