class Lecturer:
    
    def __init__(self, n='unknown', sn='unknown'):
        self.name = n
        self.staff_number = sn
        self.courses_l = []
    
    def __str__(self):
        s = self.name + ', ' + self.staff_number
        if len(self.courses_l)>0:
            s += ': '+str(self.courses_l[0])
            for c in self.courses_l[1:]:
                s += '; ' + str(c)
        return s+'.'
        
    def add_course(self, c):
        self.courses_l.append(c)    
    
    def input(self):
        self.name = input('Enter name: ')

from time import *
import sortsearch

def main():
    size=int(input('Enter the size of the list:\n'))
    print('Choose the sort method:')
    print('1. Heap Sort.')
    print('2. Bubbble Sort.')
    method=int(input())
    list_to_sort = sortsearch.generate_number_list(size)
    print('Before:',list_to_sort)
    start_time = time()	
    if method==1:
        sorted_list= sortsearch.heap_sort(list_to_sort)
    elif method==2:
        sorted_list= sortsearch.bubble_sort(list_to_sort)
    end_time = time()	 
    time_taken = end_time - start_time
    print('After:', sorted_list)
    print('Time:', time_taken)
