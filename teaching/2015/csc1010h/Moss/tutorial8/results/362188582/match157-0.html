class Lecturer:

          self.stuff_number = stuff_no
          self.courses = []
     
     def __str__(self):
          v = self.name + ', ' + self.stuff_number
          if len(self.courses)>0:
               v += ': '+str(self.courses[0])
               for course in self.courses[1:]:
                    v += '; ' + str(course)

class Student:
    
    def __init__(self, n='unknown', sn='unknown'):
        self.name = n
        self.student_number = sn
        self.courses = []
    
    def __str__(self):
        s = self.name + ', ' + self.student_number
        if len(self.courses)>0:
            s += ': '+str(self.courses[0])
            for c in self.courses[1:]:
                s += '; ' + str(c)
        return s+'.'
    
    def add_course(self, c):
        self.courses.append(c)
    
    def input(self):
        self.name = input('Enter name: ')
        self.student_number = input('Enter student number: ')

    def display(self):
        print('Name: ' + self.name)
        print('Student number: ' + self.student_number)
