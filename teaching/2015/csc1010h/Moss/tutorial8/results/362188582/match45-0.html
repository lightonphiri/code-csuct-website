class Student:
    
    def __init__(self, n='unknown', sn='unknown'):
        self.name = n
        self.student_number = sn
        self.courses = []
    
    def __str__(self):
        s = self.name + ', ' + self.student_number
        if len(self.courses)>0:
            s += ': '+str(self.courses[0])
            for c in self.courses[1:]:
                s += '; ' + str(c)
