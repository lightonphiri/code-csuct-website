from time import *
from random import *


def generate_number_list(size): # generates list of random numbers
    lst = []
    for i in range(size):
        lst.append(randint(0,size-1))
    return lst

def heap_sort(lst):
    # Heapify
    start = ((len(lst)-2)//2)
    while start>=0:
        siftDown(lst, start, len(lst)-1)
        start=start-1
    end=len(lst)-1
    # Sort
    while end>0:
        lst[0], lst[end] = lst[end], lst[0]
        end=end-1
        siftDown(lst, 0, end)
    return lst

def siftDown(lst, start, end):
    root = start
    
    while root*2+1<=end:
        child = root*2+1
        swap = root
        
        if lst[swap]<lst[child]:
            swap = child
        if child+1<=end  and lst[swap]<lst[child+1]:
            swap = child+1
        if swap==root:
            return
        else:
            lst[root], lst[swap]=lst[swap], lst[root]
            root = swap
            
def bubble_sort(lst):
    swapped = True
    top = len(lst) - 1
    while top > 0 and swapped == True:
        i = 0
        swapped = False
        while i < top:
            if lst[i] > lst[i+1]:
                swapped = True
                lst[i],lst[i+1] = lst[i+1],lst[i] # swap
            i += 1
        top -= 1
    return lst

def sorttester():
    size=int(input("Enter the size of the list:\n"))
    f = input("Choose the sort method:\n1. Heap sort.\n2. Bubble Sort.\n")
    lst1 = generate_number_list(size)
    print("Before: ",lst1)
    start_time = clock()


    if f == "1":
        sort = heap_sort(lst1)
    elif f == "2":
        sort = bubble_sort(lst1)
    end_time = clock()
    time_taken = end_time - start_time
    print("After: ",sort)
    print("Time:",time_taken)
    
sorttester()
