class Lecturer:
    
     def __init__(self, n='unknown', sn='unknown'):
          self.name = n
          self.stuff_number = sn
          self.courses = []
     
     def __str__(self):
          s = self.name + ', ' + self.stuff_number
          if len(self.courses)>0:
               s += ': '+str(self.courses[0])
               for c in self.courses[1:]:
                    s += '; ' + str(c)
          return s+'.'
  
     def add_course(self, c):
          self.courses.append(c)
       
     def input(self):
          self.name = input('Enter name: ')


          self.stuff_number = input('Enter staff number: ')
 
     def display(self):
          print('Name: ' + self.name)
          print('Staff number: ' + self.stuff_number)

from student import Student
from course import Course
from lecturer import Lecturer

def start():
    print("""University Member System: 
0. Quit.
1. Input Course.
2. Input Lecturer.
3. Input Student.
4. Enrol Student on Course.
5. Assign Course to Lecturer.
6. Find Course.
7. Find Lecturer.
8. Find Student.
9. Display Courses.
10. Display Lecturers.
11. Display Students.""")


start()
select = eval(input(">"))

lec = {}
stu = {}
cos = {}

while select!=0 :
    
    if select == 1 :  # Adding a course
        coz = Course()
        coz.input()
        cos[coz.course_code] = coz
        
    elif select == 2 : # adding a lecturer
        lect = Lecturer()
        lect.input()
        lec[lect.stuff_number] = lect
        
    elif select == 3 : # adding a student
        stud = Student()
        stud.input()
        stu[stud.student_number] = stud
    
    elif select == 4 : # enrol studdent on a course
        stu_num = input("Enter student number: ")
        cos_code = input("Enter course code: ")
        
        if stu_num not in stu:
            print("Could not find a student with that student number.")
            
        else:
            stu[stu_num].add_course(cos[cos_code])
            
    elif select == 5 :  # lecture to a course
        stuff_num = input("Enter staff number: ")
        cos_code = input("Enter course code: ")
                
        if stuff_num not in lec:
            print("Could not find a lecture with that staff number.")
                    
        else:
            lec[stuff_num].add_course(cos[cos_code])    
            
    elif select == 6 : # find a course
        cos_code = input("Enter course code: ")
        
        if cos_code not in cos:
            print("Could not find a course with that code.")
        
        else:
            print(cos[cos_code])
        
    elif select == 7 : # find a lecture 
        stuff_num = input("Enter staff number: ")
                
        if stuff_num not in lec:
            print("Could not find a member of staff with that number.")
                
        else:
            print(lec[stuff_num]) 
            
    elif select == 8 : # find student
        stu_num = input("Enter student number: ")
                  
        if stu_num not in stu:
            print("Could not find a student with that student number.")
                    
        else:
            print(stu[stu_num])
    
    elif select == 9 : # display course:
        for coz in cos :
            cos[coz].display()
    
    elif select == 10 : # display lecture
        for lect in lec :
            lec[lect].display()
    
    elif select == 11 : # display student
        for stud in stu :
            stu[stud].display()
            
    else:
        print("That selection was not recognised.")
        
    start()
    select = eval(input(">"))
