    print("Choose the sort method:")
    print("1. Heap Sort.")
    print("2. Bubble Sort.")
    option=input()
    if option=="1":
        heap=generate_number_list(size)
        print("Before:",heap)
        time_before=time.clock()
        y=heap_sort(heap)
        time_after=time.clock()
        print("After:",y)
        td=time_after-time_before
        print('Time: {0:.9f}'.format(td))
        
    elif option=="2":
        bubble=generate_number_list(size)

        y=bubble_sort(bubble)
        time_after=time.clock()
