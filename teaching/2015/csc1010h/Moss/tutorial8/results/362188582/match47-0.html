class Lecturer:

    def __init__(self, n='unknown', sn='unknown'):
        self.name = n
        self.staff_number =sn
        self.courses = []
    
    def __str__(self):
        s = self.name + ', ' + self.staff_number
        if len(self.courses)>0:
            s += ': '+str(self.courses[0])
            for c in self.courses[1:]:
                s += '; ' + str(c)
        return s+'.'    
    
    def add_course(self, c):

import time
import sortsearch 
 
def main():
    size=int(input('Enter the size of the list:\n'))
    method=int(input('Choose the sort method:\n1. Heap Sort.\n2. Bubble Sort.\n'))
    if method==1:
        list1=sortsearch.generate_number_list(size)
        print('Before:' ,list1)
        start_time=time.clock()
        after=sortsearch.heap_sort(list1)
        end_time=time.clock()
        time_taken = end_time-start_time
        print('After:',after)
        
    elif method==2:
        list2=sortsearch.generate_number_list(size)
        print('Before:',list2)
        start_time=time.clock()
        after=sortsearch.bubble_sort(list2)
        end_time=time.clock()
        time_taken = end_time-start_time
        print('After:',after)

    print('Time: {:.9f}'.format(time_taken))
    
main()
