def account():
    card_account=' Credit Card Account '
    name=input("Enter the card holder's name:\n")                               #prompt user,name
    card_number=input("Enter the 16 digit card number (no spaces please):\n")   #prompt user, card number
    date_of_issue=input("Enter the date of issue (mm:yy):\n")                   #prompt user, date of issue


    expiry_date=input("Enter the expiry date (mm:yy):\n")                       # prompt user, expiry date
    security_code=input("Enter the 3 digit security code:\n")                   # prompt user,security code
    credit_limit=float(input("Enter the credit limit (Rand):\n"))               #prompt user, credit limit
    balance=float(input("Enter the balance (Rand):\n"))                         #prompt user, balance
    
    available=credit_limit- balance                                             #calculate available amount
    
    decimal_credit_limit='{:.2f}'.format(credit_limit)    #round off amounts two demical places
    decimal_available='{:.2f}'.format(available) 
    decimal_balance='{:.2f}'.format(balance)
   
    date_of_issue= date_of_issue.replace(':','/')         #change symbol 
    expiry_date= expiry_date.replace(':','/')
    
    card_number_1= card_number[:4]                        # create spaces in between numbers,(string slicing)
    card_number_2= card_number[4:8]
    card_number_3= card_number[8:12]
    card_number_4= card_number[12:]
                                            
    print('+{0:-^48}+'.format(card_account))                                               #print and format results
    print('| {1:<15}{0:<32}|'.format(name,'Card holder:'))                                 #print card holder
    print('| {:<15}'.format('Card number:')+'{:<5}'.format(card_number_1)+'{:<5}'.format(card_number_2)+'{:<5}'.format(card_number_3)+'{:<17}|'.format(card_number_4))                                                     #print card number
    print('| {1:<15}{0:<8}'.format(date_of_issue,'Date of issue:'),'Expiry date: {0:<10}|'.format(expiry_date))  #print date of issue and expiry date
    print('| {1:<15}{0:<32}|'.format(security_code,'Security code:'))                      #print security code
    print('| {1:<15}{0:<32}|'.format(str(decimal_credit_limit) + ' Rand','Credit limit:')) #print credit limit
    print('| {1:<15}{0:<32}|'.format(str(decimal_available) + ' Rand','Available:'))       #print available
    print('| {1:<15}{0:<32}|'.format(str(decimal_balance) +' Rand','Balance:'))            #print balance
    print('+{0:-^48}+'.format(''))

def hexagon():
    import math
    length=eval(input("Enter the length of the side (cm):\n")) #prompt user,length
    area=((3*math.sqrt(3))/2)*(length**2)                      # calculate area
    print("The area of a hexagon of side {0} is {1:.3f}.".format(length,area)) #print results
