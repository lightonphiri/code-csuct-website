# calculate the area of a hexagon
# lutho nduneni
# 24 april 2015

def main():
    import math
    # get input from user
    side=eval(input("Enter the length of the side (cm):\n"))
    area=((3*(math.sqrt(3)))/2)*(side**2)
    # make three spaces available for decimals
    area=round(area,3)
    print("The area of a hexagon of side",side,"is","{0:.3f}".format(area),end=".")

import turtle
def main():
    # state pen size
    turtle.pensize(1)
    # pick up the pen and move it where it is to begin 
    turtle.penup()
    # change pen colour
    turtle.pencolor("green")
    # move pen to start position
    turtle.forward(200)
    turtle.left(90)
    turtle.forward(120)
    # put pen down and start to draw


    turtle.pendown()
    turtle.left(90)
    turtle.forward(200)
    turtle.left(90)
    turtle.forward(100)
    # pick pen up and move it to the second start
    turtle.penup()
    turtle.forward(20)    
    # change the pen colour
    turtle.pencolor("red")
    # put pen down and draw second initial
    turtle.pendown()
    turtle.left(90)
    turtle.forward(200)
    turtle.right(150)
    turtle.forward(223.6067977)
    turtle.left(150)
    turtle.forward(200)
    # exit turtle
    turtle.exitonclick()
main()
