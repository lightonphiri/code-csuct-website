card_name = input("Enter the card holder's name:\n")
card_number = input("Enter the 16 digit card number (no spaces please):\n")
temp = ""
i = 0
while i < 4:
    if i < 3:
        temp += card_number[4*(i):4*(i+1)] + " "
    else:
        temp += card_number[4*(i):4*(i+1)]
    i +=1


date_issue = input("Enter the date of issue (mm:yy):\n")
date_issue = date_issue[0:2] + "/" + date_issue[3:]
expire_date = input("Enter the expiry date (mm:yy):\n")
expire_date = expire_date[0:2] + "/" + expire_date[3:]
three_digit_code = input("Enter the 3 digit security code:\n")

credit_limit = float(input("Enter the credit limit (Rand):\n"))
card_balance = float(input("Enter the balance (Rand):\n"))

print("+------------- Credit Card Account --------------+")
print("| {0:<15}".format("Card holder:"),"{0:<32}|".format(card_name),sep="")
print("| {0:<15}".format("Card number:"),"{0:<32}|".format(temp),sep="")
print("| {0:<15}".format("Date of issue:"),"{0:<32}|".format(date_issue + "    Expiry date:" +" "+ expire_date),sep="")
print("| {0:<15}".format("Security code:"),"{0:<32}|".format(three_digit_code),sep="")
print("| {0:<15}".format("Credit limit:"),"{0:<32}|".format(str(credit_limit) + "0 Rand" ),sep="")
print("| {0:<15}".format("Available:"),"{0:<32}|".format(str(credit_limit-card_balance) + "0 Rand" ),sep="")
print("| {0:<15}".format("Balance:"),"{0:<32}|".format(str(card_balance) + "0 Rand" ),sep="")
print("+------------------------------------------------+")

import math #import maths module to calculate maths related sums   

def main():
    length = int(input("Enter the length of the side (cm):\n")) #prompt user for length of one side of the hexagon 
    area = round((((3*math.sqrt(3))/2)*length**2),3)
    print("The area of a hexagon of side",length, "is", "{:.3f}".format(area), end=".")
