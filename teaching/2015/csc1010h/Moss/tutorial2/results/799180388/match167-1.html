c_3= card_number[8:12]
c_4= card_number[12:16]

z= '{0:<15}'.format("Date of issue:")
date_issue= input("Enter the date of issue (mm:yy): \n")
date_issue = date_issue.replace(":","/")


expiry= input("Enter the expiry date (mm:yy): \n")
expiry = expiry.replace(":","/")

b= '{0:<15}'.format("Security code:")
security_code= int(input("Enter the 3 digit security code: \n"))

c= '{0:<15}'.format("Credit limit:")
credit_limit= float(input("Enter the credit limit (Rand): \n"))

d= '{0:<15}'.format("Balance:")
balance= float(input("Enter the balance (Rand): \n"))

a= '{0:<15}'.format("Available:")
available= credit_limit - balance



head="Credit Card Account"
tail=""
end="|"

print('+' + '-'* 13 + ' ' + head + ' ' + '-'* 14 + '+') 
print("| "+ x + "{0:<32}".format(name) + '|')
print("| "+ y + "{0:<32}".format(c_1+ " "+ c_2+" "+c_3+" "+ c_4) + '|')
print("| "+ z + "{0:<32}".format((date_issue+"    Expiry date: " + expiry ) ) + '|') 
print("| "+ b + "{0:<32}".format(str(security_code)) + '|')
print("| "+ c + "{0:<32}".format((str('{0:2.2f}'.format(credit_limit))+" Rand")) + '|')
print("| "+ a + "{0:<32}".format((str('{0:2.2f}'.format(available))+" Rand")) + '|')
print("| "+ d + "{0:<32}".format((str('{0:2.2f}'.format(balance))+" Rand")) + '|')
print("+{0:-^48}+".format(tail))
