def main():
    
    import math
    v=int(input("Enter the length of the side (cm): \n"))  #prompt for input
    A=math.sqrt(3)#calculation
    B=3*A
    C=B/2
    D=C*v**2
    a=round(D,3)                                           #round off answer
    print("The area of a hexagon of side {} is {}.".format(v,a))    #output-print answer
