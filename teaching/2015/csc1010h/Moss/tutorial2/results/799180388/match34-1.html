    print("+------------- Credit Card Account --------------+")
    print("| Card holder:   "+"{0:32}".format(name_1)+"|")

def main():
    side = eval(input("Enter the length of the side (cm):\n"))
    area = (((3*(math.sqrt(3))/2))*(side**2))
    the_area = '{0:4.3f}'.format(area)
    print("The area of a hexagon of side "+str(side)+" is "+str(the_area)+ ".")
