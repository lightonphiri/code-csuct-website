    cardnumber_2=cardnumber[4:8]
    cardnumber_3=cardnumber[8:12]
    cardnumber_4=cardnumber[12:]
    issuedate=input("Enter the date of issue (mm:yy):\n")
    expirydate=input("Enter the expiry date (mm:yy):\n")
    code=input("Enter the 3 digit security code:\n")
    limit=float(input("Enter the credit limit (Rand):\n"))
    balance=float(input("Enter the balance (Rand):\n"))
    limit_1="{0:.2f}".format(limit)
    balance_1="{0:.2f}".format(balance)
    avail=(limit-balance)
    avail_1="{0:.2f}".format(avail)
    
    
    head=" Credit Card Account "
    print("+{0:-^48}+".format(head))
    print("| {0:<15}{1:<32}|".format("Card holder:",name_1))

def main():
    
    import math
    length=float(input("Enter the length of the side (cm):\n"))
    area=(3*(math.sqrt(3))/2)*(length*length)
    area_1="{0:.3f}".format(area)
    print("The area of a hexagon of side",int(length),"is",area_1,end=".")
    
main()
