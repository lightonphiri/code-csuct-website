def main():
    print('Hello, I am Eugene Junior.')

    last_name=input('What is your last name?\n')
    year=int(input('Hi {}.{}.!\nWhat year is this?\n'.format(first_name[0],last_name[0])))
    age=eval(input('What age are you?\n'))
    year_of_birth=year-age
    str_year_of_birth=str(year_of_birth)
    height=eval(input('Okay, so you were born in {}/{}.\nHow tall are you (metres)?\n'.format(year_of_birth-1,str_year_of_birth[2:4])))
    height_in_feet=math.floor(height*3.28)
    height_in_inch=round(((height*3.28)%height_in_feet)*12)
    mothers_name=input('That\'s {}\'{}"!\nI\'m going to tell you your Star Wars name.\nWhat\'s your mother\'s first name?\n'.format(height_in_feet,height_in_inch))
    city=input('What\'s the name of the city nearest to your place of birth?\n')
    star_war_name1=last_name[:3] + first_name[:2]
    star_war_name2=mothers_name[-(len(mothers_name)-((len(mothers_name)-2))):] + city[-(len(city)-((len(city)-3))):]
    print('Your Star Wars name is',star_war_name1.title(),star_war_name2.title()+".")
    
main()
