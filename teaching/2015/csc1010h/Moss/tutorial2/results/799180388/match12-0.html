# A program to print out my initials using a turtle module


import turtle # allows us to use turtle

def main():
    #pick up the pen and move 80 steps to the left to centrelize the letters    
    turtle.penup()
    turtle.setx(-80)
   
                                            
    turtle.pencolor('red')                    # change the pen color to red   
    turtle.pendown()                          # put the pen down to start drawing the first letter 
    turtle.forward(200)                       # move the pen 200 steps forward(upwards direction)
    turtle.left(90)                           # rotate 90 degrees left
    turtle.forward(50)                        # move 50 steps forward
    turtle.left(180)                          # rotate the pen 180 degrees to face backwards
    turtle.forward(100)                       # move 100 steps forward
    
    turtle.penup()                            # pick up the pen and move 50 steps forward
    turtle.forward(50)
    
    
    
    turtle.pencolor('magenta')                # change the pen color to magenta
    turtle.pendown()                          # put the pen down to draw the second letter
    turtle.right(90)                          # rotate the pen 90 degrees right
    turtle.forward(200)                       # move the pen 200 steps forward
    turtle.left(90)                           # rotate the pen 90 degrees left
    turtle.forward(60)                        # move 50 steps forward
    
    
    # drawing the first curve
    turtle.left(45)                           #rotate 45 degrees left and move 40 steps forward          
    turtle.forward(40)
    turtle.left(45)                           #rotate 45 degrees left and move 143 steps forward
    turtle.forward(143)
                              
    
    
    #drawing the secod curve
    turtle.left(45)                           #rotate 45 degrees left and move 40 steps forward
    turtle.forward(40)
    turtle.left(45)                           #rotate 45 degrees left and move 60 steps forward
    turtle.forward(60)
                              
    
    
    turtle.hideturtle()                       #hide the pen
    
    
    turtle.exitonclick() # Wait for the user to close the window
    
main()
