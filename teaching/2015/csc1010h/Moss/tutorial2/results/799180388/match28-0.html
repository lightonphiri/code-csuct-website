#program to calculate area of a hexagon.

def main():
    #promt user,
    length= float(input ("Enter the length of the side (cm):\n"))
    #area calculation
    import math
    area= ((3* math.sqrt(3)) / 2 * length**2)
    #display answer
    print ('The area of a hexagon of side', int(length), 'is','{0:.3f}.'.format(area))
    
main()
