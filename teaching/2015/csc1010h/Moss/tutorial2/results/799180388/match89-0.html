def main():
    holder=input("Enter the card holder\'s name:\n")

    card_number=input("Enter the 16 digit card number (no spaces please):\n")
    cn1=card_number[0:4]
    cn2=card_number[4:8]
    cn3=card_number[8:12]
    cn4=card_number[12:16]
    cn=cn1+" "+cn2+" "+cn3+" "+cn4

    issue=input("Enter the date of issue (mm:yy):\n")
    m1=issue[0:2]+"/"+issue[3:5]

    expire=input("Enter the expiry date (mm:yy):\n")
    m2=expire[0:2]+"/"+expire[3:5]

    security=input("Enter the 3 digit security code:\n")
    
    limit=input("Enter the credit limit (Rand):\n")
    lm=limit+".00"+" "+"Rand"
    
    balance=int(input("Enter the balance (Rand):\n"))
    b=str(balance)+".00"+" "+"Rand"

    available=int(limit)-int(balance)
    av=str(available)+".00"+" "+"Rand"

    print("+------------- Credit Card Account --------------+")
    print("| Card holder:   {0: <32}".format(holder), "|", sep="")
    print("| Card number:   {0: <32}".format(cn), "|", sep="")
    print("| Date of issue: {0: <8}".format(m1), "Expiry date: {0:9}".format(m2), "|")
    print("| Security code: {0: <32}".format(security), "|", sep="")
    print("| Credit limit:  {0: <32}".format(lm), "|", sep="")
    print("| Available:     {0: <32}".format(av), "|", sep="")
    print("| Balance:       {0:<32}".format(b), "|", sep="")
    print("+------------------------------------------------+")
main()>>>> file: EugeneJunior.py
