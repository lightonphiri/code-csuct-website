import math # call the math module

def main():  # define the funcion main
    side=eval(input("Enter the length of the side (cm):\n")) # ask the user to input the size of the sides
    area=((3*math.sqrt(3))/2)*(side**2)   # formula to calculate the area
    
    print("The area of a hexagon of side {0} is {1:.3f}.".format(side,area))  # output the area
