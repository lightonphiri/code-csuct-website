name=input("Enter the card holder's name:\n")

issue_date=input("Enter the date of issue (mm:yy):\n")
expiry_date=input("Enter the expiry date (mm:yy):\n")
code=input("Enter the 3 digit security code:\n")
limit1=float(input("Enter the credit limit (Rand):\n"))
balance1=float(input("Enter the balance (Rand):\n"))

name1=card_no[:4]


name2=card_no[4:8]
name3=card_no[8:12]
name4=card_no[12:] 

name5=name1+" "+name2+" "+name3+" "+name4

date1=issue_date[:2]
date2=issue_date[3:]
issue_date1=date1+"/"+date2

dt1=expiry_date[:2]
dt2=expiry_date[3:]
expiry_date1=dt1+"/"+dt2

available1=(limit1)-(balance1)

ch="| Card holder:  "         #Card holders name
cn="| Card number:  "         #Card number
di="| Date of issue:"       #Issue date
ed="    Expiry date:"       #expiri date
sc="| Security code:"       #security code
lim="| Credit limit: "       #credit limit
cl="| Available:    "           #available funds
bal="| Balance:      "            #balance

limit='{0:.2f}'.format(limit1)
balance='{0:.2f}'.format(balance1)
available='{0:.2f}'.format(available1)

print("+------------- Credit Card Account --------------+")
print('{0: <48}'.format(ch+" "+ name),"|")
print('{0: <48}'.format(cn+" "+name5),"|")
print('{0: <48}'.format(di+ " "+issue_date1+ed+" "+expiry_date1),"|")
