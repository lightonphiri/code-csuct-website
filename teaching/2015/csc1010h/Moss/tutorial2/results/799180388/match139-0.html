# program to print a credit card account

def main():
    #get user to input account info
    #store as variables
    name= input('Enter the card holder\'s name:\n')
    number= input('Enter the 16 digit card number (no spaces please):\n')
    issue_date= input('Enter the date of issue (mm:yy):\n')
    expiry_date= input('Enter the expiry date (mm:yy):\n')
    code= input('Enter the 3 digit security code:\n')
    limit= float(input('Enter the credit limit (Rand):\n'))
    balance= float(input('Enter the balance (Rand):\n'))
    
    #display credit account
    print("{0}{1:-^48}{2}".format('+',' Credit Card Account ','+'))
    print("| Card holder:{0:3}{1:<32}|".format(" ", name))
    
    '''concatenate strings,print card num with spaces between every 4th num'''
    print("| Card number:{0:3}{1:<32}|".format(" ", number[:4]+' '+number[4:8]+' '+number[8:12]+' '+number[12:]))
    
    '''replace : with / of the issue and expiry date'''
    print("| Date of issue: {0:<9}Expiry date: {1:<10}|".format(issue_date[:2]+'/'+issue_date[3:], expiry_date[:2]+'/'+expiry_date[3:]))
      
    print("| Security code: {0:<32}|".format(code))
    
    '''format limit, available and balance to print with 2 dec places'''
    limit_string= "{0:1.2f}".format(limit)
    print("| Credit limit:{0:2}{1:<32}|".format("", limit_string+' Rand'))
    available_string="{0:1.2f}".format(limit-balance)
    print("| Available:{0:<5}{1:<32}|".format("", available_string+' Rand'))
    balance_string="{0:1.2f}".format(balance)
    print("| Balance:{0:<7}{1:<32}|".format("", balance_string+' Rand'))

    print("{0}{1:-^48}{2}".format('+','','+'))

import math
def main():
    #prompt user for name & surname,store variables
    print('Hello, I am Eugene Junior.')
    name= input('What is your first name?\n')
    name_1= name[:1]
    surname= input('What is your last name?\n')
    surname_1= surname[:1]
    print('Hi', name_1+'.'+surname_1+'.!')          #print greeting + initials
    
    #prompt user for current year & age
    year= int(input('What year is this?\n'))
    age= int(input('What age are you?\n'))
    year_born= str(year-age)
    year_born_1= str(year-age-1)
    year_born_2= year_born[2:]
    print('Okay, so you were born in {}/{}.'.format(year_born_1,year_born_2))   #print year of birth
    
    #prompt user for height in metres
    height= float(input('How tall are you (metres)?\n'))
    '''1 metre = 39.37 inches
       1 foot = 12 inches'''
    feet= str(int((height*39.37)//12))          
    inches= str(math.ceil((height*39.37)%12))       #math(ceil):returns the integer greater than or = to result of inches
    print('That\'s ' + feet + '\'' + inches +'\"!') #print height in feet and inches      
    
    #prompt user for name of mother + nearest city of birth
    print('I\'m going to tell you your Star Wars name.')
    mother= input('What\'s your mother\'s first name?\n')
    city= input('What\'s the name of the city nearest to your place of birth?\n')
    
    star_name= surname[:3] + name.lower()[:2]       #star name- 1st 3 letters of surname,1st 2 letters of name

# program to print the area of a hexagon

import math
def main():
 #prompt user, read length of side of hexagon
 side= eval(input('Enter the length of the side (cm):\n'))
 
 #calc area using formula
 area= (3*math.sqrt(3)*(side**2))/2
 
 #another way to round: (round(area,3))
 
 print('The area of a hexagon of side', side, 'is {0:1.3f}.'.format(area))  #print area result
