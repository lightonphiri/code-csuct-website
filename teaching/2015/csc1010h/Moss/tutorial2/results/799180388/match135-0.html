def main():
    import math
    side =int(input("Enter the length of the side (cm):\n"))
    area = ((3*math.sqrt(3))/2)*side*side #math.sqrt is the square of the number 
    print("The area of a hexagon of side",side,"is","{0:.3f}".format(area)+".") #round is used to round of the area into three decimal places
