    year_born2 = str(year_born -1)


    print("Okay, so you were born in " + year_born2 + "/" + year_born1[2:4] +".")

    feet = math.floor(height)
    inches = round((height - feet)*12)
    print("That's " + str(feet) + "\'" + str(inches) + "\"" + "!")
    print("I'm going to tell you your Star Wars name.")
