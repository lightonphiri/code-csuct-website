def main():
    # obtain details
    name = str(input("Enter the card holder's name: \n"))
    
    name = name.title()
    
    cardno = str(input("Enter the 16 digit card number (no spaces please): \n"))
    
    date_of_issue = str(input("Enter the date of issue (mm:yy): \n"))
    
    expiry_date = str(input("Enter the expiry date (mm:yy): \n"))
    
    security_code = input("Enter the 3 digit security code: \n")
    
    credit_limit = str(input("Enter the credit limit (Rand): \n"))
    
    balance1 = float(input("Enter the balance (Rand): \n"))
    
    # print out details
    
    no_of_spaces1 = 32 - len(name)
    
    print("+------------- Credit Card Account --------------+")
    
   
    
    print("| Card holder:   {}{}|".format(name,no_of_spaces1*' '))
    
    
    
    s1,s2,s3,s4 = cardno[:4],cardno[4:8],cardno[8:12],cardno[12:16]
    
    space = 29 - len(cardno)
    
    print("| Card number:   {} {} {} {}{}|".format(s1,s2,s3,s4,space*' '))
    
   
    
    card_date_of_issue = date_of_issue[0:2] + '/' + date_of_issue[3:5]
    card_expiry_date = expiry_date[0:2] + '/' + expiry_date[3:5]
    
    
    print("| Date of issue: {}    Expiry date: {}     |".format(card_date_of_issue,card_expiry_date))
    
    
    
    print("| Security code: {}                             |".format(security_code))
    
    no_of_spaces2 = 24 - len(credit_limit)
    
    credit_limit = str(float(credit_limit)) + '0'
    
    print("| Credit limit:  {} Rand{}|".format(credit_limit,no_of_spaces2*' '))
    
    
    
    available = str(float(credit_limit) - float(balance1)) + '0'
    
    no_of_spaces3 = 27 - len(str(available))
    
    print("| Available:     {} Rand{}|".format(available,no_of_spaces3*' '))
    
    #balance = str(float(balance)) + '0'
    
    
    balance = str(float(balance1)) + '0'
    
    no_of_spaces4 = 28 - len(str(balance))
    
    print("| Balance:       {} Rand{}|".format(float(balance),no_of_spaces4*' '))
    
    
    
    print("+------------------------------------------------+")
