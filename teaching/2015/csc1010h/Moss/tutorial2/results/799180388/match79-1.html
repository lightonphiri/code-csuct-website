def main():
    
    #Prompt the user to enter credit card details:
    name=str(input("Enter the card holder's name:\n"))
    card_number=str(input("Enter the 16 digit card number (no spaces please):\n"))
    issue_date=str(input("Enter the date of issue (mm:yy):\n"))
    expiry_date=str(input("Enter the expiry date (mm:yy)\n"))
    security_code=str(input("Enter the 3 digit security code:\n"))
    credit_limit=float(input("Enter the credit limit (Rand):\n"))
    balance=float(input("Enter the balance (Rand):\n"))
    
    available=float(credit_limit-balance)     #the available amount is credit limit minus balance
    
    
    print("+{0:-^48}+".format(" Credit Card Account ")) #print title of a credit card with its top boarders
    
    
    #string slicing for card number
    no1=card_number[0:4]         #first four card numbers
    no2=card_number[4:8]         #second four card numbers
    no3=card_number[8:12]        #third four card numbers
    no4=card_number[12:16]       #fourth four card numbers
    card_no=("{} {} {} {}".format(no1,no2,no3,no4))   #format the sliced numbers into one line
    
    
    #display the information entered by the user on a credit card.
    print("| Card holder:{0:3}{1:<25}{2:>8}".format("",name,'|'))
    print("| Card number:{0:3}{1:<19}{2:>14}".format("",card_no,"|"))
    print("| Date of issue:{0:1}{1:<5}    Expiry date: {2:<5}{3:>6}".format("",issue_date,expiry_date,"|"))
    print("| Security code:{0:1}{1:3}{2:>30}".format("",security_code,"|"))
    print("| Credit limit:  {0:<8.2f} Rand{1:>20}".format(credit_limit,"|"))
    print("| Available:     {0:<8.2f} Rand{1:>20}".format(available,"|"))
    print("| Balance:       {0:<7.2f} Rand{1:>21}".format(balance,"|"))
    
    
    print("+{0:-^48}+".format("")) #bottom boarder of a credit card

#program that is given the length of side of a regular hexagon, and it computes and prints the area, rounded and formatted to 3 decimal places.
 #Name:Vusi Skosana
 #Student No: SKSVUS001
 #Due date: 05 May 2015




def main():
 import math
 side=eval(input("Enter the length of the side (cm):\n")) #prompt the user to enter the length
 
 area=(3*(math.sqrt(3))/2)*((side)**2)                    #formula to calculate the area of a hexagon
 
 print("The area of a hexagon of side {0} is {1:<0.3f}.".format(side,area))  #displays the output 
main()
