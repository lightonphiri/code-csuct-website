import math
def main():
    area=math.sqrt(3)
    length=eval(input("Enter the length of the side (cm):\n"))
    area2=3*area
    area3=area2/2
    area4=area3*(length*length)
    area5="{:1.3f}".format(area4)
    area6=str(area5)+"."
    print("The area of a hexagon of side",length,"is",area6)
main()
