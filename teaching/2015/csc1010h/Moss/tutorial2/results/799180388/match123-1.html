def main():
    a = input("Enter the card holder's name:\n") # prompts the user to enter the card holder's name
    b = input("Enter the 16 digit card number (no spaces please):\n") # prompts the user to enter their card number
    c = input("Enter the date of issue (mm:yy):\n") # prompts the user to enter the issue date
    d = input("Enter the expiry date (mm:yy):\n") # prompts the user to enter the card expiry date
    e = input("Enter the 3 digit security code:\n") # prompts the user to enter the 3 digit security code
    f = int(input("Enter the credit limit (Rand):\n")) # prompts the user to enter card credit limit
    g = int(input("Enter the balance (Rand):\n")) # prompts the user to enter balance 
    
    name = " Credit Card Account " # create variable called name and store credit card accound in it
    avl = f - g # available balance
    h_1 = b[0:4] # slices the first 4 digits of card number
    h_2 = b[4:8] # slices the 2nd 4 digits of card number
    h_3 = b[8:12] # slices the 3rd 4 digits of card number
    h_4 = b[12:16] # slices the 4rd 4 digits of card number
    i_1 = c[0:2] # slices the 1st 2 digits of date of issue
    i_2 = c[3:5] # slices the last 2 digits of date of issue with no colon
    j_1 = d[0:2] # slices the 1st 2 digits of date of expiry
    j_2 = d[3:5] # slices the last 2 digits of date of expiry with no colon
    
    sp = "" # variable which stores spaces
    r = ".00 Rand" 
    x = str(f) # credit limit casted into a string
    y = str(avl) # available balance casted into a string
    z = str(g) # balance casted into a string
    
    print("+""{0:-^48}".format(name)+"+")
    print("| Card holder:   ""{0:<31} |".format(a))
    print("| Card number:""{0:^26}".format(h_1+" " +h_2+" "+h_3+" "+ h_4),"{0:>8}".format(sp)+"|")
    print("| Date of issue:""{0:^8}".format(i_1+"/"+i_2)," ""Expiry date:", j_1+"/"+j_2,"{0:>4}".format(sp)+"|")
    print("| Security code:""{0:^6}".format(e),"{0:>26}".format(sp)+"|")
    print("| Credit limit:  ""{0:<31} |".format(x+str(r)))
    print("| Available:     ""{0:<31} |".format(y+str(r)))
    print("| Balance:       ""{0:<31} |".format(z+str(r)))
    print("+""{0:-^48}".format(sp)+"+")
      
main()
