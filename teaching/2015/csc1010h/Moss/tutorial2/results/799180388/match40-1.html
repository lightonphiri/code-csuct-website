def credit_card():
    # Inputs    
    name = input("Enter the card holder's name:\n")
    card_num = input("Enter the 16 digit card number (no spaces please):\n")
    date_of_issue = input("Enter the date of issue (mm:yy):\n")
    exp_date = input("Enter the expiry date (mm:yy):\n")
    cvv = input("Enter the 3 digit security code:\n")
    card_limit = float(input("Enter the credit limit (Rand):\n"))
    card_limit = round(card_limit,2)
    balance = float(input("Enter the balance (Rand):\n"))
    balance = round(balance,2)
    
    # Formatting galore
    head = " Credit Card Account "
    print("+{0:-^48}+".format(head))
    print("| {0:<15}{1:<32}|".format("Card holder:",name))
    slice_card_num = card_num[0:4]+" "+card_num[4:8]+" "+card_num[8:12]+" "+card_num[12:16]   # Concatenation not , , is used why doesn't the other work?
    print("| {0:<15}{1:<32}|".format("Card number:",slice_card_num))
    date_of_issue2 = date_of_issue.replace(":","/") 
    exp_date2 = exp_date.replace(":","/")
    print("| {0:<15}{1:<9}{2:<13}{3:10}|".format("Date of issue:",date_of_issue2,"Expiry date:",exp_date2))
    print("| {0:<15}{1:<32}|".format("Security code:",cvv))
    length = len("{:.2f}".format(card_limit))
    print("| {0:<15}{1:<.2f}".format("Credit limit:",card_limit), "Rand", ' '*(25-length),"|") # spacing before
    available = card_limit - balance
    length = len("{:.2f}".format(available))
    print("| {0:<15}{1:<.2f}".format("Available:",available), "Rand", ' '* (25-length),"|") 
    length = len("{:.2f}".format(balance))
    print("| {0:<15}{1:<.2f}".format("Balance:",balance), "Rand", ' '*(25-length),"|") # of rands ie spacing after the length of the balance
    print("+{0:-<48}+".format(""))
credit_card()

import math
def area():  
    side = eval(input("Enter the length of the side (cm):\n"))
    area_form = ((3*math.sqrt(3))/2)*(side**2)
    print("The area of a hexagon of side",side,"is {0:<.3f}.".format(area_form))
