def creditcard():
    crd_holder_name=input("Enter the card holder's name:\n")
    card_number=input("Enter the 16 digit card number (no spaces please):\n")
    issue_date=input("Enter the date of issue (mm:yy):\n")
    issue2=issue_date.replace(':','/')
    expiry_date=input("Enter the expiry date (mm:yy):\n")
    expiry2=expiry_date.replace(':',"/")
    security_code=input("Enter the 3 digit security code:\n")
    limit=float(input("Enter the credit limit (Rand):\n"))


    limit2="{0:1.2f}".format(limit)
    STR_limit=str(limit2)+" Rand"
    balance=float(input("Enter the balance (Rand):\n"))
    balance2="{0:1.2f}".format(balance)
    STR_balance=str(balance2)+" Rand"
    avail=limit-float(balance)
    avail2='{0:1.2f}'.format(avail)
    STR_avail=str(avail2)+" Rand"
    a=card_number[0:4]
    b=card_number[4:8]
    c=card_number[8:12]
    d=card_number[12:]
    
    print("+------------- Credit Card Account --------------+")
    print("| {0:15}{1:31} |".format('Card holder:',crd_holder_name))
    print("| {0:15}{1:5}{2:5}{3:5}{4:16} |".format('Card number:',a,b,c,d))
    print("| {0:15}{1:5}    Expiry date: {2:9} |".format('Date of issue:',issue2,expiry2))
    print("| {0:15}{1:31} |".format('Security code:',security_code))
    print("| {0:15}{1:31} |".format('Credit limit:',STR_limit))
    print("| {0:15}{1:31} |".format('Available:',STR_avail))
    print("| {0:15}{1:31} |".format('Balance:',STR_balance))
