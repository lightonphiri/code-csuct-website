name=input("Enter the card holder's name:\n")
card_number=input("Enter the 16 digit card number (no spaces please):\n")
date_of_issue=input("Enter the date of issue (mm:yy):\n")


expiry_date=input("Enter the expiry date (mm:yy):\n")
security_code=input("Enter the 3 digit security code:\n")
credit_limit=float(input("Enter the credit limit (Rand):\n"))
balance=float(input("Enter the balance (Rand):\n"))
available=credit_limit-balance
available="{0:.2f} Rand".format(available)
balance="{0:.2f} Rand".format(balance)
credit_limit="{0:.2f} Rand".format(credit_limit)
print("+------------- Credit Card Account --------------+")
print("| Card holder:"," ","{0:<31} ".format(name)+"|")
print("| Card number:","","",card_number[0:4],card_number[4:8],card_number[8:12],card_number[12:16]+"{0:13}".format('')+"|")
print("| Date of issue:",date_of_issue[0:2]+"/"+date_of_issue[3:]+"    "+"Expiry date: "+expiry_date[0:2]+"/"+expiry_date[3:]+"     |")
print("{0:<17}{1:<32}|".format('| Security code:',security_code))
print("{0:<17}{1:<32}|".format("| Credit limit:",credit_limit))
print("{0:<17}{1:<32}|".format("| Available:",available))
print("{0:<17}{1:<32}|".format("| Balance:",balance))
print('+------------------------------------------------+')
