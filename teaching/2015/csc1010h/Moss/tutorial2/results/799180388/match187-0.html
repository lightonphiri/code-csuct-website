﻿import math
# Area of Triangle
# Stephan Jamieson
# 19/12/2014

# s=(a+b+c)/2
# A=sqrt(s(s-a)(s-b)(s-c))


def main():
    a = eval(input('Enter the length of the first side (cm):\n'))
    b = eval(input('Enter the length of the second side (cm):\n'))
    c = eval(input('Enter the length of the third side (cm):\n'))
    
    if a+b<c or b+c<a or a+c<b:
        print('Sorry, those numbers don\'t describe a triangle.')
    else:
        s=(a+b+c)/2
        A=math.sqrt(s*(s-a)*(s-b)*(s-c))
        
        print('The area of the triangle is {0:.3f}'.format(A), 'cm'+chr(178)+'.')
    
    
main()
