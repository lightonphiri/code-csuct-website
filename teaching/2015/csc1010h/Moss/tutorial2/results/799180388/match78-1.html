def main():
    a=eval(input("Enter the length of the side (cm) :\n"))
    Area= ((3*math.sqrt(3))/2)*a**2

def main():
    a = input("Enter the card holder's name:\n")
    b = str(input("Enter the 16 digit card number (no spaces please):\n"))
    c = input("Enter the date of issue (mm:yy):\n")
    d = input("Enter the expiry date (mm:yy):\n")
    e = input("Enter the 3 digit security code:\n")
    f = float(input("Enter the credit limit (Rand):\n"))
    g = float(input("Enter the balance (Rand):\n"))
    k = str("{:.2f}".format(f))
    h = f-g
    m = str("{:.2f}".format(h))
    n = str("{:.2f}".format(g))
    print("+------------- Credit Card Account --------------+")
    print("{:16}{:<33}|".format("| Card holder:"," "+a))
    print("{:16}{:<4} {:<4} {:<4} {:<17}|".format("| Card number:"," "+b[0:4],b[4:8],b[8:12],b[12:]))
    print("{:16}{:<10}{:<23}|".format("| Date of issue:"," "+c[0:2]+"/"+c[3:],"Expiry date: "+d[:2]+"/"+d[3:]))
    print("{:16}{:<33}|".format("| Security code:"," "+e))
    print("{:16}{:<33}|".format("| Credit limit:"," "+k+" Rand"))
    print("{:16}{:<33}|".format("| Available:"," "+m+" Rand"))
    print("{:16}{:<33}|".format("| Balance:"," "+n+" Rand"))
    print("+------------------------------------------------+")         
main()
