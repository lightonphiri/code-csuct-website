import math

print("Hello, I am Eugene Junior.")

def main():                 
    '''function to ask for input of the user and work out star war name'''
    name = input("What is your first name?\n")
    surname = input("What is your last name?\n")
    year = int(input("Hi {}.{}.!\nWhat year is this?\n".format(name[0],surname[0])))
    age = eval(input("What age are you?\n"))
    year_of_birth = year - age                   # calculating year of birth
    str_year_of_birth = str(year_of_birth)
    height = eval(input("Okay, so you were born in {}/{}.\nHow tall are you (metres)?\n".format(year_of_birth -1,str_year_of_birth[2:4])))
    height_in_feet = math.floor(height*3.28084)
    height_in_inch = math.ceil(((height*3.28084)%height_in_feet)*12)
    mothers_name = input("That's {}'{}\"!\nI'm going to tell you your Star Wars name.\nWhat's your mother's first name?\n".format(height_in_feet,height_in_inch))
    city = input("What's the name of the city nearest to your place of birth?\n") 
    star_war_name = surname[:3] + name[:2] + " " + mothers_name[-(len(mothers_name)-((len(mothers_name)-2))):] + city[-(len(city)-((len(city)-3))):]
    print("Your Star Wars name is ",star_war_name.title(),'.',sep = "")
    
main()

import math

def main():
    '''To get length of side from the user and calculate area of the hexagon'''
    length_of_side = eval(input("Enter the length of the side (cm):\n"))#get length of side from user
    area = ((3*math.sqrt(3))/2)*(length_of_side**2)# calculate the area of the hexagon
    print("The area of a hexagon of side {} is {:.3f}.".format(length_of_side,area))  # To print out the area of the hexagon
