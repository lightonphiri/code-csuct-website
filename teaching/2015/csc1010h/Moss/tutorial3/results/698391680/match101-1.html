def main():
    z = eval(input("Enter the minimum number of inches (not less than 0):\n"))
    y = eval(input("Enter the maximum number of inches (not greater than 11):\n"))
    print("Inches:",end="")
    for o in range(z,y+1):
        print("{0:>5}".format(o),end="")
    print()
    print("Metres:",end="")
    for s in range(z,y+1):
        m = s*0.0254
        print("{0:>5.2f}".format(m),end="")
