        _class_ = "C"  
    
    if _trailer_[0] == "y" :
        _trailer_mass_=eval(input("What is gross vehicle mass of the trailer (in kg)?\n"))
        if _trailer_mass_ > 750 :
            print("This vehicle is class E{}.".format(_class_))
        elif _trailer_mass_ <= 750 :
            print("This vehicle is class {}.".format(_class_))
    
    if _trailer_[0] == "n" :
        _art_ = input("Is the vehicle articulated?\n")         
        if _art_ == "n" :
            print("This vehicle is class {}.".format(_class_))
        else :
            print("This vehicle is class E{}.".format(_class_))
