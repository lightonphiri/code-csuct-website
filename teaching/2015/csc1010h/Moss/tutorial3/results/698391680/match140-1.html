def flowerpattern():
    import turtle
    petals=turtle.numinput("petals","Enter the number of petals (must be an even number): ")
    stemlength=turtle.numinput("stemlength","Enter the length of stem: ")
    petaltype=turtle.textinput("petal type","Enter the petal type (triangle, square, or pentagon): ")


    petallength=turtle.numinput("petallength","Enter the length of side: ")
    stemcolour=turtle.textinput("stemcolour","Enter the stem colour: ")
    firstpetal=turtle.textinput("firstpetal","Enter the first petal colour: ")
    secondpetal=turtle.textinput("secondpetal","Enter the second petal colour: ")
    petals1=int(petals)
    
    for i in range(petals1):
        turtle.pencolor(stemcolour)
        turtle.forward(stemlength)
        if (i%2)==0:
            turtle.pencolor(firstpetal)
        else:
            turtle.pencolor(secondpetal)
            
        if petaltype=="triangle":
            turtle.right(30)
            turtle.forward(petallength)
            turtle.left(120)
            turtle.forward(petallength)
            turtle.left(120)
            turtle.forward(petallength)
            turtle.right(30)
            turtle.pencolor(stemcolour)
            turtle.forward(stemlength)
            
            
        elif petaltype=="pentagon":
            turtle.right(54)
            turtle.forward(petallength)
            turtle.left(72)
            turtle.forward(petallength)
            turtle.left(72)
            turtle.forward(petallength)
            turtle.left(72)
            turtle.forward(petallength)
            turtle.left(72)
            turtle.forward(petallength)
            turtle.right(54)
            turtle.pencolor(stemcolour)
            turtle.forward(stemlength)
            
            
        elif petaltype=="square":
            turtle.right(45)
            turtle.forward(petallength)
            turtle.left(90)
            turtle.forward(petallength)
            turtle.left(90)
            turtle.forward(petallength)
            turtle.left(90)
            turtle.forward(petallength)
            turtle.right(45)
            turtle.pencolor(stemcolour)
            turtle.forward(stemlength)
            


        turtle.left(180-(360/petals))
        
        
    turtle.exitonclick()
        
flowerpattern()
