# program to print a conversion table of feet to metres
def main():
    minimum= int(input('Enter the minimum number of feet (not less than 0):\n'))
    maximum= int(input('Enter the maximum number of feet (not more than 99):\n'))
    
    if minimum>=0 and maximum<=99:
        for feet in range (minimum,maximum+1):
            metres= feet*0.3048
            print("  {0:>2}' |   {1:1.2f}m".format(feet,metres), end="\n")

# program to draw flower patterns

import turtle
def main():
    petals=int(input('Enter the number of petals (must be an even number): '))
    stem_length=int(input('Enter the length of stem: '))


    petal_type=input('Enter the petal type (triangle, square, or pentagon): ')
    side_length=int(input('Enter the length of side: '))
    stem_colour=input('Enter the stem colour: ')
    petal_1_colour=input('Enter the first petal colour: ')
    petal_2_colour=input('Enter the second petal colour: ')
    if petal_type=='triangle':
        ext_angle= 120
        int_angle= 60
        sides= 3
        rotate_angle= 180
    elif petal_type=='square':
        ext_angle= 90
        int_angle= 90
        sides= 4
        rotate_angle= 180
    else:
        ext_angle= 72
        int_angle= 108
        sides= 5
        rotate_angle= 180
        
    for i in range(petals):                     #draw stem and shape (petals) many times
        turtle.right(360//petals)
        turtle.pencolor(stem_colour),turtle.forward(stem_length)
        turtle.left(int_angle//2)
        if i%2 == 0:                            #draw even i's one colour
            turtle.pencolor(petal_1_colour)
        else:                                   #draw odd i's another colour
            turtle.pencolor(petal_2_colour)            
        for j in range(sides):                  #draw shape(triangle/square/pentagon)
                turtle.forward(side_length)
                turtle.right(ext_angle)
        turtle.pencolor(stem_colour),turtle.left(int_angle//2+ext_angle)
        turtle.forward(stem_length)
        turtle.left(rotate_angle)               #rotate to repeat stem and shape(petal)
        
    turtle.exitonclick()

    if minimum>=0 and maximum<=11:
        print('Inches: ',end='')        #print inches row from min no. of inches to max no. of inches


        for inches_row in range (minimum,maximum+1):    
            print("{:>4}".format(inches_row), end=" ")

        for inches in range (minimum,maximum+1):
            metres= inches/39.370
            print("{:>4.2f}".format(metres), end=" ")

def main():
    print('{0:*^30}'.format(' Vehicle classifier '))
    GVM= int(input('What is the gross vehicle mass (in kg)?\n'))
    trailer= input('Does the vehicle have a trailer?\n')
    if trailer=='y':
        gvm_trailer= int(input('What is gross vehicle mass of the trailer (in kg)?\n'))
    
        if gvm_trailer<=750:
            if GVM<=3500:
                print('This vehicle is class B.')
            elif 3500<GVM<=16000:
                print('This vehicle is class C1.')
            
            else:                       #16000<GVM
                print('This vehicle is class C.')
            
        elif gvm_trailer>750:
            if GVM<=3500:
                print('This vehicle is class EB.')
            elif 3500<GVM<=16000:
                print('This vehicle is class EC1.')

                print('This vehicle is class EC.')

        articulate=input('Is the vehicle articulated?\n')

            if GVM<=3500:
                print('This vehicle is class EB.')

                print('This vehicle is class EC1.')
            else:                           #16000<GVM:
                print('This vehicle is class EC.')
