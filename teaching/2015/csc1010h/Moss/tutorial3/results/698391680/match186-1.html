import turtle

def main(): 
    petal = eval(input("Enter the number of petals (must be an even number): "))
    stem = eval(input("Enter the length of stem: "))
    petal_type = input("Enter the petal type (triangle, square, or pentagon):")
    side = eval(input("Enter the length of side: "))
    colourS = input("Enter the stem colour: ")
    colour1 = input("Enter the first petal colour: ")
    colour2 = input("Enter the second petal colour: ")
    if petal_type == "triangle":
        for i in range (petal):
            turtle.pendown()
            turtle.color(colourS)
            turtle.forward(stem)
            turtle.right(30)
            if (i+1)%2 != 0 :
                turtle.color(colour1)
            else:
                turtle.color(colour2)
            for n in range (3):
                turtle.forward(side)   
                turtle.left(120)
            turtle.penup()
            turtle.home()
            turtle.right((360/petal)*(i+1))
        turtle.exitonclick()
            
    elif petal_type == "square":
        for i in range (petal):
            turtle.pendown()
            turtle.color(colourS)
            turtle.forward(stem)
            turtle.right(45)
            if (i+1)%2 != 0 :
                turtle.color(colour1)
            else:
                turtle.color(colour2)            
            for n in range (4):
                turtle.forward(side)    
                turtle.left(90)
            turtle.penup()
            turtle.home()
            turtle.right((360/petal)*(i+1))
        turtle.exitonclick()
    else:
        for i in range (petal):
            turtle.pendown()
            turtle.color(colourS)
            turtle.forward(stem)
            turtle.right(36)
            if (i+1)%2 != 0 :
                turtle.color(colour1)
            else:
                turtle.color(colour2)            
            for n in range (5):
                turtle.forward(side)
                turtle.left(72)
            turtle.penup()
            turtle.home()
            turtle.right((360/petal)*(i+1))  
        turtle.exitonclick()

main()
