import turtle
def main():

    petal_number=int(input('Enter the number of petals (must be an even number): '))
    stem_length=int(input('Enter the length of stem: '))
    petal_type=input('Enter the petal type (triangle, square, or pentagon): ')
    side_length=int(input('Enter the length of side: '))
    stem_colour=input('Enter the stem colour: ')
    first_petal=input('Enter the first petal colour: ')
    second_petal=input('Enter the second petal colour: ')
    angle=360/petal_number
    
    
    if petal_type=='triangle':
        angle=180-(360/petal_number)
        for i in range(petal_number):
                turtle.right(180)
                turtle.forward(stem_length)
                turtle.pencolor(first_petal)
                
                if i%2==0:
                    turtle.pencolor(second_petal)            
                turtle.right(330)
                
                for i in range(3):
                    turtle.forward(side_length)
                    turtle.right(120)
                turtle.right(30)
                turtle.pencolor(stem_colour)
                turtle.back(stem_length)
                turtle.right(angle)        
            
    elif petal_type=='square':
        for i in range(petal_number):
            turtle.forward(stem_length)
            turtle.left(45)
            turtle.pencolor(first_petal)
            
            if i%2==0:
                turtle.pencolor(second_petal)    
            turtle.forward(stem_length)
            
            for i in range(3):
                turtle.right(90)
                turtle.forward(stem_length)
            turtle.right(-45)
            turtle.pencolor(stem_colour)
            turtle.forward(stem_length)
            turtle.right(180)
            turtle.right(angle)
    
    
    else:
        for i in range(petal_number):
            turtle.forward(stem_length)
            turtle.left(54)
            turtle.pencolor(first_petal)
            
            if i%2==0:
                turtle.pencolor(second_petal)    
            turtle.forward(stem_length)
            
            for i in range(4):
                turtle.right(72)
                turtle.forward(stem_length)
            turtle.right(-54)
            turtle.pencolor(stem_colour)
            turtle.forward(stem_length)
            turtle.right(180)
            turtle.right(angle)
    turtle.exitonclick()
