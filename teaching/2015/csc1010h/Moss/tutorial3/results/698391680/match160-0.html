def main():
    print("***** Vehicle classifier *****")
    car=eval(input("What is the gross vehicle mass (in kg)?\n"))
    trailer=input("Does the vehicle have a trailer?\n")
    
    if trailer=='y':
        mass=eval(input("What is gross vehicle mass of the trailer (in kg)?\n"))
        if mass>750 and car<=3500:
            print("This vehicle is class EB.")
        elif mass>750 and 3500<car<=16000:
            print("This vehicle is class EC1.")
        elif mass>750 and car>16000:
            print("This vehicle is class EC.")
        elif mass<=750 and car<=3500:
            print("This vehicle is class B.")
        elif mass<=750 and 3500<car<=16000:
            print("This vehicle is class C1.")
        elif mass<=750 and 16000<car:
            print("This vehicle is class C.")
            
    else:
        art=input("Is the vehicle articulated?\n")
        if art=='y':
            if car<=3500:
                print("This vehicle is class EB.")
            elif 3500<car<=16000:
                print("This vehicle is class EC1.")
            elif 16000<car:
                print("This vehicle is class EC.")
        else:
            if car<=3500:
                print("This vehicle is class B.")
            elif 3500<car<=16000:
                print("This vehicle is class C1.")
            elif 16000<car:
                print("This vehicle is class C.")            
        
main()
