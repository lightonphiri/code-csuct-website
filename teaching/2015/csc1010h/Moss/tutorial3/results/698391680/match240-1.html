from turtle import forward
from turtle import right
from turtle import left
from turtle import color

n = eval(input("Enter the number petals (must be an even number): ")) # number of leaves
l = eval(input("Enter the length of stem: ")) # length stem
t = input("Enter the petal type (triangle, square, or pentagon): ").lower() # type
s = eval(input("Enter the length of side: ")) # side length
stc = input("Enter the stem colour: ").lower() # stem colour
fc = input("Enter the first petal colour: ").lower() # first petal colour
sc = input("Enter the second petal colour: ").lower() # second petal colour

a = 360//n # stem turn angle

if t == "triangle":
    r = 3
    k = 60
    
elif t == "pentagon":
    r = 5
    k = 108
elif t == "square":
    r = 4
    k = 90

left(90)
for i in range(1,n+1):
    if (i%2==0):
        z = sc
    else:
        z = fc
    forward(l)
    right(k//2)

    for i in range(r):
        color(z)
        forward(s)
        right(k+180)
    color(stc)    
    right(180-(k//2)) 
    forward(l)
    right(a+180)  
turtle.exitonclick()

def main():
    n = input("Enter a time ([h]h:mm [am|pm]):\n").lower()
    
    if n.index(":") == 1:
        n = n.replace(n[0:1],'0'+n[0:1])    
    if len(n) == 8: 
        if n[:2] == '12' and "am" in n:
            n = n.replace(n[:2],'00')
            n = n.replace("am","")
        
        if n[:2] == '12' and "pm" in n:
            n = n.replace("pm","")
        
        elif "am" in n:
            n = n.replace("am","")
            
        elif "pm" in n:
            n = n.replace(n[:2],str(int(n[:2])+12))
            n = n.replace("pm","")
        print(n)
        
    elif len(n) == 5:
        if int(n[0:2]) >= 1  and int(n[0:2]) < 12:
            print(n[1:5]+' am')
        elif int(n[0:2]) > 12 and int(n[0:2]) < 24:
            p = str(int(n[0:2])-12)
            print('0'+p+n[2:5]+' pm')
        elif int(n[0:2]) == 0:
            p = str(int(n[0:2])+12)
            print(p+n[2:5]+' am')    
        else:
            print(n+' pm')
