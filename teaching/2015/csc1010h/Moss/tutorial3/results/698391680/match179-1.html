    max_feet = eval(input("Enter the maximum number of feet (not more than 99):\n"))

        conversion = float(i/3.28)

            print("  "+str(i) + "' |    " + str(round(conversion,2))+"m")

import turtle   

def main():
    

    petal = int(input("Enter the number of petals (must be an even number):"))
    
    stem_length = int(input("Enter the length of stem:"))
    shape = input("Enter the petal type (triangle, square, or pentagon):")
    side_length = int(input("Enter the length of side:"))
    stem_colour = input("Enter the stem colour:")
    petal_color = input("Enter the first petal colour:")
    
    turtle.left(90)     
    for i in range(petal):
        
        turtle.pencolor(stem_colour)	
        turtle.forward(stem_length) 
            
        if shape == "triangle": 


            for i in range(3):
                turtle.pencolor(petal_color)
                turtle.left(30)
                turtle.forward(side_length)
                turtle.right(120)
                turtle.right(30)
                
        elif shape == "square":


            for i in range(4):
                turtle.pencolor(petal_color)
                turtle.left(45)
                turtle.forward(side_length)
                turtle.right(90)
                turtle.right(45)
                
        elif shape == "pentagon":
            for i in range(5):
                turtle.pencolor(petal_color)
                turtle.left(55)
                turtle.forward(side_length)
                turtle.right(72)
                turtle.right(55)
                
            
        
        turtle.pencolor(stem_colour)	        
        turtle.back(stem_length)
    
        turtle.right(360/petal)

    print("Inches:", end = "  " )
    for i in range(min_inches , max_inches+1):
        conversion = float(i/39.37)
        print(" ",i, end = "  ", )
        
    print("\nMetres:", end = " ")
    for i in range(min_inches , max_inches+1):
        conversion = float(i/39.37)
