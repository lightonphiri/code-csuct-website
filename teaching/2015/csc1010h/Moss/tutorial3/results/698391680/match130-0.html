def main():
    min_num = int(input('Enter the minimum number of inches (not less than 0): '))
    max_num = int(input('Enter the maximum number of inches (not greater than 11): '))
    for i in range(min_num,max_num+1):
        metre = i*0.0254
        metre = round(metre,2)
        print('Inches:',i,end=' ')
        print('Metres:',metre,end=' ')
