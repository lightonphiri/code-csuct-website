def main():
    print("***** Vehicle classifier *****")
    GVM_V = int(input("What is the gross vehicle mass (in kg)?\n"))
    a = input("Does the vehicle have a trailer?\n")
    if a == "y":
        GVM_T = int(input("What is gross vehicle mass of the trailer (in kg)?\n")) 
        if GVM_V <= 3500 and GVM_T <= 750:
            print("This vehicle is class B.")
        elif 3500 < GVM_V <= 16000 and GVM_T <= 750:
            print("This vehicle is class C1.")
        elif GVM_V > 16000 and GVM_T <= 750:
            print("This vehicle is class C.")
        elif GVM_V <= 3500 and GVM_T > 750:
            print("This vehicle is class EB.")
        elif 3500 < GVM_V <= 16000 and GVM_T > 750:
            print("This vehicle is class EC1.")
        elif GVM_V > 16000 and GVM_T > 750:
            print("This vehicle is class EC.")        
        
            
    elif a == "n":
        c = input("Is the vehicle articulated?\n")
        if c == "n":
            if GVM_V <= 3500:
                print("This vehicle is class B.")
            elif 3500 < GVM_V <= 16000:
                print("This vehicle is class C1.")
            elif GVM_V > 16000:
                print("This vehicle is class C.")
        elif c == "y":
            if GVM_V <= 3500:
                print("This vehicle is class EB.")
            elif 3500 < GVM_V <= 16000:
                print("This vehicle is class EC1.")            
            elif GVM_V > 16000:
                print("This vehicle is class EC.")
    
main()
