#prompt user
print ("***** Vehicle classifier *****")
gross_mass= int(input('What is the gross vehicle mass (in kg)?\n'))
trailer= input('Does the vehicle have a trailer?\n')
if trailer=='y':
    gross_mass_t= int(input('What is gross vehicle mass of the trailer (in kg)?\n'))
    if gross_mass <= 3500:
        if  gross_mass_t > 750:
            print ('This vehicle is class EB.')
        else:
            print ('This vehicle is class B.')
    elif 3500 <gross_mass<= 16000:
        if  gross_mass_t > 750:
            print ('This vehicle is class EC1.')
        else:
            print ('This vehicle is class C1.')
    elif gross_mass > 16000:
        if  gross_mass_t > 750:
            print ('This vehicle is class EC.')
        else:
            print ('This vehicle is class C.')


elif trailer=='n':
    articulated= input('Is the vehicle articulated?\n')
    if articulated== 'y' and gross_mass <= 3500:
        print ('This vehicle is class EB.')
    elif articulated== 'y' and 3500< gross_mass <= 16000:
        print ('This vehicle is class EC1.')
    elif articulated== 'y' and gross_mass > 16000:
        print ('This vehicle is class EC.')   
    elif articulated== 'n' and gross_mass <= 3500:
        print ('This vehicle is class B.')
    elif articulated== 'n' and 3500< gross_mass <= 16000:
        print ('This vehicle is class C1.')
    elif articulated== 'n' and gross_mass > 16000:
        print ('This vehicle is class C.')
