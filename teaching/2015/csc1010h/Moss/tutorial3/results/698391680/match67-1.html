        turtle.left(30)

    classVehicle='C1.'
elif gvm>16000:
    classVehicle='C.'

    gvmTrailer = int(input("What is gross vehicle mass of the trailer (in kg)?\n"))

        classTogether = 'E'+classVehicle

        classTogether=classVehicle
