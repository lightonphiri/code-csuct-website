    r = 5
    k = 108

    k = 90

def main():
    time=input('Enter a time in 12 hour clock ([h]h:mm [am|pm]):\n')
    length=len(time)
    
    if length==8:
        if time[-2:]=="pm" and int(time[0:2])==12:
            print("12:{}".format(time[3:5]))
        elif time[-2:]=="pm" and int(time[0:2])!=12:
            print("{}:{}".format((int(time[0:2])+12),time[3:5]))
            
            
        elif time[-2:]=="am" and int(time[0:2])==12:
            print("00:{}".format(time[3:5]))
        elif int(time[0:2])!=12 and time[-2:]=="am":
            print("{}:{}".format( time[0:2],time[3:5]))
    elif length==7:
        if time[-2:]=="am":
            print("0{}{}".format( time[0:2],time[2:4]))
        elif time[-2:]=="pm":
            print("{}:{}".format((int(time[0])+12),time[2:4]))
main()

def main():
    time=input('Enter a time in 24 hour clock (hh:mm):\n')
    a= int(time[0:2])
    if a==00:
        print('12'+':'+ time[-2:],'am')
    elif 0<a<12:
        print('{}:{} am'.format(a,time[-2:]))
    elif a==12:
        print('{}:{} pm'.format(a,time[-2:]))
    elif 12<a<24:
        print('{}:{} pm'.format((a-12),time[-2:]))
main()
