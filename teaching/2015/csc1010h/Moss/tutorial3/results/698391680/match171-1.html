import turtle 
def main():
    turtle.title("Draw Flower")

    len_stem=turtle.numinput('Stem Length Window','Enter the length of stem:')
    petal_type=turtle.textinput('Petal Type Window','Enter the petal type (triangle, square, or pentagon):')
    len_side=turtle.numinput('Side Length Window','Enter the length of side:')
    stem_colour=turtle.textinput('Stem Colour Window','Enter the stem colour:')
    first_petal=turtle.textinput('First Petal Window', 'Enter the first petal colour:')
    second_petal=turtle.textinput('Second Petal Window','Enter the second petal colour:')
    
    
    turtle.pencolor(stem_colour)
    for i in range(int(num_petals)):
        turtle.forward(len_stem)
        if petal_type=="triangle":
                turtle.left(30)
                turtle.pencolor(first_petal)

def main(): 
    lowest=input("Enter the minimum number of inches (not less than 0):\n")
    highest=input("Enter the maximum number of inches (not greater than 11):\n")
        
    print("Inches:",end="")
    for j in range(int(lowest),int(highest)+1):
        print("{0:5}".format(j),end="")

    print() #new row
    
    print("Metres: ",end="")
    for k in range(int(lowest),int(highest)+1):
        metres=round((k/39.320),2)
        print("{0:.2f}".format(metres),end=" ")
main()
