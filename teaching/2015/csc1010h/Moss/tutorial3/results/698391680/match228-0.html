    if time[-1]=="m":
        if hour_time_hour==12:
            if time[-2]=="a":
                print("00"+time_minutes)        
            else:
                print(time[:5])
        elif time[-2]=="p":
            print(str(hour_time_hour+12)+str(time_minutes))
        elif time[-2]=="a":
            if 0<hour_time_hour<10:
                print("0"+time[:5])
            else:
                print(time[:5])
    else:
        if 0<hour_time_hour<12:
            print(str(hour_time_hour)+time[time.find(":"):],"am")
        elif hour_time_hour==12:
            print(time_hour+time[-3:],"pm")
        elif hour_time_hour>12:
            print(str(hour_time_hour-12)+time[2:1],"pm")
        elif hour_time_hour==0:
            print("12"+time[2:],"am")
