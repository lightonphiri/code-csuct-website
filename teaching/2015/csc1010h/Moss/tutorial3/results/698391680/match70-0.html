def flower():
    
    import turtle
    stem=turtle.numinput('Petals','Enter the number of petals(must be an even number):')  #prompt useer,number of petals
    lenght=turtle.numinput('Lenght of stem','Enter the lenght of stem:') #prompt user,lenght of stem
    petal_type=turtle.textinput('Petal type','Enter the petal type (triangle,square, or pentagon:') #prompt user,type of petal
    side=turtle.numinput('Lenght of side','Enter the lenght of side:') #prompt user,lenght of side
    stem_colour=turtle.textinput('First petal colour','Enter the stem colour:') #prompt user, the stem colour
    colour_1=turtle.textinput('First petal colour','Enter the first petal colour:') #prompt user, the colour of the first petal
    colour_2=turtle.textinput('Second petal colour','Enter the second petal colour:') # prompt user, tthe colour of the second petal
    
    for i in range(int(stem)):                #number of stems
        turtle.pencolor(stem_colour)          #stem colour
        turtle.right(360/stem)                #rotating to draw the stem
        turtle.forward(lenght)                #drawing the stem
        if i%2==0:                     
            turtle.pencolor(colour_1)         #change colour
        else:
            turtle.pencolor(colour_2)         #change colour
            
    #drawing a triangle petal
        if petal_type=='triangle':            
            turtle.left(30)                    
            for i in range(2):                 
                turtle.forward(side)
                turtle.right(360/3)
            turtle.forward(side)
            turtle.left(30)
            turtle.pencolor(stem_colour)
            turtle.forward(lenght)
            turtle.left(180)
            
       #drawing a pentagon petal     
        elif petal_type=='pentagon':
            turtle.left(54)
            for i in range(4):
                turtle.forward(side)
                turtle.right(360/5)
            turtle.forward(side)
            turtle.left(54)
            turtle.pencolor(stem_colour)
            turtle.forward(lenght)
            turtle.left(180)
            
        # drawing square petals
        elif petal_type=='square':
            turtle.left(45)
            for i in range(3):
                turtle.forward(side)
                turtle.right(360/4)
            turtle.forward(side)
            turtle.left(45)
            turtle.pencolor(stem_colour)
            turtle.forward(lenght)
            turtle.left(180)
            
    turtle.exitonclick()

   print("Inches:",end="")                                                               #print heading inches
   for i in range(min_inches,max_inches+1):                                              # loop
      print('{0:>5}'.format(i),end='')                                                   # print results 

   print("")                                                                          # skip a line
   print("Metres:",end='')                                                            # print heading metres
   for i in range(min_inches,max_inches+1):                                           # loop
