    count = minimum_feet

def main():
    print("***** Vehicle classifier *****")
    print("What is the gross vehicle mass (in kg)?")
    vehicle_mass = int(input())
    
    print("Does the vehicle have a trailer?")
    trailer = input()
    if trailer == 'y':
        print("What is gross vehicle mass of the trailer (in kg)?")
        trailer_mass = int(input())


        if trailer_mass > 750:
            if vehicle_mass <= 3500:
                print("This vehicle is class EB.")
            elif 3500 < vehicle_mass <= 16000:
                print("This vehicle is class EC1.")
            else:
                print("This vehicle is class EC.")  
        else:
            if vehicle_mass <= 3500:
                print("This vehicle is class B.")
            elif 3500 < vehicle_mass <= 16000:
                print("This vehicle is class C1.")
            else:
                print("This vehicle is class C.")            
    else:
        print("Is the vehicle articulated?")
        articulated = input()


        if articulated == 'y':
            if vehicle_mass <= 3500:
                print("This vehicle is class EB.")
            elif 3500 < vehicle_mass <= 16000:
                print("This vehicle is class EC1.")
            else:
                print("This vehicle is class EC.")
        else:
            if vehicle_mass <= 3500:
                print("This vehicle is class B.")
            elif 3500 < vehicle_mass <= 16000:
                print("This vehicle is class C1.")
            else:
                print("This vehicle is class C.")
