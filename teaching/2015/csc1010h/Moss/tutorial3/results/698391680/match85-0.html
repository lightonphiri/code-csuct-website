import turtle

petalnumber = int(input('Enter the number of petals (must be an even number) : '))
stemlength = int(input('Enter the length of the stem: '))
petaltype = input('Enter the petal type (triangle, square or pentagon) : ')
petallength = int(input('Enter the length of side: '))
stemcolor = input('Enter the stem colour: ')
firstcolor = input('Enter the first colour: ')
secondcolor = input('Enter the second colour: ')

if petaltype == 'pentagon':
        for i in range (int(petalnumber)): # number of stems


                turtle.forward(stemlength) #stem length
                turtle.left(54)# half of the int angle (60/2)
                turtle.pencolor(firstcolor) # pen color1
                if i%2 == 0:
                        turtle.pencolor(secondcolor)
                turtle.forward(petallength)
                for i in range (4): #number of sides - 1 in shape
                        turtle.right(72) # ext angle of petal
                        turtle.forward(petallength) #petal length
                turtle.right(-54) # angle to go back home
                turtle.pencolor(stemcolor) #stem color
                turtle.forward(stemlength)
                turtle.right(180)
                turtle.right(360/int(petalnumber)) #rotation for next stem

elif petaltype == 'square':
        for i in range (int(petalnumber)): # number of stems


                turtle.forward(stemlength) #stem length
                turtle.left(45)# half of the int angle (60/2)
                turtle.pencolor(firstcolor) # pen color1
                if i%2 == 0:
                        turtle.pencolor(secondcolor)
                turtle.forward(petallength)

    time = input('Enter a time ([h]h:mm [am|pm]): \n')
    
    if time.endswith('m'):
        if time[1] == ':':
            if time.endswith('am'):
                print('0'+time[0:4])
            elif time.endswith('pm'):
                print(str(int(time[0])+12)+time[1:4])                               
        if time[2] == ':':
            if time == '12:00 am':
                print('00:00')
            if time == '12:00 pm':
                print ('12:00')
            if time[0:2] == '12' and time.endswith('pm'):
                print(time[0:5])
            if time[0:2] == '12' and time.endswith('am'):
                print('00'+time[2:5])
            elif time.endswith('am') and not time[0:2] == '12':
                print(time[0:5])
            elif time.endswith('pm') and not time[0:2] == '12':
                print(str(int(time[0:2])+12)+time[2:5])

    if not time.endswith('m'):        
            if time == '00:00':
                print('12:00 am')
            if time == '12:00':
                print ('12:00 pm')
            if time[0:2] == '12':
                print(time[0:5]+' pm')
            if time[0:2] == '00':
                print('12'+time[2:5]+' am')
            elif int(time[0:2]) <= 11:
                print(time[1:5]+' am')
            elif int(time[0:2]) >= 13:
                print(str(int(time[0:2])-12)+time[2:5]+' pm')
