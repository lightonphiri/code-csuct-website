def main():
    min_num = int(input("Enter the minimum number of inches (not less than 0):\n"))
    max_num = int(input("Enter the maximum number of inches (not greater than 11):\n"))
    
    print("Inches:", end='')
    for row in range(min_num,max_num+1):
        print("{0:>5}".format(row),end="")
    print("\n""Metres:",end="")
    for i in range(min_num,max_num+1):
        metres = (i * 0.0254)
        print("{0:5.2f}".format(metres),end="")
