print("***** Vehicle classifier *****")


mass=int(input("What is the gross vehicle mass (in kg)?\n"))
trail=input("Does the vehicle have a trailer?\n")

if trail=="y":
    masst=int(input("What is gross vehicle mass of the trailer (in kg)?\n"))
    if masst>750:
        if mass<=3500:
            print("This vehicle is class EB.")
        elif 3500<mass<=16000:
            print("This vehicle is class EC1.")
        elif 16000<mass:
            print("This vehicle is class EC.")         
        
    elif masst<=750:
        if mass<=3500:
            print("This vehicle is class B.")
        elif 3500<mass<=16000:
            print("This vehicle is class C1.")
        elif 16000<mass:
            print("This vehicle is class C.")        
        
        
elif trail=="n":
    vart=input("Is the vehicle articulated?\n")
