import turtle

def main():
    turtle.title("Draw flowers")
    no_petals = turtle.numinput("Number of petal","Enter the number of petals (must be an even number): ")
    l_stem = turtle.numinput("length of stem","Enter the length of stem: ")
    petal_t = turtle.textinput("petal type","Enter the petal type (triangle, square, pentagon): ")
    l_sides = turtle.numinput("length of sides","Enter the length of side: ")
    stem_co = turtle.textinput("stem colour","Enter the stem colour: ")
    f_petal_co = turtle.textinput("first petal colour","Enter the first petal colour: ")
    s_petal_co = turtle.textinput("second petal colour","Enter the second petal colour: ")
    m = 0
    if petal_t=="triangle":


        for k in range(int(no_petals)):
            turtle.pencolor(stem_co)
            turtle.fd(l_stem)        
            if k %2 !=0:
                turtle.pencolor(f_petal_co)
            else:
                turtle.pencolor(s_petal_co)
            
            turtle.left(30)        
            turtle.fd(l_sides)
            turtle.right(120)
            turtle.fd(l_sides)
            turtle.right(120)
            turtle.fd(l_sides)
            turtle.left(30)
            turtle.pencolor(stem_co)
            turtle.fd(l_stem)
            turtle.home()
            m = m + 360/no_petals
            turtle.right(m)
    elif petal_t=="pentagon":


        for k in range(int(no_petals)):
            turtle.pencolor(stem_co)
            turtle.fd(l_stem)

            turtle.fd(l_sides)

            turtle.home()
            m = m + 360/no_petals
            turtle.right(m)
    elif petal_t=="square":


        for k in range(int(no_petals)):
            turtle.pencolor(stem_co)
            turtle.fd(l_stem)        
            if k %2 !=0:
                turtle.pencolor(f_petal_co)

def main():
    print("***** Vehicle classifier *****")
    gvm = eval(input("What is the gross vehicle mass (in kg)?\n"))
    trailer = input("Does the vehicle have a trailer?\n")
    
    if trailer[0]=="y":
        mass_trailer = eval(input("What is gross vehicle mass of the trailer (in kg)?\n"))
        if mass_trailer > 750:
            if gvm <= 3500:
                print("This vehicle is class EB.")
            elif 3500 < gvm <= 16000:
                print("This vehicle is class EC1.")
            elif 16000 < gvm:
                print("This vehicle is class EC.")
        else:
            if gvm <= 3500:
                print("This vehicle is class B.")
            elif 3500 < gvm <= 16000:
                print("This vehicle is class C1.")
            elif 16000 < gvm:
                print("This vehicle is class C.")            
    else:
        art = input("Is the vehicle articulated?\n")
        if art[0]=="y":
            if gvm <= 3500:
                print("This vehicle is class EB.")
            elif 3500 < gvm <= 16000:
                print("This vehicle is class EC1.")
            elif 16000 < gvm:
                print("This vehicle is class EC.")
        else:
            if gvm <= 3500:
                print("This vehicle is class B.")
            elif 3500 < gvm <= 16000:
                print("This vehicle is class C1.")
            elif 16000 < gvm:
                print("This vehicle is class C.")
