import turtle
def main():
    petals = turtle.numinput("Petals","Enter the number of petals (must be an even number):")
    stem = turtle.numinput("Stem","Enter the length of stem:")
    petaltype = turtle.textinput("Petal Type","Enter the petal type (triangle, square, or pentagon):")


    side = turtle.numinput("Side","Enter the length of side:")
    stemcolour = turtle.textinput("Stem Colour","Enter the stem colour:")
    firstpetal = turtle.textinput("First Petal","Enter the first petal colour:")
    secondpetal = turtle.textinput("Second Petal","Enter the second petal colour:")
    b = 0
    count = 0
    for i in range (int(petals)):
        turtle.pencolor(stemcolour)
        turtle.forward(stem)
        if petaltype == "pentagon":
            turtle.left(54)
            if b == 0 or b%2 == 0:
                for i in range(5):
                    turtle.pencolor(secondpetal)
                    turtle.forward(side)
                    turtle.right(72)
                turtle.pencolor(stemcolour)
                turtle.home()
                angle = 360/petals
                count+= angle
                turtle.right(count)
            else:
                for i in range(5):
                    turtle.pencolor(firstpetal)
                    turtle.forward(side)
                    turtle.right(72)
                turtle.pencolor(stemcolour)
                turtle.home()
                angle = 360/petals
                count+= angle
                turtle.right(count)
            b+=1
            
        elif petaltype == "triangle":
            turtle.left(30)
            if b == 0 or b%2 == 0:

def main():
    minimum = eval(input("Enter the minimum number of inches (not less than 0):\n"))
    maximum = eval(input("Enter the maximum number of inches (not greater than 11):\n"))
    print("Inches:", end=" ")
    for i in range(minimum, maximum+1):
        metres = i*0.0254
        print("{0:>4}".format(i), end=" ")
    print("\nMetres:", end=" ")
    for i in range(minimum, maximum+1):
        metres = i*0.0254    
        print("{0:>4.2f}".format(metres),end=" ")
