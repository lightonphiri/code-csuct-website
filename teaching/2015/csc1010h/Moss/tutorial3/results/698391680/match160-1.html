def main():
    print("***** Vehicle classifier *****")

    gvm = int(input("What is the gross vehicle mass (in kg)?\n"))
    a = input("Does the vehicle have a trailer?\n")
    if a == 'y':
        gvm_of_t = int(input("What is gross vehicle mass of the trailer (in kg)?\n"))
        if gvm_of_t>750 and gvm<=3500:
            print("This vehicle is class EB.")
            
        elif gvm_of_t>750 and 3500<gvm<=16000:
            print("This vehicle is class EC1.")
        elif  gvm>16000 and  gvm_of_t>750:
            print("This vehicle is class EC.")
        elif gvm_of_t<=750 and gvm<=3500:
            print("This vehicle is class B.")
        elif gvm_of_t<=750 and 3500<gvm<=16000:
            print("This vehicle is class C1.")
        elif  gvm>16000 and  gvm_of_t<=750:
            print("This vehicle is class C.")        
            
        
    elif a=="n":
        c = input("Is the vehicle articulated?\n")
        
        if gvm<=3500 and c=="y":
            print("This vehicle is class EB.")
            
        elif c=="y" and 3500<gvm<=16000:
            print("This vehicle is class EC1.")
        elif  gvm>16000 and c=="y":
            print("This vehicle is class EC.")
            
        elif c=="n" and gvm<=3500:
            print("This vehicle is class B.")
                        
        elif c=="n"and 3500<gvm<=16000:
            print("This vehicle is class C1.")
        elif  gvm>16000 and  c=="n":
            print("This vehicle is class C.")
