from turtle import *

def flowerpattern():
    number_of_petals=numinput('Enter the number of petals (must be an even number):','')
    length_of_stem=numinput('Enter the length of stem:','')
    petal_type=textinput('Enter the petal type (triangle, square, or pentagon):','')
    length_of_side=numinput('Enter the length of side:','')
    stem_colour=textinput('Enter the stem colour:','')
    first_petal_colour=textinput('Enter the first petal colour:','')
    second_petal_colour=textinput('Enter the second petal colour:','')
    angle=360/int(number_of_petals)
    
    if petal_type=='triangle':
        for i in range(1, int(number_of_petals)+1):
            i-=1
            if i%2==0:
                i+=1
                pencolor(stem_colour)
                forward(length_of_stem)
                left(30)
                pencolor(second_petal_colour)
                for n in range(3):
                    forward(length_of_side)
                    right(120)
                penup()
                home()
                pendown()
                left(i*int(angle))
            
            elif i%2!=0:
                i+=1
                pencolor(stem_colour)
                forward(length_of_stem)
                left(30)
                pencolor(first_petal_colour)  
                for n in range(3):
                    forward(length_of_side)
                    right(120)
                penup()
                home()
                pendown()
                left(i*int(angle))
    
    elif petal_type=='square':
        for i in range(1, int(number_of_petals)+1):
            i-=1
            if i%2==0:
                i+=1        
                pencolor(stem_colour)
                forward(length_of_stem)
                left(45)

                    forward(length_of_side)
                    right(90)
                penup()
                home()
                pendown()
                left(i*int(angle))
                
    elif petal_type=='pentagon':
        for i in range(1, int(number_of_petals)+1):

                i+=1

def main():
    time=input('Enter a time ([h]h:mm [am|pm]):\n')
    colon=time.find(':')
    time1=time[:colon]
    time2=time[colon+1:5]
    
    if time.endswith('am') or time.endswith('pm'):         
        if 0<int(time1)<10 and time.endswith('am'):
            print('0'+str(time1)+':'+time2)
        
        elif 10<=int(time1)<=11 and time.endswith('am'):
            print(str(time1)+':'+time2)
        
        elif 0<int(time1)<12 and time.endswith('pm'):
            time_in_pm=int(time1)+12
            print(str(time_in_pm)+':'+time2)    
                
        elif int(time1)==12 and time.endswith('pm'):
            time_pm=int(time1)
            print(str(time_pm)+':'+time2)
            
        elif time1=='12' and time.endswith('am'):
            print('00'+':'+time2)       
        
    else:        
        if 0<int(time1)<12:
            print(str(int(time1))+':'+time2+' am')
            
        elif time1=='00':
            hour_am=int(time1)+12
            print(str(hour_am)+':'+time2+' am')
                
        elif time1==12:
            print(str(time1)+':'+time2+' pm')
                
        elif 13<=int(time1)<=23:
            time1_am=int(time1)-12
            print(str(time1_am)+':'+time2+' pm')
