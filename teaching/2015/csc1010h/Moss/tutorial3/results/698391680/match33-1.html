def main():
    x = eval(input("Enter the minimum number of inches (not less than 0):\n"))
    y = eval(input("Enter the maximum number of inches (not greater than 11):\n"))
    
    print("Inches:",end='')    
    for i in range (x, y+1):
            metre = i*0.0254
            metre = round(metre,2)
            print('{0:>5}'.format(i)+'', end='')
    print('\nMetres: ',end='')
    for i in range(x,y+1):
        metre = i*0.0254
        print('{0:2.2f}'. format(metre),end=' ')

    if trailer =="n":
        articulated = str(input("Is the vehicle articulated?\n"))
        grossTrailer = 0
    else:    
        grossTrailer = eval(input("What is gross vehicle mass of the trailer (in kg)?\n"))
        articulated = "n"
        
    E = False
    if((articulated == 'y') or (grossTrailer>750)):
        E = True
              
    if(gross_vehicle<= 3500 and (E == False)):
        print("This vehicle is class B.")    
    elif(gross_vehicle>16000 and (E == False)):
        print("This vehicle is class C.")
    elif (3500<gross_vehicle<=16000 and (E == False)):    
        print("This vehicle is class C1.")
     
     
    #E series 
    if(gross_vehicle<= 3500 and (E == True)):
        print("This vehicle is class EB.")    
    elif(gross_vehicle>16000 and (E == True)):
        print("This vehicle is class EC.")
    elif (3500<gross_vehicle<=16000 and (E == True)):    
        print("This vehicle is class EC1.")     

       
    # E series for towing > 750 or articulated
