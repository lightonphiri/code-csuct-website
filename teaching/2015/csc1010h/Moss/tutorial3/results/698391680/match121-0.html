    max_num = int(input('Enter the maximum number of inches (not greater than 11):\n'))
    print('Inches:',end="")
    
    
    
    for i in range(min_num,max_num+1):
        metre = i*0.0254
        metre = round(metre,2)
        print("{0:>5}".format(i)+"",end="")
        
    print('\nMetres: ',end="")
    for i in range(min_num,max_num+1):
        metre = i*0.0254
        #metre = round(metre,2)
        print("{0:2.2f}".format(metre),end=" ")
