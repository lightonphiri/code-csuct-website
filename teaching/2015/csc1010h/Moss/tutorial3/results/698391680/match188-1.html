def main():
    #12 to 24
    time=input("Enter a time ([h]h:mm [am|pm]):\n")
    end=time[-2:]
    length=len(time)
    
    if end=="am" and length==8 and int(time[0:2])==12:
        print("{}:{}".format("00",time[3:5]))
    elif end=="pm" and length==7:
        print("{}:{}".format(int(time[0:1])+12,time[2:4]))
    elif end=="am" and length==7:
        print("0{}:{}".format(time[0:1],time[2:4]))
    elif end=="pm" and length==8 and int(time[0:2])<12:
        print("{}:{}".format(int(time[0:2])+12,time[3:5]))
    elif end=="am" and length==8:
        print("{}:{}".format(int(time[0:2]),time[3:5]))
        
    elif end=="pm" and length==8 and int(time[0:2])==12:
        print("{}:{}".format("12",time[3:5]))    
  
    # 24 to 12
    elif end!="pm" and end!="am" and length==5 and 0<int(time[0:2])<10:
        print("{}:{} am".format(time[1:2],time[-2:]))
    elif end!="pm" and end!="am" and length==5 and int(time[0:2])>=10 and int(time[0:2])<=12:
        print("{}:{} pm".format(time[0:2],time[-2:]))    
        
    elif end!="pm" and end!="am" and length==5 and int(time[0:2])>12:
        print("{}:{} pm".format(int(time[0:2])-12,time[-2:]))
    elif end!="pm" and end!="am" and length==5 and int(time[0:2])==0:
        print("{}:{} am".format("12",time[-2:]))

def main():
    print("***** Vehicle classifier *****")
    
    GVM=eval(input("What is the gross vehicle mass (in kg)?\n"))
    trailer=input("Does the vehicle have a trailer?\n")
    
    if trailer=="y":
        gvm=eval(input("What is gross vehicle mass of the trailer (in kg)?\n"))
        
        if gvm<=750 and GVM<=3500:
            print("This vehicle is class B.")
        elif gvm>750 and GVM<=3500:
            print("This vehicle is class EB.")
        elif gvm>750 and 16000>=GVM>3500:
            print("This vehicle is class EC1.")
        elif gvm>750 and GVM>16000:
            print("This vehicle is class EC.")
        elif gvm<=750 and 16000>=GVM>3500:
            print("This vehicle is class C1.")
        elif gvm<=750 and GVM>16000:
            print("This vehicle is class C.")
    elif trailer=="n":
        
        articulated=input("Is the vehicle articulated?\n")
        if articulated=="n" and articulated=="n" and GVM<=3500:
            print("This vehicle is class B.")
        elif articulated=="n" and articulated=="n" and 16000>=GVM>3500:
            print("This vehicle is class C1.")
        elif articulated=="n" and articulated=="n" and GVM>16000:
            print("This vehicle is class C.")         
        
        elif GVM<=3500 and articulated=="y":
            print("This vehicle is class EB.")
        elif 16000>=GVM>3500 and articulated=="y":
            print("This vehicle is class EC1.")
        elif GVM>16000 and articulated=="y":
            print("This vehicle is class EC.")
