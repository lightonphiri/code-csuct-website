import turtle

num=turtle.numinput("","Enter the number of petals (must be an even number):")
num=int(num)
stem=turtle.numinput("","Enter the length of stem:")
petal=turtle.textinput("","Enter the petal type (triangle, square, or pentagon):")
side=turtle.numinput("","Enter the length of side:")
colour1=turtle.textinput("","Enter the stem colour:")
colour2=turtle.textinput("","Enter the first petal colour:")
colour3=turtle.textinput("","Enter the second petal colour:")

r=0
turtle.speed(0)
for i in range(num):
    r+=360/num
    turtle.color(colour1)
    turtle.fd(stem)
    if petal =="triangle":
        if i% 2 ==0:
            turtle.color(colour2)
        else:
            turtle.color(colour3)
        turtle.left(30)
        turtle.fd(side)
        turtle.right(120)
        turtle.fd(side)
        turtle.right(120)
        turtle.fd(side)
        turtle.color(colour1)
        
        turtle.home()
        turtle.right(r)
        
        
    elif petal=="square":
        if i% 2 ==0:
            turtle.color(colour2)
        else:
            turtle.color(colour3)
            
        turtle.left(90)
        turtle.fd(stem/2)
        turtle.right(90)
        turtle.fd(stem)
        turtle.right(90)
        turtle.fd(stem)
        turtle.right(90)
        turtle.fd(stem)
        turtle.right(90)
        turtle.fd(stem/2)
        turtle.color(colour1)
        
        turtle.home()
        turtle.right(r)
        
    elif petal == "pentagon":
        if i% 2 ==0:               #colour code
            turtle.color(colour2)
        else:
            turtle.color(colour3)  #pentagon code
        turtle.left(45)
        turtle.fd(stem)
        turtle.right(72)
        turtle.fd(stem)
        turtle.right(72)
        turtle.fd(stem)
        turtle.right(72)
        turtle.fd(stem)
        turtle.right(72)
        turtle.fd(stem)
        turtle.color(colour1)
        
        turtle.home()
        turtle.right(r)
        
turtle.exitonclick()
