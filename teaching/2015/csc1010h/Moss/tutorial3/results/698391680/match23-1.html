def main():
    print('Enter the minimum number of inches (not less than 0):')
    min=eval(input())
    print('Enter the maximum number of inches (not greater than 11):')
    max=eval(input())
    
    print("Inches: ",end = '')
    for inches in range(min,max+1):


        print("{0:>4}".format(inches),end=' ')
    
    print()
    print("Metres: ",end='')
    for inches in range(min, max+1):    
        for metres in range(1):
            metres=round((inches/39.37),2)
            print(format(metres,'.2f'),end=' ')
