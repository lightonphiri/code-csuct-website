# convert one format of telling the time to the other
# lutho nduneni

def main():
    time=input("Enter a time ([h]h:mm [am|pm]):\n")
    if len(time)>=6:
        if time[-2]=="a":
            if time[0:2]!="12":
                a=time.find("a")
                
                print("0"+time[0:a-1])
            else:
                print("00:"+time[3:5])
        elif time[-2]=="p":
            if time[0:2]!="12":
                p=time.find("p")
                if p==6:
                    time1=int(time[0:2])+12
                    print(str(time1)+time[2:5])
                elif p==5:
                    time1=int(time[0:1])+12
                    print(str(time1)+time[1:5])            
            else:
                print(time[0:5])
    elif len(time)==5:
        if time[0:2].isnumeric():
            time1=int(time[0:2])-12
            if -12<time1<0:
                time1=abs(time1)
                time1=str(time1)
                print(time[1:2]+":"+time[3:5],"am")
            elif time1==-12:
                time1=abs(time1)
                print(str(time1)+":"+time[3:5],"am")
            elif time1>0:
                time1=str(time1)
                print(time1+":"+time[3:5],"pm")
            elif time1==0:
                print("12:"+time[3:5],"pm")

def main():
    print("***** Vehicle classifier *****")
    gross_mass=eval(input("What is the gross vehicle mass (in kg)?\n"))
    trailer_choice=input("Does the vehicle have a trailer?\n")
    if gross_mass<=3500:
        if trailer_choice=="y":
            trailer_mass=eval(input("What is gross vehicle mass of the trailer (in kg)?\n"))
            if trailer_mass>750:
                print("This vehicle is class EB.")
            else:
                print("This vehicle is class B.")
        else:
            articulation=input("Is the vehicle articulated?\n")
            if articulation=="y":
                print("This vehicle is class EB.")
            else:
                print("This vehicle is class B.")
