def main():
    import turtle
    petal_number=int(input("Enter the number of petals (must be an even number): "))
    stem_length=int(input("Enter the length of stem: "))
    petal_type=input("Enter the petal type (triangle, square, or pentagon): ")
    side_length=int(input("Enter the length of side: "))
    colour_stem=input("Enter the stem colour: ")
    first_petal_colour=input("Enter the first petal colour: ")
    second_petal_colour=input("Enter the second petal colour: ")
    turtle.speed(6)
    if petal_type=='pentagon':
        for i in range(petal_number):
            turtle.color(colour_stem)
            turtle.forward(stem_length)
            turtle.left(54)
            if i%2==0:
                turtle.color(second_petal_colour)
            else:
                turtle.color(first_petal_colour)


            turtle.forward(side_length)
            for i in range(4):
                turtle.right(72)
                turtle.forward(side_length)
            turtle.left(54)
            turtle.color(colour_stem)
            turtle.forward(stem_length)
            turtle.right(180)
            turtle.right(int(360/petal_number))
        turtle.exitonclick()
    elif petal_type=='triangle':
        for i in range(petal_number):
                turtle.color(colour_stem)
                turtle.forward(stem_length)
                turtle.left(30)
                if i%2==0:
                    turtle.color(second_petal_colour)
                else:
                    turtle.color(first_petal_colour)


                turtle.forward(side_length)
                for i in range(2):
                    turtle.right(120)
                    turtle.forward(side_length)
                turtle.left(30)
                turtle.color(colour_stem)
                turtle.forward(stem_length)
                turtle.right(180)
                turtle.right(int(360/petal_number))
        turtle.exitonclick()
    elif petal_type=='square':
        for i in range(petal_number):
            turtle.color(colour_stem)
            turtle.forward(stem_length)
            turtle.left(45)
            if i%2==0:
                turtle.color(second_petal_colour)
            else:
                turtle.color(first_petal_colour)


            turtle.forward(side_length)
            for i in range(3):
                turtle.right(90)
                turtle.forward(side_length)
            turtle.left(45)
            turtle.color(colour_stem)
            turtle.forward(stem_length)
            turtle.right(180)
            turtle.right(int(360/petal_number))
        turtle.exitonclick()

def main():
    minimum=int(input("Enter the minimum number of inches (not less than 0):\n"))
    maximum=int(input("Enter the maximum number of inches (not greater than 11):\n"))
    inches=0.0254
    print("Inches:",end="")
    for i in range(minimum,maximum+1):
        print("{0:>5}".format(i),end="")
    
    print('\nMetres:',end='')
    for b in range(minimum,maximum+1):
        print("{0:>5.2f}".format(round(b*inches,2)),end="")
main()

def main():
    print("***** Vehicle classifier *****")
    
    gvm=int(input("What is the gross vehicle mass (in kg)?\n"))
    trailer=input("Does the vehicle have a trailer?\n")
    if trailer.startswith('y'):
        trailer_gvm=int(input("What is gross vehicle mass of the trailer (in kg)?\n"))
        if gvm<=3500:
            if trailer_gvm<=750:
                print("This vehicle is class B.")
            else:
                print("This vehicle is class EB.")
        elif 3500<gvm<=16000:
            if trailer_gvm<=750:
                print("This vehicle is class C1.")
            else:
                print("This vehicle is class EC1.")
        else:
            if gvm>16000:
                if trailer_gvm<=750:
                    print("This vehicle is class C.")
                else:
                    print("This vehicle is class EC.")
            
    elif trailer=='n':
        articulated=input("Is the vehicle articulated?\n")
        if articulated.startswith('y'):
            if gvm<=3500:
                print("This vehicle is class EB.")
            elif 3500<gvm<=16000:
                print("This vehicle is class EC1.")
            else:
                print("This vehicle is class EC.")
        else:
            if articulated.startswith('n'):
                if gvm<=3500:
                    print("This vehicle is class B.")
                elif 3500<gvm<=16000:
                    print("This vehicle is class C1.")
                elif gvm>16000:
                    print("This vehicle is class C.")
