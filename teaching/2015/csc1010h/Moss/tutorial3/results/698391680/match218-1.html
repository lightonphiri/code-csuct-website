import turtle
def main():
    petal = turtle.numinput("Petals","Enter the number of petals (must be an even number):")
    petal = int(petal)
    stem = turtle.numinput("Stem length","Enter the stem length:")
    side = turtle.numinput("Side length","Enter the side length:")
    petal_type = turtle.textinput("Petal type"," Enter the type of petal:")
    stem_col  = turtle.textinput("Stem colour"," Enter the stem colour:")
    petal_col1 = turtle.textinput("1st petal colour","Enter the first petal colour:")
    petal_col2 = turtle.textinput("2nd petal colour","Enter the second petal colour:")
    

    x = 0
    for i in range(petal):
        x +=360/petal
    
        if petal_type =="triangle":
            
            turtle.color(stem_col)
            turtle.fd(stem)
            if i%2 == 0:
                turtle.color(petal_col1)
            else:
                turtle.color(petal_col2)
                
            turtle.left(30)   
            turtle.fd(side)
            turtle.right(120)
            turtle.fd(side)
            turtle.right(120)  
            turtle.fd(side)
            turtle.color(stem_col)
        
            turtle.home()
            turtle.right(x)
            
        elif petal_type =="pentagon":
            turtle.color(stem_col)
            turtle.fd(stem)
            if i%2 ==0:
                turtle.color(petal_col1)
            else:
                turtle.color(petal_col2)
            turtle.left(72)
            turtle.fd(side)

            turtle.fd(side)

            turtle.fd(side)
            turtle.right(72)
            turtle.fd(side)
            turtle.color(stem_col)
            turtle.home()
            turtle.right(x)        
        
        elif petal_type=="square":
            
            turtle.color(stem_col)
            turtle.fd(stem)            
            
            if i%2 == 0:
                turtle.color(petal_col1)

                turtle.color(petal_col2)
