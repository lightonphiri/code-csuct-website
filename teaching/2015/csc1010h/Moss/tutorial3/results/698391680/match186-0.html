def main():
    minimum = int(input("Enter the minimum number of feet (not less than 0):\n"))
    maximum = int(input("Enter the maximum number of feet (not more than 99):\n"))
    for ft in range(minimum,maximum+1):
        metre = ft/3.2808
        metre_str = "{0:>.2f}".format(metre) # could be done away with and intergrated into print statement
        print("{0:>4}' |  {1:>5}m".format(ft,metre_str))

    # square

        for i in range(petal_number):                         # petal 
            if i%2 == 0:

def main():
    time = input("Enter a time ([h]h:mm [am|pm]):\n")
    suffix = "m"  
    if time.endswith(suffix):
        length = len(time)
        if length == 7:
            time1 = time[0]
            time2 = time[2:4]
            meridian = time[5:7]
            # am converter
            if meridian == "am":
                print("0" + time1 + ":" + time2) 
            elif meridian == "pm":
                time1 = str(int(time1) + 12)
                print(time1 + ":" + time2)
        elif length == 8:
            time1 = time[0:2]
            time2 = time[3:5]
            meridian = time[6:8]  
            # am converter at 12
            if meridian == "am" and time1 == "12":
                time1 = "00"
                print(time1 + ":" + time2)
            # pm converter at 12
            elif meridian == "pm" and time1 == "12":
                print(time1 + ":" + time2)
            # am not 12
            elif meridian == "am" and time1 != "12":
                print(time1 + ":" + time2)
            # pm not 12
            elif meridian == "pm" and time1 != "12":
                time1 = str(int(time1) + 12)
                print(time1 + ":" + time2)          
    
    
    else: # it is 24 hour
        time1 = time[0:2]
        time2 = time[3:5]
        
        if time1 > "12":
            time1 = int(time1) - 12
            print("{0:<}".format(time1) + ":" + time2 + " pm")               
        elif time1 < "12" and time1 != "00":
            print(time1[1] + ":" + time2 + " am")
        elif time1 == "00":
            time1 = "12"
            print(time1 + ":" +time2 + " am")
        elif time1 == "12":
            time = "12"
            print(time1 + ":" + time2 + " pm")                     
main()
