"""Program that calculates the minimum amount of coins in change in R5, R2,
and R1 coins, given a total change amount in rand"""



def main():                                               #define main() function
    change_amount= eval(input("Enter change amount:"))    #Prompt user to enter change value
    fiveRAND= (change_amount)//5                          #Calculate minimum amount of coins of change in R5 coins
    two_rand= (change_amount-fiveRAND*5)//2               #Calculate minimum amount of coins of change in R2 coins
    rand1= ((change_amount)-(fiveRAND*5)-(two_rand*2))//1 #Calculate minimum amount of coins of change in R1 coins 
    
    print("The change:\n",fiveRAND,"R5 coins\n",two_rand,"R2 coins\n",rand1,"R1 coins") #Print the output
main()                                                    #Call main() function
