# Change calculator




def main():
    amount = int(input('Enter change amount:\n'))
    print('The change:')
    coins = amount//5
    amount = amount%5
    print('{0} R5 coins'.format(coins))
    coins = amount//2
    amount = amount%2
    print('{0} R2 coins'.format(coins))
    print('{0} R1 coins'.format(amount))
    
main()
