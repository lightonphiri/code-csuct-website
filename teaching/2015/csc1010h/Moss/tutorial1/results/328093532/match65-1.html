def main():
    
    change = eval(input("Enter change amount:\n",))  #prompt the user to enter a change number
    rand_5 = change // 5          #finding no of R5 coins
    remainder_1 = change % 5      #fidning the remainder after the 5 coins is taken out 
    rand_2 = remainder_1 // 2     #finding the no of R2 coins from the frist reminder
    remainder_2 = (change % 5) % 2  #Finding the reminder after the 2 coins are taken out
    rand_1 = remainder_2 // 1     #finding the no of 1 rand coins
    
    print("The change:")           
    print(rand_5, "R5 coins")
    print(rand_2, "R2 coins")
    print(rand_1, "R1 coins")
main()
