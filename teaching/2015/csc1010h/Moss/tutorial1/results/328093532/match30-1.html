def main():
    change = eval(input("Enter change amount:\n")) # entering the change value
    print("The change:")


    n = change//5     # is the change divisible by 5 
    print(n, 'R5 coins') # if so print the whole number 
    b = (change - (n*5)) //2  # is the change divisible by 2
    print(b, "R2 coins") # if so print the whole number
    c = (change - (n*5) - (b*2))//1 # is the change minus the 5 and the 2 still divisible by 1
    print (c, "R1 coins") # if so print that whole number
main()
