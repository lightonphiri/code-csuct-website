# take the given amount to be first be divided by a 5.
# the remainder will then be divided by 2
# the reamainder will be the 1 rands obtained.




rand = int(input("Enter change amount:\n"))
five_rand = rand//5             #divide by 5, to see the number of 5 rand coins we can obtain
two_rand = (rand%5)//2          #divide by 2, from the remainder of the factor divided by 5 to see the 2 rand we ca obtain
one_rand = int((rand%5)%2)//1
# the remaining number from dividing by 5 and then by 2, will be the number of 1 rand coins

print("The change:")
print(five_rand,"R5 coins")
print(two_rand,"R2 coins")
print(one_rand,"R1 coins")
