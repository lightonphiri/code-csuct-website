print ("Enter change amount:")


val = eval (input(""))

# calculate how many R5 coins there are
if val >= 5:
    FiveRand = val // 5
    remainder = val % 5
    
    # calculate how many R2 coins there are
    if remainder >=2:
        TwoRand = remainder // 2
        leftovers = remainder % 2
        
        # calculate how many R1 coins there are
        if leftovers >=1:
            OneRand = leftovers
            
        else:
            OneRand = 0
        
    else:
        TwoRand = 0
        OneRand = remainder
        
else:
    FiveRand = 0
    TwoRand = val // 2
    change = val % 2
    
    if change >= 1:
        OneRand = change
        
    else:
        OneRand = 0
    
print ("The change:")
