# calculate the simplest change for money
# lutho nduneni
# 14 april 2015

def main():


    amount_to_be_changed=eval(input("Enter change amount:\n"))  # Enter the amount that has to be changed
    # Check how many R5 coins can be found in the amount
    R5_change=amount_to_be_changed//5
    # Check how much cannot be changed to R5 coins
    remaining=amount_to_be_changed%5
    # Check if amount left can be changed to R2 coins
    R2_change=remaining//2
    # Change the remaining amount to R1 coins
    R1_change=remaining%2
    # Print the results
    print("The change:")
    print(R5_change,"R5 coins")
    print(R2_change,"R2 coins")
    print(R1_change,"R1 coins")
main()
