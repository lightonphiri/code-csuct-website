#programPython program that calculates the minimum amount of coins in change in R5, R2,                                           and R1 coins, given a total change amount in rand.
#phuti mokoti
#19 april 2015

def main():
    t=eval(input("Enter the total change amount: \n")) # get input from user
    if t>0:


        five=t//5                                 #divide total change by 5 to get number of R5 coins
        two=(t%5)//2                              #divide remainder of totalchange divided by 5 to get number of R2 coins
        one=((t%5)%2)                             # the remainder after number of R2 coins is the number of R1 coins
    
    
    print("The change: ")    #print output
    print(five, "R5 coins")
    print(two, "R2 coins")
    print(one, "R1 coins") 
main()
