# get input from user


amount = (int(input('Enter change amount:\n')))

# def main function
def main():
    R5_coins = amount//5
    R2_coins = (amount%5)//2
    R1_coins = (amount%5)%2
    print('The change:\n',R5_coins,' R5 coins\n',R2_coins,' R2 coins\n',R1_coins,' R1 coins',end = "",sep = "")

main()
