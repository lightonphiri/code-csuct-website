# calculate minimum amount of coins in rands for an amount

change_amount=int(input('Enter change amount:\n'))#Ask user to enter number(amount in rands)

def main():   #create main function


    i=change_amount//5  # find maximum amount of times it is divisible by 5.
    k=(change_amount%5)//2  # calculate how many times the remainder of step 3 is divisible by 2.
    l=((change_amount%5)%2)//1 # calculate how many times the remainder of step 4 is divisible by 1.
    print('The change:')
    print(i,'R5 coins')
    print(k,'R2 coins')
    print(l,'R1 coins') # display the output

main() #call the function
