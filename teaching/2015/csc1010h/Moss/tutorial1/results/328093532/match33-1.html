# a programe to calculate change in three types of coins,R5,R2 and R1.
# Siphetho Velaphi[VLPSIP001]
# 17 April 2015
change=int(input("Enter change amount:\n"))

def main():


    R5=change//5
    R2=((change)-R5*5)//2
    R1= ((change)-R5*5-R2*2)//1
    print("The change:\n", R5,"R5 coins\n",R2,"R2 coins\n",R1,"R1 coins")
main()
