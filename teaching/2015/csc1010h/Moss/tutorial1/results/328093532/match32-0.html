change_amount=int(input('Enter change amount:\n'))
#calculation for the R5 coin
R5_coins=change_amount//5
amount_left=change_amount-R5_coins*5
#calculation for the R2 coin
R2_coins=amount_left//2
amount_left1=amount_left-R2_coins*2
#calculation for the R1 coin
R1_coins=amount_left1//1
#display the output 
print('The change:')
print(R5_coins,'R5 coins')
print(R2_coins,'R2 coins')
print(R1_coins,'R1 coins')
