    # display the answer
    print("The change:")


    print(num_R5coins,"R5 coins")
    # calculate minimum amount of coins in change in R2 coins
    change_amount2=(change_amount)%5

    # dispay the answer
    print(num_R2coins,"R2 coins")

    change_amount3=(change_amount2)%2
