value = int(input("Enter change amount: \n"))
print ("The change:")
         
r5 = value//5
print(r5,"R5 coins")
    
amount = value%5 #amount remaining from value

r2 = amount//2
print (r2,"R2 coins")

amount2 = value-(r5*5+r2*2)
print (amount2, "R1 coins")
