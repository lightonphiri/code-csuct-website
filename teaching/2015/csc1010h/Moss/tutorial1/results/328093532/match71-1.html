def main():
    c = int(input("Enter change amount:\n")) #Ask the user to enter the change amount
#read the amount
    R5_change = c//5 #.calculate the number of R5 coins
    R2_change = (c%5)//2 #.calculate the number of R2 coins
    R1_change = (c%5)%2 #.calculate the number of R1 coins 
    print("The change:")
    print(R5_change, "R5 coins")
    print(R2_change, "R2 coins")
    print(R1_change, "R1 coins")  #.display the answer
    
main()
