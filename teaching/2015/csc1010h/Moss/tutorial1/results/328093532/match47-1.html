# Fucntion definition
def main():
    # Input
    cng = int(input("Enter change amount:\n"))
    
    # Process    
    cng_in_5s =  cng//5


    cng_in_2s = (cng - (cng_in_5s * 5))//2 
    cng_in_1s =  cng - (cng_in_5s * 5 + cng_in_2s * 2)
    
    # Output
    print("The change:")
    print(cng_in_5s,"R5 coins")
    print(cng_in_2s,"R2 coins")
    print(cng_in_1s,"R1 coins")
main()
