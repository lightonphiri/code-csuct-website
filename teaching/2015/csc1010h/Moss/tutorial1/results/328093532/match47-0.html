#Program to calculate the minimum change 

amount = int(input("Enter change amount:\n"))
def main():
    amountOfR5= amount//5
    if amountOfR5*5==amount:
        amountOfR2=0
        amountOfR1=0       

    else:


        amountOfR2= (amount-(amountOfR5*5))//2
        amountOfR1= (amount-(amountOfR5*5+amountOfR2*2))
    
    print("The change:")
    print(amountOfR5,"R5 coins")
    print(amountOfR2,"R2 coins")
    print(amountOfR1,"R1 coins")

main()
