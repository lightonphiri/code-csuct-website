def main():
     # get the user to enter the change amount
     change_amount = int(input("Enter the change amount: "))
     # calculate how many R5 coins
     a = change_amount//5
     # calculate how many R2 coins


     b = change_amount%5


     c = b//2
     # calculate how many R1 coins 
     d = b%2
     # display the answer 
     print("\n","The change:","\n",a,"R5 coins","\n",c,"R2 coins","\n",d,"R1 coins")

main()
