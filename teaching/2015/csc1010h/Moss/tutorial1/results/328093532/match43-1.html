#Program that calculates the minimum amount of coins in change R5, R2 and R1 coins, given a total change amount in rand.
#Name: Vusi Skosana
#student number: SKSVUS001



def main():
    
    n=eval(input("Enter change amount:"))  #prompts the user to enter a number

    # formula to calculate change in rands:

    coin_R5=(n//5)            #calculates the number of R5 coins
    coin_R2=(n%5)//2          #calculates the number of R2 coins
    coin_R1=((n%5)%2)//1      #calculates the number of R1 coins
    
    #displays the results:
    
    print("The change:")
    print(coin_R5, "R5 coins")
    print(coin_R2, "R2 coins")
    print(coin_R1, "R1 coins")
    
main()
