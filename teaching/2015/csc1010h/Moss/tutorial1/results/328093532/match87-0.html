x = int(input("Enter change amount:\n")) #prompts the user to enter a change amount

def main ():
    print ("The change:")  


    five = x//5             #calculates the number of R5 coins in change required by dividing the amount by 5
    print (five,"R5 coins")
    two = (x%5)         #calculates the remainder amount if change is divided by 5 
    two_1 = two//2      #calculates the number of R2 coins required in change
    print (two_1, "R2 coins")
    one = (two%2)//1    #calculates the number of R1 coins needed in change
    print (one, "R1 coins")
    
    
main ()
