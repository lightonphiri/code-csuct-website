def process_line(line,line_number):


    parts = line.split(' ')
    name = parts[0]
    trials = (parts[1])
    average =(parts[2])
    try:
        trials = int(parts[1])
    except ValueError:
        print("check line"+str(line_number))
        print(line)
        print("Please enter an integer value. See that input values follow the following format (name,trials and average)\n Do not enter more than 1 space between input values")
    try:
        average = float(parts[2])
    except ValueError:
        print("check line"+str(line_number))
        print(line)
        print("Please enter an floating value. See that input values follow the following format (name,trials and average)\n Do not enter more than 1 space between input values")   
    return (name, trials, average)

def get_max(entries):
    max = entries[0]
    for i in range(1, len(entries)):
        if entries[i][2] <max[2]:
            max = entries[i]
            
    entries.remove(max)
    return max

def sort(entries):
    sorted_entries = []
    while len(entries)>0:
        sorted_entries.append(get_max(entries))
    return sorted_entries

def output(entries):
    print('Subject        | Average Time')
    print('************** | ************')
    for entry in entries:
        print('{:14} | {:.6}'.format(entry[0], entry[2]))

import time
import random

def flush_input():
    try:
        import msvcrt
        while msvcrt.kbhit():
            msvcrt.getch()
    except ImportError:
        import sys, termios
        termios.tcflush(sys.stdin, termios.TCIOFLUSH)
        


def results_file():
    x = open('results.txt', 'w')
    print("Subject        | Average Time", file = x)
    print("************** | ************", file = x)
    x.close()
    
def main():
    results_file()
    output_file = str(input("Enter the name of the output file:\n"))
    f = open(output_file.lower(), 'a')
    d={}
    e=[]
    while True:
        subject_name = str(input("Enter subject name (or press 'enter' to quit):\n"))
        if subject_name == '': break
        else:
            total = 0
            tests = int(input("Enter the number of tests:\n"))
            for i in range(tests):
                value = random.randrange(1, 6, 1)   # chooses random number from 1 to 5 for pausing time
                print("Get ready...", flush = True)
                time.sleep(value)
                print("Press 'enter' NOW!", end = '')
                start_time = time.time()
                flush_input()
                input()
                end_time = time.time()
                reaction_time = round(end_time - start_time, 6)
                print("Reaction time:", reaction_time)
                total += reaction_time
            average = round(total / tests, 6)
            d[average] = subject_name
            e.append(average)
            print("Average reaction time:", average)
            print()
    e.sort()
    for av in e:
        print("{0:<14} | {1:<12}".format(d[av], av), file = f)   # prints to results file
    f.close()
        
main()
