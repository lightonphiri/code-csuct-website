def linear_search(val,lst):
    index = 0
    found_index = None
    while index < len(lst) and found_index == None:
        if val == lst[index]: # value found
            found_index = index
            break
        index += 1
    return found_index

def split_line(s):
    r = []
    s = s.split()
    for i in s:
        k = i.strip('?!.,;:)([]{}\'" \n')
        r.append(k)
    return r

def process_line(line):
    parts = line.split(' ')
    name = parts[0]
    trials = int(parts[1])
    average = float(parts[2])
    return (name, trials, average)

def get_max(entries):
    max = entries[0]
    for i in range(1, len(entries)):
        if entries[i][2] <max[2]:
            max = entries[i]
            
    entries.remove(max)
    return max

def sort(entries):
    sorted_entries = []
    while len(entries)>0:
        sorted_entries.append(get_max(entries))
    return sorted_entries

def output(entries):

            break
        number_of_tests=int(input("Enter the number of tests:"))

        total_time=0
