def process_line(line):
    try:
        parts = line.split(' ')
        name = parts[0]
        trials = int(parts[1])
        average = float(parts[2])
        return (name, trials, average)
    except ValueError as VE :
        print("In line number {} \n {} Must be a digit \n \n {} ".format(line,parts[1],VE))
    except TypeError as TE :
        print("incorrect type used in line" , TE )
        
def get_max(entries):
    max = entries[0]
    for i in range(1, len(entries)):
        if entries[i][2] <max[2]:
            max = entries[i]
            
    entries.remove(max)
    return max

def sort(entries):
    sorted_entries = []
    while len(entries)>0:
        sorted_entries.append(get_max(entries))
    return sorted_entries
