    filea=open(filea,"a")
    average=0
    for i in range(tests):
        print("Get ready...",flush=True)
        filea.write("Get ready...\n")
        time.sleep(random.randint(1,5))
        start=time.time()
        d=input("Press 'enter' NOW!")
        end=time.time()
        reaction_time=end-start
        average+=reaction_time
        print("Reaction time:","{0:.6f}".format(reaction_time))
        filea.write("Reaction time:"+str("{0:.6f}".format(reaction_time))+"\n")
    print("Tests complete.")
    filea.write("Tests complete."+"\n")
    print("Average reaction time:"+str("{0:.6f}".format(average/tests)))
