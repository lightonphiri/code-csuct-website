def process_line(line):
    parts = line.split(' ')
    name = parts[0]
    trials = int(parts[1])
    average = float(parts[2])
    return (name, trials, average)

def get_max(entries):
    max = entries[0]
    for i in range(1, len(entries)):
        if entries[i][2] <max[2]:
            max = entries[i]
            
    entries.remove(max)
    return max

def sort(entries):
    sorted_entries = []
    while len(entries)>0:
        sorted_entries.append(get_max(entries))
    return sorted_entries

def output(entries):


    print('Subject        | Average Time')
    print('************** | ************')

        print('{:14} | {:.6}'.format(entry[0], entry[2]))

import time
import random

def flush_input():
    try:
        import msvcrt


        while msvcrt.kbhit():
            msvcrt.getch()
    except ImportError:
        import sys, termios
        termios.tcflush(sys.stdin, termios.TCIOFLUSH)

def main():
    output= input('Enter the name of the output file:\n')
    f= open(output, 'a')
    subject=input('Enter subject name (or press \'enter\' to quit):\n')  
    num=int(input('Enter the number of tests:\n'))
    total=0
   
    while True:
        f= open(output, 'a')
        for i in range(num):
            print('Get ready...', flush=True)
            time.sleep(random.randint(1,5))
            print('Press \'enter\' NOW!', end='')
            flush_input()
            start=time.time()
            reaction=input()
            if reaction=='':
                end=time.time()
            seconds=end-start
            total+=seconds
            print('Reaction time: {0:2.6f}'.format(seconds))
        print('Tests complete.')
        average=total/num
        print('Average reaction time: {0:2.6f}'.format(average))
        print('{0} {1} {2:2.6f}'.format(subject,num,average), file=f)
        print()
        subject=input('Enter subject name (or press \'enter\' to quit):\n')
        if subject =='': break
        num=int(input('Enter the number of tests:\n'))
    f.close()

main()
