def process_line(line):
    parts = line.split(' ')
    name = parts[0]
    trials = int(parts[1])
    avrg = float(parts[2])
    return (name, trials, avrg)

def get_maximum(entries):
    maximum = entries[0]
    for i in range(1, len(entries)):
        if entries[i][2] <maximum[2]:
            maximum = entries[i]
            
    entries.remove(maximum)
    return maximum

def sort(entries):
    sorted_entries = []
    while len(entries)>0:
        sorted_entries.append(get_maximum(entries))
    return sorted_entries

def output(entries):
    print('Subject        | Average Time')
    print('************** | ************')
    for entry in entries:
        print('{:14} | {:.6}'.format(entry[0], entry[2]))
