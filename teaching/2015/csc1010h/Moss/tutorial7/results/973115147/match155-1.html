def process_line(line):
    words = line.split(' ') #split line into individual words
    name = words[0]
    trials = int(words[1])
    average = float(words[2])
    return (name, trials, average)

def get_max(entries):
    max_entry = entries[0]
    for i in range(1, len(entries)):
        if entries[i][2] <max_entry[2]:
            max_entry = entries[i]
            
    entries.remove(max_entry)
    return max_entry

def sort(entries):
    sorted_entries = []
    while len(entries)>0:
        sorted_entries.append(get_max(entries))
    return sorted_entries
