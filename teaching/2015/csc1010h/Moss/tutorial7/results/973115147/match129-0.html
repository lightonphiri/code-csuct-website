import time
from random import randint

output = input("Enter the name of the output file:\n")
f = open(output,'a')
name = input("Enter subject name (or press 'enter' to quit):\n")

while name !="" :
    num = int(input("Enter the number of tests:\n"))
    add = 0
    for z in range (num) :
        rand = randint(1,5)
        
        print("Get ready...")
        
        time.sleep(rand)
        now = time.clock() 
        rec = input("Press 'enter' NOW!")
       
        if (rec==""):
            after = time.clock()
            recT = after-now
            print('{} {:.6}'.format("Reaction time:",recT))
            add += recT
            
    average = add/num
    print("Tests complete.")
    print('{} {:.6}'.format('Average reaction time:',average),"\n")
    print('{} {} {:.6}'.format(name,num,average),file = f)
    
    name = input("Enter subject name (or press 'enter' to quit):\n")

f.close()
