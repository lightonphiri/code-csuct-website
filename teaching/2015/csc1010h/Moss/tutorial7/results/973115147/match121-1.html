# A program that calculates a particular subject's reaction time

import random
import time


file_name = input('Enter the name of the output file:\n')
file_name = open(file_name,'w')

while True:  
    
    subject_name = input("Enter subject name (or press 'enter' to quit):\n")
    
    if subject_name == "":
        break
    
    number_of_tests = int(input('Enter the number of tests:\n'))
    
    for i in range(number_of_tests):
        
        total = 0
        
        print('Get ready...')
        
        time.sleep(random.randint(1,5))
        
        start_time = time.time()        # starts the time
        
        n = input("Press 'enter' NOW!") # gets the user press enter 
        
        end_time = time.time()
        
        reaction_time = end_time-start_time  # time elasped 
        
        print('Reaction time: {0:1.6f}'.format(reaction_time))
        
        total += reaction_time # total reaction time
        
        average = total/number_of_tests  # calculates tghe average reaaction time 
    

    print(subject_name,number_of_tests,round(average,6),file=file_name) # prints results to a file 
    print('Test Complete.')
    print('Average reaction time {0:1.6f}'.format(average))

file_name.close()
