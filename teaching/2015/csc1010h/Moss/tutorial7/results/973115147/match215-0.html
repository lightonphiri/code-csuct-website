import time
import random

def main():
    print("Enter the name of the output file:")
    file = input()
    candidates = []
    while True:
        print("Enter subject name (or press 'enter' to quit):")
        name = input()
        
        if name == (""):
            break
        
        print("Enter the number of tests:")
        num_tests = int(input())
        
        times = 0
        
        for i in range(num_tests):
            print("Get ready...", flush = True)
            random_int = random.randint(1,5)
            time.sleep(random_int)
            print("Press 'enter' NOW!", end=(""))
            start_time = time.time()
            input()
            end_time = time.time()
            reaction_time = end_time - start_time
            reaction_time = round(reaction_time,6)
            print("Reaction time:", reaction_time)
            times = times + reaction_time
            
        print("Tests complete.")
        average_time = times / num_tests
        average_time = round(average_time,6)
        print("Average reaction time", average_time, "\n")
        
        details = str(name) + " " + str(num_tests) + " " + str(average_time)
        candidates.append(details)
        
    f = open(file, 'w')
    for person in candidates:    
        print(person, file = f)
        
    f.close()
    
main()
