def process_line(line):
    parts = line.split(' ')
    name = parts[0]
    trials = int(parts[1])
    average = float(parts[2])
    return (name, trials, average)

def get_max(entries):
    max = entries[0]
    for i in range(1, len(entries)):
        if entries[i][2] <max[2]:
            max = entries[i]
            
    entries.remove(max)
    return max

def sort(entries):
    sorted_entries = []
    while len(entries)>0:
        sorted_entries.append(get_max(entries))
    return sorted_entries

def output(entries):
    print('Subject        | Average Time')
    print('************** | ************')
    for entry in entries:
        print('{:14} | {:.6}'.format(entry[0], entry[2]))
              
def main():
   
    while True:
        file_name = input('Enter the name of the file:\n')
        try:
            file_input = open(file_name, 'r')

                n=1

import random
import time


file_name = input('Enter the name of the output file:\n')
file_name = open(file_name,'w')

while True:  
    
    subject_name = input("Enter subject name (or press 'enter' to quit):\n")
    
    if subject_name == "":
        break
    
    number_of_tests = int(input('Enter the number of tests:\n'))
    
    for i in range(number_of_tests):
        
        total = 0
        
        print('Get ready...')
        
        time.sleep(random.randint(1,5))
        
        start_time = time.time()
        
        n = input("Press 'enter' NOW!")
        
        end_time = time.time()
        
        reaction_time = end_time-start_time
        
        print('Reaction time: {0:1.6f}'.format(reaction_time))
        
        total += reaction_time
        
        average = total/number_of_tests

    print(subject_name,number_of_tests,round(average,6),file=file_name)
    print('Test Complete.')
    print('Average reaction time {0:1.6f}'.format(average))

file_name.close()
