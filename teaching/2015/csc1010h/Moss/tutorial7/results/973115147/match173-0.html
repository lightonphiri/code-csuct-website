def process_line(line):
    parts = line.split(' ')
    name = parts[0]
    trials = int(parts[1])
    average = float(parts[2])
    return (name, trials, average)

def get_max(entries):
    max = entries[0]
    for i in range(1, len(entries)):
        if entries[i][2] <max[2]:
            max = entries[i]
            
    entries.remove(max)
    return max

def sort(entries):
    sorted_entries = []
    while len(entries)>0:
        sorted_entries.append(get_max(entries))
    return sorted_entries

def output(entries):


    print('Subject        | Average Time')
    print('************** | ************')

# Program to calculate average reaction time

from random import *
from time import*

def flush_input():
    try:
        import msvcrt


        while msvcrt.kbhit():
            msvcrt.getch()
    except ImportError:
        import sys, termios
        termios.tcflush(sys.stdin, termios.TCIOFLUSH)

def main():   
    file=input("Enter the name of the output file:\n")
    f=open(file,"w")
    f.write('Subject        | Average Time\n')
    f.write('************** | ************\n')
    f.close()
    while True:
        subject_name=input("Enter subject name (or press 'enter' to quit):\n")
        if subject_name=='':
            break
        number_tests=int(input("Enter the number of tests:\n"))
        sum_time=0
        for i in range(number_tests):
            print("Get ready...",flush=True)
            sleep(randint(1,5))
            print("Press 'enter' NOW!",end='')
            flush_input()
            start_time=time()
            response=input()
            if response=='':
                end_time=time()
                reaction_time=float('{:.6}'.format(end_time-start_time))
                sum_time+=(reaction_time)
                print("Reaction time:",reaction_time)
        average=float('{:.6}'.format(sum_time/number_tests))
        print("Tests complete.")
        print("Average reaction time:",average,'\n')
        f=open(file,"a")
        f.write('{:14} | {:.6}\n'.format(subject_name, average))
        f.close()
main()
